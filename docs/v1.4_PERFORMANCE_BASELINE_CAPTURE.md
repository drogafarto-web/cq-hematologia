### 3b. Extract Chunk Sizes

```bash
# Create script: scripts/analyze-bundle-size.mjs

#!/usr/bin/env node

import * as fs from 'fs';
import * as path from 'path';
import { createGzip } from 'zlib';
import { promisify } from 'util';

const gzip = promisify(createGzip());
const readFile = promisify(fs.readFile);

const distDir = './dist/assets';
const chunks = fs.readdirSync(distDir).filter(f => f.endsWith('.js'));

(async () => {
  const sizes = {};
  
  for (const chunk of chunks) {
    const filePath = path.join(distDir, chunk);
    const content = await readFile(filePath);
    const compressed = await new Promise((resolve, reject) => {
      const gzipStream = createGzip();
      let output = Buffer.alloc(0);
      gzipStream.on('data', (chunk) => {
        output = Buffer.concat([output, chunk]);
      });
      gzipStream.on('end', () => resolve(output));
      gzipStream.on('error', reject);
      gzipStream.write(content);
      gzipStream.end();
    });
    
    sizes[chunk] = {
      raw_bytes: content.length,
      raw_kb: (content.length / 1024).toFixed(2),
      gzip_bytes: compressed.length,
      gzip_kb: (compressed.length / 1024).toFixed(2),
      compression_ratio: (compressed.length / content.length * 100).toFixed(1) + '%'
    };
  }
  
  console.log(JSON.stringify(sizes, null, 2));
})();

# Run
node scripts/analyze-bundle-size.mjs > bundle-sizes-2026-05-07.json
```

### 3b. PowerShell Alternative (Windows)

```powershell
# scripts/analyze-bundle-size.ps1

$distDir = ".\dist\assets"
$chunks = Get-ChildItem $distDir -Filter "*.js" -File

$sizes = @{}

foreach ($chunk in $chunks) {
    $filePath = $chunk.FullName
    $rawSize = (Get-Item $filePath).Length
    
    # Gzip compression via .NET
    $inputStream = [System.IO.File]::OpenRead($filePath)
    $outputStream = [System.IO.MemoryStream]::new()
    $gzipStream = [System.IO.Compression.GZipStream]::new($outputStream, [System.IO.Compression.CompressionMode]::Compress)
    $inputStream.CopyTo($gzipStream)
    $gzipStream.Close()
    $inputStream.Close()
    $gzipSize = $outputStream.Length
    
    $sizes[$chunk.Name] = @{
        raw_bytes = $rawSize
        raw_kb = [math]::Round($rawSize / 1024, 2)
        gzip_bytes = $gzipSize
        gzip_kb = [math]::Round($gzipSize / 1024, 2)
        compression_ratio = [math]::Round(($gzipSize / $rawSize * 100), 1).ToString() + "%"
    }
}

$sizes | ConvertTo-Json | Out-File "bundle-sizes-2026-05-07.json"
```

### 4b. Extract Latency Percentiles

```bash
# Script: scripts/parse-firestore-latency.mjs

#!/usr/bin/env node

import * as fs from 'fs';

const logs = JSON.parse(fs.readFileSync('firestore-logs-24h.json', 'utf-8'));

const latencies = logs
  .filter(log => log.jsonPayload?.latencyUs)
  .map(log => log.jsonPayload.latencyUs / 1000) // Convert μs → ms
  .sort((a, b) => a - b);

if (latencies.length === 0) {
  console.error('No Firestore query logs found. Check filter and time range.');
  process.exit(1);
}

const p50 = latencies[Math.floor(latencies.length * 0.50)];
const p95 = latencies[Math.floor(latencies.length * 0.95)];
const p99 = latencies[Math.floor(latencies.length * 0.99)];
const mean = latencies.reduce((a, b) => a + b, 0) / latencies.length;
const max = Math.max(...latencies);

console.log(JSON.stringify({
  capturedAt: new Date().toISOString(),
  sampleSize: latencies.length,
  metrics: {
    p50_ms: p50.toFixed(2),
    p95_ms: p95.toFixed(2),
    p99_ms: p99.toFixed(2),
    mean_ms: mean.toFixed(2),
    max_ms: max.toFixed(2)
  },
  distribution: {
    under_50ms: latencies.filter(l => l < 50).length,
    under_100ms: latencies.filter(l => l < 100).length,
    under_500ms: latencies.filter(l => l < 500).length,
    over_500ms: latencies.filter(l => l >= 500).length
  }
}, null, 2));
```

### 5b. Manual Timing Test

```bash
#!/bin/bash
# scripts/test-pdf-export-latency.sh

ITERATIONS=10
TOTAL_TIME=0

for i in {1..ITERATIONS}; do
  START=$(date +%s%3N)
  
  # Trigger PDF export via curl (requires auth token)
  curl -X POST "https://southamerica-east1-hmatologia2.cloudfunctions.net/generatePdfExport" \
    -H "Authorization: Bearer $FIREBASE_TOKEN" \
    -H "Content-Type: application/json" \
    -d '{
      "labId": "test-lab-001",
      "runIds": ["run-1", "run-2", ..., "run-50"],
      "format": "pdf"
    }' \
    -o /dev/null
  
  END=$(date +%s%3N)
  ELAPSED=$((END - START))
  TOTAL_TIME=$((TOTAL_TIME + ELAPSED))
  
  echo "Run $i: ${ELAPSED}ms"
done

AVERAGE=$((TOTAL_TIME / ITERATIONS))
echo "Average TTI (PDF export, 50 pages): ${AVERAGE}ms"
```

### 5c. PowerShell Version

```powershell
# scripts/test-pdf-export-latency.ps1

$iterations = 10
$totalTime = 0
$results = @()

for ($i = 1; $i -le $iterations; $i++) {
    $start = [System.Diagnostics.Stopwatch]::StartNew()
    
    $response = Invoke-WebRequest `
        -Uri "https://southamerica-east1-hmatologia2.cloudfunctions.net/generatePdfExport" `
        -Method Post `
        -Headers @{
            "Authorization" = "Bearer $env:FIREBASE_TOKEN"
            "Content-Type" = "application/json"
        } `
        -Body (@{
            "labId" = "test-lab-001"
            "runIds" = @("run-1", "run-2", "run-3")
            "format" = "pdf"
        } | ConvertTo-Json) `
        -OutFile $null `
        -ErrorAction SilentlyContinue
    
    $start.Stop()
    $elapsed = $start.ElapsedMilliseconds
    $totalTime += $elapsed
    
    $results += @{
        run = $i
        elapsed_ms = $elapsed
    }
    
    Write-Host "Run $i`: ${elapsed}ms"
}

$average = [math]::Round($totalTime / $iterations)
Write-Host "Average TTI (PDF export, 50 pages): ${average}ms"

# Export results
$results | ConvertTo-Json | Out-File "pdf-export-latency-2026-05-07.json"
```

## Troubleshooting

### Lighthouse CI Not Set Up

**Symptom:** `npx @lhci/cli` command not found or fails.

**Solution:**

1. **Install LHCI globally:**

```bash
npm install -g @lhci/cli@0.14.x
lhci --version  # Should output 0.14.x
```

2. **Create `.lighthouserc.json` in project root:**

```json
{
  "ci": {
    "collect": {
      "staticDistDir": "./dist",
      "url": [
        "http://localhost:3000/hub",
        "http://localhost:3000/features/bioquimica",
        "http://localhost:3000/features/analytics",
        "http://localhost:3000/features/export",
        "http://localhost:3000/portal"
      ],
      "numberOfRuns": 3
    },
    "upload": {
      "target": "temporary-public-storage"
    },
    "assert": {
      "preset": "lighthouse:recommended"
    }
  }
}
```

3. **Run locally:**

```bash
npm run build
npx @lhci/cli@latest autorun
```

### CrUX API Not Available

**Symptom:** `403 Forbidden` on CrUX query.

**Solution:**

1. Verify API is enabled: [Google Cloud Console](https://console.cloud.google.com/apis/library/chromeuxreport.googleapis.com)
2. Create API key in Credentials
3. Use `--api-key` flag:

```bash
curl -s "https://chromeuxreport.googleapis.com/v1/records:queryRecord?key=$CRUX_API_KEY" \
  -d '{"url": "https://hmatologia2.web.app/"}' | jq .
```

### Firestore Logs Empty

**Symptom:** `gcloud logging read` returns 0 results.

**Solution:**

1. Verify project: `gcloud config get-value project`
2. Check permission: `gcloud projects get-iam-policy hmatologia2 --flatten="bindings[].members" --filter="bindings.members:$USER"`
3. Adjust time range (logs may be older than 24h):

```bash
gcloud logging read \
  'resource.type="cloud_firestore"' \
  --limit 1000 \
  --format json \
  --project hmatologia2 \
  --freshness 72h  # Past 72 hours instead of 24h
```

### PDF Export Endpoint Requires Auth

**Symptom:** `401 Unauthorized` on Cloud Function call.

**Solution:**

1. Get Firebase ID token:

```bash
firebase auth:export auth-users.json --project hmatologia2
# Or: use existing token from browser DevTools (Network tab)
```

2. Use token in curl:

```bash
FIREBASE_TOKEN=$(gcloud auth print-identity-token)

curl -X POST "https://southamerica-east1-hmatologia2.cloudfunctions.net/generatePdfExport" \
  -H "Authorization: Bearer $FIREBASE_TOKEN" \
  ...
```

### Bundle Analysis Takes Too Long

**Symptom:** `ANALYZE=true npm run build` hangs after build.

**Solution:**

```bash
# Run without opening visualization
ANALYZE=true npm run build
# Wait for build to complete (3-5 min)
# Open dist/stats.html manually
open dist/stats.html
```
