```typescript
import { OCRParseResult, ParsedAnalyte } from '../types/ocr';

export const GEMINI_OCR_PROMPT = `You are analyzing a clinical chemistry analyzer display (H550, Abbott, Siemens, or similar).
Your task: Extract all visible analyte results as structured JSON.

For EACH analyte shown:
1. Extract the analyte NAME (exactly as displayed, e.g., "Glicose", "Ureia", "ALT")
2. Extract the numeric RESULT VALUE
3. Extract the UNIT (mg/dL, mmol/L, U/L, %, etc.)
4. Estimate your confidence in this extraction [0-1]

CRITICAL: Respond ONLY with valid JSON, no markdown, no explanations:
{
  "analytes": [
    {
      "name": "Glicose",
      "value": 105.2,
      "unit": "mg/dL",
      "confidence": 0.98
    }
  ],
  "overallConfidence": 0.95,
  "parseTime": 1200
}

If you cannot read the image clearly, set confidence < 0.7 and include an "errors" field.`;

interface GeminiVisionRequest {
  contents: {
    parts: {
      text: string;
      inline_data?: { mime_type: string; data: string };
      file_data?: { mime_type: string; file_uri: string };
    }[];
  }[];
  generation_config?: {
    temperature?: number;
    top_p?: number;
    top_k?: number;
    max_output_tokens?: number;
  };
}

interface GeminiVisionResponse {
  candidates?: { content?: { parts?: { text?: string }[] } }[];
  error?: { code: number; message: string };
}

/**
 * Parse image with Gemini 2.5 Flash Vision API.
 * Handles base64, URLs, and raw bytes.
 */
export async function parseImageWithGemini(
  imageData: ArrayBuffer | string,  // base64 or file path
  apiKey: string,
  imageMediaType: string = 'image/jpeg'  // jpeg, png, webp, gif
): Promise<OCRParseResult> {
  const startTime = Date.now();

  try {
    let base64Data: string;

    // Convert ArrayBuffer to base64
    if (imageData instanceof ArrayBuffer) {
      const uint8 = new Uint8Array(imageData);
      base64Data = btoa(String.fromCharCode(...uint8));
    } else if (typeof imageData === 'string' && imageData.startsWith('data:')) {
      // Already base64-encoded data URL
      base64Data = imageData.split(',')[1];
    } else {
      base64Data = imageData;
    }

    // Call Gemini 2.5 Flash API
    const response = await fetch('https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        contents: [{
          parts: [
            { text: GEMINI_OCR_PROMPT },
            {
              inline_data: {
                mime_type: imageMediaType,
                data: base64Data,
              },
            },
          ],
        }],
        generation_config: {
          temperature: 0.1,      // Low temp for deterministic extraction
          top_p: 0.9,
          max_output_tokens: 1024,
        },
      } as GeminiVisionRequest),
    } as RequestInit);

    if (!response.ok) {
      const error = await response.json() as { error?: { message: string } };
      throw new Error(`Gemini API error: ${error.error?.message || response.statusText}`);
    }

    const data = (await response.json()) as GeminiVisionResponse;

    if (data.error) {
      throw new Error(`Gemini API error: ${data.error.message}`);
    }

    const textResponse = data.candidates?.[0]?.content?.parts?.[0]?.text || '';

    if (!textResponse) {
      throw new Error('Gemini returned empty response');
    }

    // Parse JSON response
    const jsonMatch = textResponse.match(/\{[\s\S]*\}/);
    if (!jsonMatch) {
      throw new Error('Gemini response did not contain valid JSON');
    }

    const parsed = JSON.parse(jsonMatch[0]) as {
      analytes: ParsedAnalyte[];
      overallConfidence: number;
      errors?: string[];
    };

    // Validate and sanitize
    const analytes = (parsed.analytes || []).map((a: ParsedAnalyte) => ({
      name: String(a.name).trim(),
      value: Number(a.value),
      unit: String(a.unit).trim(),
      confidence: Math.max(0, Math.min(1, Number(a.confidence) || 0)),
      rawLine: a.rawLine,
    }));

    return {
      analytes,
      overallConfidence: Math.max(0, Math.min(1, parsed.overallConfidence || 0.5)),
      parseTime: Date.now() - startTime,
      parserVersion: 'gemini-2.5-flash-v1',
      errors: parsed.errors,
      geminiRawResponse: data,  // Debug
    };
  } catch (err) {
    const parseTime = Date.now() - startTime;
    const error = err instanceof Error ? err.message : String(err);

    return {
      analytes: [],
      overallConfidence: 0,
      parseTime,
      parserVersion: 'gemini-2.5-flash-v1',
      errors: [error],
    };
  }
}

/**
 * Estimate cost of parsing (for billing visibility).
 * Gemini 2.5 Flash: ~$0.075 per 1M input tokens, ~$0.3 per 1M output tokens.
 * Average OCR parse: ~500 tokens input, ~200 tokens output.
 */
export function estimateOCRCost(): number {
  const inputTokens = 500;  // Typical for image + prompt
  const outputTokens = 200;
  const inputPrice = 0.075 / 1_000_000;  // per token
  const outputPrice = 0.3 / 1_000_000;
  return (inputTokens * inputPrice) + (outputTokens * outputPrice);
}
```

# Bioquímica OCR Integration — End-to-End Flow

## Architecture

\`\`\`
Operator uploads image
  ↓

  ↓

  ↓

\`\`\`

## Gemini Prompt Template

See \`geminiVisionService.ts\` for the full prompt. Key features:
- **Input:** Analyzer image (base64 or URL)
- **Output:** JSON with analytes, confidence scores
- **Determinism:** Temperature 0.1 (low randomness)
- **Format:** Strict JSON (no markdown, no explanations)

## Cost Model

- **Per image:** ~$0.0005 (Gemini 2.5 Flash)
- **10 images/day:** ~$0.005/day = $0.15/month
- **100 images/day:** ~$0.05/day = $1.50/month
- **Production:** Monitor via Cloud Logs + budget alert

## Error Handling & Retries

| Scenario | Behavior | Retry |
|----------|----------|-------|
| Network timeout (5s) | Show error, offer manual entry | Yes, 2x |
| Gemini API error (429, 5xx) | Exponential backoff | Yes, 3x |
| Image quality too low | confidence < 0.7, force manual review | No |
| Response parse fails | Return empty analytes, log error | No |

## Fuzz Matching Threshold

- **>95% confidence:** Auto-accept (analyst sees green checkmark)
- **85–95%:** Yellow warning, manual review required
- **70–85%:** Orange caution, operator decides
- **<70%:** Red reject, operator must override or reject

## Test Scenarios

### Scenario 1: Perfect Image (H550, clear display)
- Input: High-quality H550 image with 5 analytes
- Expected: overallConfidence > 0.95, all analytes matched to seed
- User action: Approve → proceeds to Westgard

### Scenario 2: Low Confidence (grainy image, partial display)
- Input: Blurry photo, some values obscured
- Expected: overallConfidence 0.60–0.80, 2 analytes unmatchable
- User action: Manual review, override fuzzy matches or enter manually

### Scenario 3: Unrecognized Analyte (new test, not in seed)
- Input: Image with "CustomMarker-01" (not in 25+ seeded list)
- Expected: fuzzyMatch returns no-match, requiresOperatorReview = true
- User action: Operator selects from seed or creates custom

### Scenario 4: Timeout (slow network, API unavailable)
- Input: Trigger with network throttled to <1 Mbps
- Expected: Timeout after 5s, show "Parsing failed" + retry button
- User action: Retry or fall back to manual entry

### Scenario 5: Malformed Image (not a valid image file)
- Input: Text file with .jpg extension
- Expected: Gemini returns error, overallConfidence = 0, analytes = []
- User action: Show error, allow re-upload

## Rollback Procedure

If Gemini API becomes unavailable:
1. Disable OCR upload button (show "OCR temporarily unavailable")
2. Keep manual entry UX working
3. Operator manually enters values as in Phase 9.1
4. Once API restored, re-enable with same prompt + prompt version

- **RDC 978 Art. 161:** Operator ID + timestamp logged with each OCR decision
- **DICQ 4.3 (audit trail):** All parse → match → validation steps appended to \`audit/{logId}\`
- **Gemini terms:** Processed data does not train future models (Flash tier)

```typescript
import { Analito } from '../types/analito';
import { ParsedAnalyte, MatchResult } from '../types/ocr';

/**
 * Levenshtein distance: measures difference between two strings.
 */
function levenshteinDistance(a: string, b: string): number {
  const aLen = a.length;
  const bLen = b.length;
  const matrix: number[][] = Array(aLen + 1)
    .fill(null)
    .map(() => Array(bLen + 1).fill(0));

  for (let i = 0; i <= aLen; i++) matrix[i][0] = i;
  for (let j = 0; j <= bLen; j++) matrix[0][j] = j;

  for (let i = 1; i <= aLen; i++) {
    for (let j = 1; j <= bLen; j++) {
      const cost = a[i - 1] === b[j - 1] ? 0 : 1;
      matrix[i][j] = Math.min(
        matrix[i - 1][j] + 1,      // deletion
        matrix[i][j - 1] + 1,      // insertion
        matrix[i - 1][j - 1] + cost  // substitution
      );
    }
  }

  return matrix[aLen][bLen];
}

/**
 * Normalize string for matching (lowercase, remove accents, strip whitespace).
 */
function normalize(str: string): string {
  return str
    .toLowerCase()
    .normalize('NFD')
    .replace(/[̀-ͯ]/g, '')  // Remove diacritics
    .trim();
}

/**
 * Semantic similarity check (simple heuristics).
 * E.g., "glucose" ≈ "glicose", "albumin" ≈ "albumina"
 */
function semanticSimilarity(a: string, b: string): number {
  const normalized_a = normalize(a);
  const normalized_b = normalize(b);

  // Exact match after normalization
  if (normalized_a === normalized_b) return 1.0;

  // Common Portuguese/English endings (albumin/albumina, etc.)
  const patterns = [
    ['albumin', 'albumina'],
    ['glucose', 'glicose'],
    ['urea', 'ureia'],
    ['triglyceride', 'trigliceride'],
    ['cholesterol', 'colesterol'],
  ];

  for (const [en, pt] of patterns) {
    if (
      (normalized_a.includes(en) && normalized_b.includes(pt)) ||
      (normalized_a.includes(pt) && normalized_b.includes(en))
    ) {
      return 0.95;
    }
  }

  return 0;
}

/**
 * Calculate fuzzy match confidence [0-1].
 */
function calculateConfidence(distance: number, maxLen: number): number {
  if (maxLen === 0) return 0;
  const normalized = 1 - distance / maxLen;
  return Math.max(0, Math.min(1, normalized));
}

/**
 * Fuzzy match parsed analyte against seed.
 */
export function fuzzyMatchAnalyte(
  parsedName: string,
  seededAnalitos: Analito[],
  threshold: number = 0.7
): MatchResult {
  if (!parsedName || seededAnalitos.length === 0) {
    return {
      parsedAnalyte: { name: parsedName, value: 0, unit: '', confidence: 0 } as ParsedAnalyte,
      matchConfidence: 0,
      matchReason: 'no-match',
      requiresOperatorReview: true,
      overrideAllowed: true,
    };
  }

  let bestMatch: Analito | null = null;
  let bestScore = 0;
  let matchReason = 'no-match';

  const normalizedParsed = normalize(parsedName);

  for (const analito of seededAnalitos) {
    const normalizedSeed = normalize(analito.nome);

    // Try exact match first
    if (normalizedParsed === normalizedSeed) {
      return {
        parsedAnalyte: { name: parsedName, value: 0, unit: '', confidence: 0 } as ParsedAnalyte,
        matchedAnalitoId: analito.id,
        matchedAnalito: analito,
        matchConfidence: 1.0,
        matchReason: 'exact',
        requiresOperatorReview: false,
        overrideAllowed: false,
      };
    }

    // Try semantic similarity
    const semanticScore = semanticSimilarity(normalizedParsed, normalizedSeed);
    if (semanticScore > bestScore) {
      bestScore = semanticScore;
      bestMatch = analito;
      matchReason = 'semantic';
    }

    // Try Levenshtein distance
    const distance = levenshteinDistance(normalizedParsed, normalizedSeed);
    const maxLen = Math.max(normalizedParsed.length, normalizedSeed.length);
    const confidence = calculateConfidence(distance, maxLen);

    if (confidence > bestScore && confidence >= threshold) {
      bestScore = confidence;
      bestMatch = analito;
      matchReason = `fuzzy-${Math.round(confidence * 100)}`;
    }
  }

  if (bestMatch && bestScore >= threshold) {
    return {
      parsedAnalyte: { name: parsedName, value: 0, unit: '', confidence: 0 } as ParsedAnalyte,
      matchedAnalitoId: bestMatch.id,
      matchedAnalito: bestMatch,
      matchConfidence: bestScore,
      matchReason,
      requiresOperatorReview: bestScore < 0.85,
      overrideAllowed: true,
    };
  }

  return {
    parsedAnalyte: { name: parsedName, value: 0, unit: '', confidence: 0 } as ParsedAnalyte,
    matchConfidence: 0,
    matchReason: 'no-match',
    requiresOperatorReview: true,
    overrideAllowed: true,
  };
}
```

```typescript
import { useState } from 'react';
import { parseImageWithGemini, estimateOCRCost } from '../services/geminiVisionService';
import { fuzzyMatchAnalyte } from '../utils/ocrFuzzyMatch';
import { OCRParseResult, OCRValidationResult, MatchResult } from '../types/ocr';
import { Analito } from '../types/analito';

export interface UseGeminiVisionState {
  parsing: boolean;
  result: OCRParseResult | null;
  validationResult: OCRValidationResult | null;
  error: string | null;
  progress: 'idle' | 'uploading' | 'parsing' | 'matching' | 'validating';
  costEstimate: number;
}

export function useGeminiVision() {
  const [state, setState] = useState<UseGeminiVisionState>({
    parsing: false,
    result: null,
    validationResult: null,
    error: null,
    progress: 'idle',
    costEstimate: 0,
  });

  async function parseImage(
    imageFile: File,
    seededAnalitos: Analito[]
  ): Promise<OCRValidationResult | null> {
    setState(s => ({ ...s, parsing: true, error: null, progress: 'uploading' }));

    try {
      // Read file as ArrayBuffer
      const arrayBuffer = await imageFile.arrayBuffer();

      // Call Gemini (with timeout)
      setState(s => ({ ...s, progress: 'parsing' }));
      const parsePromise = parseImageWithGemini(
        arrayBuffer,
        import.meta.env.VITE_GEMINI_API_KEY as string,
        imageFile.type
      );

      const timeoutPromise = new Promise<OCRParseResult>((_, reject) =>
        setTimeout(() => reject(new Error('Parsing timeout (5s)')), 5000)
      );

      const result = await Promise.race([parsePromise, timeoutPromise]);

      setState(s => ({ ...s, result, progress: 'matching' }));

      // Fuzzy match parsed analytes against seed
      const matches: MatchResult[] = result.analytes.map(parsed => ({
        ...fuzzyMatchAnalyte(parsed.name, seededAnalitos),
        parsedAnalyte: parsed,
      }));

      setState(s => ({ ...s, progress: 'validating' }));

      // Aggregate validation result
      const unrecognizedAnalytes = matches
        .filter(m => !m.matchedAnalitoId)
        .map(m => m.parsedAnalyte.name);

      const requiresOperatorReview = matches.some(m => m.requiresOperatorReview);

      const validationResult: OCRValidationResult = {
        matches,
        overallConfidence: result.overallConfidence,
        requiresOperatorReview,
        unrecognizedAnalytes,
        readyForWestgard: !requiresOperatorReview && unrecognizedAnalytes.length === 0,
        costEstimate: estimateOCRCost(),
        errorMessages: result.errors,
      };

      setState(s => ({
        ...s,
        validationResult,
        parsing: false,
        progress: 'idle',
        costEstimate: validationResult.costEstimate || 0,
      }));

      return validationResult;
    } catch (err) {
      const errorMsg = err instanceof Error ? err.message : 'Unknown error';
      setState(s => ({
        ...s,
        parsing: false,
        error: errorMsg,
        progress: 'idle',
      }));
      return null;
    }
  }

  function reset() {
    setState({
      parsing: false,
      result: null,
      validationResult: null,
      error: null,
      progress: 'idle',
      costEstimate: 0,
    });
  }

  return {
    ...state,
    parseImage,
    reset,
  };
}
```

```typescript
import { collection, addDoc, Timestamp } from 'firebase/firestore';
import { db } from '../../../shared/services/firebase';
import { OCRAuditEntry, OCRValidationResult, ParsedAnalyte } from '../types/ocr';
import { generateLogicalSignature } from '../../../shared/utils/signatureHelpers';

/**
 * Record OCR decision to audit trail.
 */
export async function recordOCRaudit(
  labId: string,
  operatorId: string,
  imageUrl: string,
  parseResult: any,  // OCRParseResult
  validationResult: OCRValidationResult,
  operatorDecision: {
    approvedMatches: string[];
    rejectedMatches: string[];
    overrides: { parsedName: string; overriddenAnalitoId: string }[];
  }
): Promise<string> {
  const signature = generateLogicalSignature(operatorId);

  const entry: OCRAuditEntry = {
    id: '',  // Firestore generates
    labId,
    operatorId,
    imageUrl,
    parseResult,
    validationResult,
    operatorDecision,
    timestamp: Timestamp.now(),
    chainHash: signature.hash,  // RDC 978 5.3 audit trail
  };

  const auditRef = collection(db, `labs/${labId}/bioquimica/audit`);
  const docRef = await addDoc(auditRef, {
    ...entry,
    timestamp: Timestamp.now(),
    operatorId,  // Redundant for RLS
    labId,       // Redundant for RLS
  });

  return docRef.id;
}

/**
 * Log OCR event (non-critical, for analytics).
 */
export async function logOCREvent(
  labId: string,
  operatorId: string,
  eventType: 'parse' | 'match' | 'approve' | 'reject',
  metadata: Record<string, unknown>
): Promise<void> {
  try {
    const eventsRef = collection(db, 'events');
    await addDoc(eventsRef, {
      event: `ocr-${eventType}`,
      labId,
      operatorId,
      timestamp: Timestamp.now(),
      metadata,
    });
  } catch (err) {
    console.error('[OCR Event Log Error]', err);
  }
}
```
