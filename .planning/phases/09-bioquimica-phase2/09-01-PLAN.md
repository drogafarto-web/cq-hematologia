```typescript
export function evaluateRules(
  points: number[],                    // Control values measured (in order)
  config: WestgardConfig,              // Which rules enabled
  sigma: number = 1                    // Std dev for this analyte (computed or from bula)
): RuleViolation[] | null {
  if (!config.enabled || points.length === 0) return null;
  
  const violations: RuleViolation[] = [];
  
  // Implement each rule:
  
  // 1-2s: Any single point >2σ
  if (config.rules["1-2s"]) {
    for (let i = 0; i < points.length; i++) {
      if (Math.abs(points[i]) > 2 * sigma) {
        violations.push({
          ruleName: "1-2s",
          reason: `Point ${i + 1} exceeds 2σ (value: ${points[i]}, threshold: ±${2 * sigma})`,
          affectedPoints: [i],
          severity: "warn",
        });
      }
    }
  }
  
  // 1-3s: Any single point >3σ (replaces 1-2s as reject)
  if (config.rules["1-3s"]) {
    for (let i = 0; i < points.length; i++) {
      if (Math.abs(points[i]) > 3 * sigma) {
        violations.push({
          ruleName: "1-3s",
          reason: `Point ${i + 1} exceeds 3σ (value: ${points[i]}, threshold: ±${3 * sigma})`,
          affectedPoints: [i],
          severity: "reject",
        });
      }
    }
  }
  
  // 2-2s: 2 consecutive points >2σ (same side)
  if (config.rules["2-2s"]) {
    for (let i = 0; i < points.length - 1; i++) {
      const p1 = points[i];
      const p2 = points[i + 1];
      if (
        Math.abs(p1) > 2 * sigma &&
        Math.abs(p2) > 2 * sigma &&
        (p1 > 0 === p2 > 0)  // Same side
      ) {
        violations.push({
          ruleName: "2-2s",
          reason: `Points ${i + 1}–${i + 2} both exceed 2σ on same side`,
          affectedPoints: [i, i + 1],
          severity: "reject",
        });
      }
    }
  }
  
  // R-4s: Range of 2 points exceeds 4σ
  if (config.rules["R-4s"]) {
    for (let i = 0; i < points.length - 1; i++) {
      const range = Math.abs(points[i] - points[i + 1]);
      if (range > 4 * sigma) {
        violations.push({
          ruleName: "R-4s",
          reason: `Range between points ${i + 1}–${i + 2} exceeds 4σ (range: ${range.toFixed(2)}, threshold: ${(4 * sigma).toFixed(2)})`,
          affectedPoints: [i, i + 1],
          severity: "reject",
        });
      }
    }
  }
  
  // Extended rules (if enabled):
  
  // 4-1s: 4+ consecutive >1σ (same side)
  if (config.rules["4-1s"]) {
    for (let i = 0; i <= points.length - 4; i++) {
      const window = points.slice(i, i + 4);
      if (
        window.every(p => Math.abs(p) > sigma) &&
        window.every(p => p > 0) || window.every(p => p < 0)  // All same side
      ) {
        violations.push({
          ruleName: "4-1s",
          reason: `4 consecutive points exceed 1σ on same side`,
          affectedPoints: [i, i + 1, i + 2, i + 3],
          severity: "reject",
        });
      }
    }
  }
  
  // 10x: 10+ consecutive on same side of mean (0)
  if (config.rules["10x"]) {
    for (let i = 0; i <= points.length - 10; i++) {
      const window = points.slice(i, i + 10);
      if (window.every(p => p > 0) || window.every(p => p < 0)) {
        violations.push({
          ruleName: "10x",
          reason: `10 consecutive points on same side of mean`,
          affectedPoints: Array.from({ length: 10 }, (_, j) => i + j),
          severity: "reject",
        });
      }
    }
  }
  
  // 6T (trend): 6+ consecutive increasing or decreasing
  if (config.rules["6T"]) {
    let ascendingCount = 1;
    let descendingCount = 1;
    for (let i = 1; i < points.length; i++) {
      if (points[i] > points[i - 1]) {
        ascendingCount++;
        descendingCount = 1;
      } else if (points[i] < points[i - 1]) {
        descendingCount++;
        ascendingCount = 1;
      } else {
        ascendingCount = 1;
        descendingCount = 1;
      }
      
      if (ascendingCount >= 6 || descendingCount >= 6) {
        const affectedStart = i - (ascendingCount >= 6 ? ascendingCount - 1 : descendingCount - 1);
        violations.push({
          ruleName: "6T",
          reason: `6 consecutive ${ascendingCount >= 6 ? "ascending" : "descending"} points`,
          affectedPoints: Array.from({ length: 6 }, (_, j) => affectedStart + j),
          severity: "reject",
        });
        break;  // Report once per trend
      }
    }
  }
  
  // 6X (alternating): 6+ alternating high/low
  if (config.rules["6X"]) {
    let alternatingCount = 1;
    for (let i = 1; i < points.length; i++) {
      if ((points[i] > 0 && points[i - 1] < 0) || (points[i] < 0 && points[i - 1] > 0)) {
        alternatingCount++;
      } else {
        alternatingCount = 1;
      }
      
      if (alternatingCount >= 6) {
        const affectedStart = i - 5;
        violations.push({
          ruleName: "6X",
          reason: `6 alternating points around mean`,
          affectedPoints: Array.from({ length: 6 }, (_, j) => affectedStart + j),
          severity: "reject",
        });
        break;
      }
    }
  }
  
  return violations.length > 0 ? violations : null;
}
```

```typescript
import { evaluateRules } from '../utils/westgardRulesCLSI';

export function useAcceptanceEngine(analito: Analito, nivel: ControlLevel) {
  const [points, setPoints] = useState<number[]>([]);
  const [violations, setViolations] = useState<RuleViolation[] | null>(null);
  
  function addPoint(value: number) {
    const newPoints = [...points, value];
    setPoints(newPoints);
    
    // Compute sigma from CV or historical data
    const sigma = computeSigma(analito, nivel);
    
    // Evaluate rules
    const newViolations = evaluateRules(newPoints, analito.westgardRules, sigma);
    setViolations(newViolations);
  }
  
  function clearPoints() {
    setPoints([]);
    setViolations(null);
  }
  
  return {
    points,
    violations,
    status: violations?.some(v => v.severity === 'reject') ? 'reject' : 
            violations?.some(v => v.severity === 'warn') ? 'warn' : 'ok',
    addPoint,
    clearPoints,
  };
}

function computeSigma(analito: Analito, nivel: ControlLevel): number {
  // If bula has cvIntra, use it; otherwise use historical internal stats
  // sigma = mean * (cvIntra / 100)
  if (analito.cvIntra) {
    return nivel.expectedValue * (analito.cvIntra / 100);
  }
  // Fallback: query internalStats for this analito/nivel/equipment
  return 1.0;  // Placeholder
}
```

```typescript
  run: RunToEvaluate,
  analitoMap: Map<string, Analito>  // Pre-fetched analito metadata
): Promise<Map<string, RunComplianceResult>> {
  const results = new Map<string, RunComplianceResult>();
  
  for (const point of run.analitos) {
    const analito = analitoMap.get(point.analitoId);
    if (!analito) {
      results.set(point.analitoId, {
        accepted: false,
        violations: [{
          ruleName: 'metadata',
          reason: `Analito ${point.analitoId} not found in metadata`,
          affectedPoints: [],
          severity: 'reject',
        }],
        applicableRules: [],
        complianceScore: 0,
      });
      continue;
    }
    
    // Rules suspended if bula pending
    if (analito.bulaPendente) {
      results.set(point.analitoId, {
        accepted: true,  // Accept, but note suspension
        violations: [{
          ruleName: 'bulaPendente',
          reason: 'Westgard rules suspended: bula not yet processed',
          affectedPoints: [],
          severity: 'warn',
        }],
        applicableRules: [],
        complianceScore: 0.5,
      });
      continue;
    }
    
    // Fetch historical points for this analito/nivel/equipment
    const historicalPoints = await fetchHistoricalPoints(
      point.analitoId,
      point.nivelId,
      point.equipmentId
    );
    
    const allPoints = [...historicalPoints, point.value];
    const sigma = computeSigma(analito, allPoints);
    
    // Evaluate rules
    const violations = evaluateRulesServer(
      allPoints,
      analito.westgardRules,
      sigma
    );
    
    const accepted = !violations.some(v => v.severity === 'reject');
    const applicableRules = Object.keys(analito.westgardRules.rules).filter(
      r => analito.westgardRules.rules[r as keyof typeof analito.westgardRules.rules]
    );
    
    const complianceScore = accepted ? 1.0 : (1.0 - (violations.length * 0.1));
    
    results.set(point.analitoId, {
      accepted,
      violations,
      applicableRules,
      complianceScore,
    });
  }
  
  return results;
}

function computeSigma(analito: Analito, points: number[]): number {
  // If analito has cvIntra, use it
  if (analito.cvIntra && points.length > 0) {
    const mean = points.reduce((a, b) => a + b, 0) / points.length;
    return mean * (analito.cvIntra / 100);
  }
  // Otherwise, compute from points (if enough data)
  if (points.length >= 5) {
    const mean = points.reduce((a, b) => a + b, 0) / points.length;
    const variance = points.reduce((sum, p) => sum + Math.pow(p - mean, 2), 0) / points.length;
    return Math.sqrt(variance);
  }
  return 1.0;
```
