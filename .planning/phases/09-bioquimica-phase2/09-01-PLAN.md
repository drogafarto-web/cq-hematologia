## SA-03 — `src/features/bioquimica/types/ocr.ts`

Criar types para todo o pipeline OCR: parse → match → validação → audit.

```typescript
// Exports obrigatórios:
export interface ParsedAnalyte {
  name: string;
  value: number;
  unit: string;
  confidence: number; // [0-1] do Gemini
  rawLine?: string;
}

export interface OCRParseResult {
  analytes: ParsedAnalyte[];
  overallConfidence: number;
  parseTime: number;
  parserVersion: string;
  errors?: string[];
}

export interface MatchResult {
  parsedAnalyte: ParsedAnalyte;
  matchedAnalitoId?: string;
  matchedAnalito?: import('./analito').Analito;
  matchConfidence: number;
  matchReason: 'exact' | 'semantic' | `fuzzy-${number}` | 'no-match';
  requiresOperatorReview: boolean;
  overrideAllowed: boolean;
}

export interface OCRValidationResult {
  matches: MatchResult[];
  overallConfidence: number;
  requiresOperatorReview: boolean;
  unrecognizedAnalytes: string[];
  readyForWestgard: boolean;
  costEstimate?: number;
}

export interface OCRAuditEntry {
  id: string;
  labId: string;
  operatorId: string;
  imageUrl: string;
  parseResult: OCRParseResult;
  validationResult: OCRValidationResult;
  operatorDecision: {
    approvedMatches: string[];
    rejectedMatches: string[];
    overrides: { parsedName: string; overriddenAnalitoId: string }[];
  };
  timestamp: import('firebase/firestore').Timestamp;
  chainHash?: string;
}
```
