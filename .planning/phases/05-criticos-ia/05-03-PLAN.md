**Callable:** `parseImmunologyStrip(labId, imageUrl, equipment: "yumizen-h550", metadata: {})`

**Requirements:**

```typescript

interface ParsedStripResult {
  success: boolean;
  analites: {
    analitoId: string;
    analitoName: string;
    resultado: number | null;    // null if not detected
    unit: string;
    confidence: number;          // 0.0 - 1.0 (Gemini confidence)

      colorCalibration: boolean;  // color bar visible & detected
      glareDetected: boolean;
      blurDetected: boolean;
    };
  }[];
  overallConfidence: number;     // 0.0 - 1.0 (avg of analites)
  rawResponse: string;           // Gemini's complete response (for audit)
  processingTimeMs: number;
  modelVersion: string;          // "gemini-2.5-flash-001" or current
  costEstimate: number;          // USD, for cost tracking
}
```

**Gemini prompt design:**

```
You are an expert medical laboratory analyzer. You will analyze an image 
of an immunology analyzer strip (Yumizen H550) and extract all visible analyte 
results.

For EACH analyte visible:
1. Identify the analyte name (e.g., IgE, IgG, IgA, IgM, C3, C4, etc.)
2. Extract the numerical result
3. Provide confidence 0.0-1.0 for this extraction
4. Note any anomalies (image quality, glare, blur, missing edges, color issues)

Response format (JSON):
{
  "analites": [
    {
      "analitoName": "IgE",
      "resultado": 234.5,
      "unit": "IU/mL",
      "confidence": 0.95,
      "anomalies": ["none"]
    },
    ...
  ],
  "overallConfidence": 0.92,
  "imageQuality": "GOOD",
  "anomalyFlags": {
    "glareDetected": false,
    "blurDetected": false,
    "colorCalibrationVisible": true,
    "stripEdgeComplete": true
  }
}
```

**Integration pattern:**

```typescript
import { GoogleGenerativeAI } from "@google/generative-ai";

const genai = new GoogleGenerativeAI(process.env.GEMINI_API_KEY);

export async function parseImmunologyStrip(
  labId: string,
  imageUrl: string,
  equipment: "yumizen-h550"
): Promise<ParsedStripResult> {
  const startTime = Date.now();
  
  try {
    const model = genai.getGenerativeModel({ model: "gemini-2.5-flash" });
    
    // Fetch image from Cloud Storage (public URL or signed)
    const imageData = await fetch(imageUrl).then(r => r.arrayBuffer());
    const base64Image = Buffer.from(imageData).toString("base64");
    
    const response = await model.generateContent([
      {
        inlineData: {
          mimeType: "image/jpeg",
          data: base64Image,
        },
      },
      {
        text: STRIP_PARSING_PROMPT,  // defined above
      },
    ]);
    
    const jsonResponse = JSON.parse(response.text());
    
    return {
      success: true,
      analites: jsonResponse.analites.map((a: any) => ({
        analitoId: normalizeAnalitoName(a.analitoName),
        analitoName: a.analitoName,
        resultado: a.resultado,
        unit: a.unit || "IU/mL",
        confidence: a.confidence,
        anomalyFlags: {
          imageQuality: jsonResponse.imageQuality,
          edgeDetection: getEdgeDetection(jsonResponse.anomalyFlags),
          colorCalibration: jsonResponse.anomalyFlags.colorCalibrationVisible,
          glareDetected: jsonResponse.anomalyFlags.glareDetected,
          blurDetected: jsonResponse.anomalyFlags.blurDetected,
        },
      })),
      overallConfidence: jsonResponse.overallConfidence,
      rawResponse: response.text(),
      processingTimeMs: Date.now() - startTime,
      modelVersion: "gemini-2.5-flash",
      costEstimate: calculateGeminiCost(imageUrl),
    };
  } catch (error) {
    return {
      success: false,
      analites: [],
      overallConfidence: 0,
      rawResponse: `Error: ${error.message}`,
      processingTimeMs: Date.now() - startTime,
      modelVersion: "gemini-2.5-flash",
      costEstimate: 0,
    };
  }
}
```

**Error handling:**

- Image URL invalid/timeout: return { success: false, error: "IMAGE_FETCH_FAILED" }
- Gemini API error: return { success: false, error: "API_ERROR", retryable: true }
- Timeout (>30s): abort + return { success: false, error: "TIMEOUT" }
- No analites detected: return { success: true, analites: [], overallConfidence: 0 }

### 3. Result Reconciliation with Analyzer

**After OCR parsing, system must reconcile with analyzer-generated reference:**

```typescript
interface StripReconciliation {
  stripId: string;
  runId?: string;                // link to analyzer run if exists
  matchStatus: "PERFECT" | "CLOSE" | "DIVERGENT" | "MISSING_ANALYZER"; 
  
  // For each analite
  analites: {
    analitoId: string;
    ocrResult: number;
    analyzerResult?: number;
    delta: number;               // |ocr - analyzer|
    deltaPercent: number;        // (delta / analyzer) * 100
    isAcceptable: boolean;       // delta <= threshold (usually 2-5%)
    conflictReason?: string;
  }[];
  
  reconciliationScore: number;   // 0-100, % of analites acceptable
  recommendedAction: "ACCEPT" | "REVIEW" | "REJECT";
  notes: string;
}
```

**Reconciliation logic:**

```typescript
export function reconcileStripResults(
  ocrResults: ParsedStripResult,
  analyzerRun?: BioquimicaRun
): StripReconciliation {
  if (!analyzerRun) {
    return {
      matchStatus: "MISSING_ANALYZER",
      recommendedAction: "ACCEPT",  // no conflict if no reference
      // ...
    };
  }
  
  const analites = ocrResults.analites.map(ocr => {
    const analyzer = analyzerRun.analitos.find(a => a.id === ocr.analitoId);
    const delta = Math.abs(ocr.resultado - (analyzer?.resultado || 0));
    const threshold = getAcceptableThreshold(ocr.analitoId);
    
    return {
      analitoId: ocr.analitoId,
      ocrResult: ocr.resultado,
      analyzerResult: analyzer?.resultado,
      delta,
      deltaPercent: (delta / (analyzer?.resultado || 1)) * 100,
      isAcceptable: delta <= threshold,
    };
  });
  
  const acceptableCount = analites.filter(a => a.isAcceptable).length;
  const score = (acceptableCount / analites.length) * 100;
  
  return {
    matchStatus: score > 95 ? "PERFECT" : score > 80 ? "CLOSE" : "DIVERGENT",
    recommendedAction: score > 90 ? "ACCEPT" : score > 70 ? "REVIEW" : "REJECT",
    analites,
    reconciliationScore: score,
  };
}
```
