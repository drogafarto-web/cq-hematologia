import { doc, getDoc, collection, query, where, getDocs } from 'firebase/firestore';
import { getFirestore } from 'firebase/app';
import { CriticoDocument, LabCriticosSettings } from '../types/machine';

export interface EscalationQueueItem {
  id: string;
  resultId: string;
  labId: string;
  status: 'pending' | 'sent' | 'failed' | 'acknowledged';
  createdAt: number;
  payload: {
    paciente: { nome: string };
    analyte: { nome: string; valor: number; unidade: string };
    criticalRange: { min?: number; max?: number };
  };
  attempts: number;
  lastAttemptAt?: number;
  sentAt?: number;
  acknowledgedAt?: number;
  deliveryLog: Array<{
    method: 'sms' | 'whatsapp' | 'email';
    timestamp: number;
    status: 'pending' | 'sent' | 'failed';
    error?: string;
    messageId?: string;
  }>;
}

export interface NotificationTemplate {
  sms: string;
  whatsapp: string;
  email: { subject: string; body: string };
}

/**
 * Escalation Service: Queue processor + template builder
 * Consumed by Cloud Functions + scheduled monitors
 */
export class EscalationService {
  /**
   * Fetch next pending escalation from queue
   */
  static async getPendingEscalations(
    labId: string,
    limit: number = 10,
  ): Promise<EscalationQueueItem[]> {
    try {
      const q = query(
        collection(getFirestore(), `labs/${labId}/criticos`),
        where('state', '==', 'ALERTADO'),
        where('escalationQueue', 'array-contains', { status: 'pending' }),
      );
      // Note: Firestore doesn't support array subcollection queries in SDK
      // Workaround: fetch all ALERTADOs, filter client-side, then query subcollections
      const criticosSnap = await getDocs(q);
      const escalations: EscalationQueueItem[] = [];

      for (const criticoDoc of criticosSnap.docs) {
        const queueSnap = await getDocs(
          query(collection(criticoDoc.ref, 'escalationQueue'), where('status', '==', 'pending')),
        );
        for (const queueDoc of queueSnap.docs) {
          escalations.push({ id: queueDoc.id, ...queueDoc.data() } as EscalationQueueItem);
        }
      }

      return escalations.slice(0, limit);
    } catch (err) {
      console.error(`Error fetching escalations for lab ${labId}:`, err);
      return [];
    }
  }

  /**
   * Build notification text/email templates based on critico data + lab branding
   */
  static buildNotificationTemplate(
    critico: CriticoDocument,
    labName: string,
    acknowledgmentUrl: string,
    labSettings?: LabCriticosSettings,
  ): NotificationTemplate {
    const { paciente, analyte, threshold } = critico;
    const slaMinutes = labSettings?.slaThresholdMs
      ? Math.round(labSettings.slaThresholdMs / 60000)
      : 5;

    // SMS (160 chars, concise)
    const sms =
      `🚨 RESULTADO CRÍTICO - ${labName}\n` +
      `Paciente: ${paciente.nome}\n` +
      `${analyte.nome}: ${analyte.valor} ${analyte.unidade}\n` +
      `Intervalo crítico: ${threshold.criticalMin || '—'} a ${threshold.criticalMax || '—'}\n` +
      `Confirme em ${slaMinutes}min: ${acknowledgmentUrl}`;

    // WhatsApp (supports longer messages + rich formatting)
    const whatsapp =
      `🚨 *RESULTADO CRÍTICO* - ${labName}\n\n` +
      `*Paciente:* ${paciente.nome}\n` +
      `*Analyte:* ${analyte.nome}\n` +
      `*Valor:* ${analyte.valor} ${analyte.unidade}\n` +
      `*Intervalo Normal:* ${threshold.normalMin}–${threshold.normalMax}\n` +
      `*Intervalo Crítico:* ${threshold.criticalMin || '—'} a ${threshold.criticalMax || '—'}\n\n` +
      `Confirme recebimento em *${slaMinutes} minutos*:\n` +
      `${acknowledgmentUrl}`;

    // Email (full details for supervisor)
    const emailSubject = `[CRÍTICO] ${analyte.nome} - ${paciente.nome} - ${labName}`;
    const emailBody = `
<!DOCTYPE html>
<html>
<body style="font-family: Arial; color: #333;">
  <h2 style="color: red;">⚠️ RESULTADO CRÍTICO DETECTADO</h2>
  
  <h3>Informações do Paciente</h3>
  <p><strong>Nome:</strong> ${paciente.nome}</p>
  <p><strong>Laboratório:</strong> ${labName}</p>
  
  <h3>Resultado</h3>
  <p><strong>Análise:</strong> ${analyte.nome}</p>
  <p><strong>Valor Obtido:</strong> <strong style="color: red; font-size: 1.2em;">${analyte.valor} ${analyte.unidade}</strong></p>
  <p><strong>Intervalo Normal:</strong> ${threshold.normalMin}–${threshold.normalMax} ${analyte.unidade}</p>
  <p><strong>Intervalo Crítico (Limite):</strong> ${threshold.criticalMin || '—'} a ${threshold.criticalMax || '—'} ${analyte.unidade}</p>
  
  <h3>Ações Necessárias</h3>
  <p>A equipe on-call foi notificada. Pressione o botão abaixo para confirmar recebimento.</p>
  <p style="text-align: center; margin: 20px 0;">
    <a href="${acknowledgmentUrl}" style="padding: 10px 20px; background-color: green; color: white; text-decoration: none; border-radius: 5px; display: inline-block;">
      ✓ Confirmar Recebimento
    </a>
  </p>
  
  <p style="color: #666; font-size: 0.9em;">
    <strong>SLA:</strong> Confirme em até ${slaMinutes} minutos da notificação.<br>
    <strong>Data/Hora:</strong> ${new Date().toLocaleString('pt-BR')}
  </p>
</body>
</html>
    `;

    return {
      sms: sms.substring(0, 160), // SMS limit
      whatsapp,
      email: { subject: emailSubject, body: emailBody },
    };
  }

  /**
   * Mark escalation as sent (update queue status + delivery log)
   */
  static async markEscalationSent(
    labId: string,
    resultId: string,
    queueId: string,
    method: 'sms' | 'whatsapp' | 'email',
    messageId?: string,
  ): Promise<void> {
    try {
      const queueRef = doc(
        getFirestore(),
        `labs/${labId}/criticos/${resultId}/escalationQueue/${queueId}`,
      );
      await updateDoc(queueRef, {
        status: 'sent',
        sentAt: Date.now(),
        deliveryLog: arrayUnion({
          method,
          timestamp: Date.now(),
          status: 'sent',
          messageId,
        }),
      });
    } catch (err) {
      console.error(`Error marking escalation sent:`, err);
    }
  }

  /**
   * Mark escalation as failed (retry logic handled by scheduled function)
   */
  static async markEscalationFailed(
    labId: string,
    resultId: string,
    queueId: string,
    method: 'sms' | 'whatsapp' | 'email',
    error: string,
  ): Promise<void> {
    try {
      const queueRef = doc(
        getFirestore(),
        `labs/${labId}/criticos/${resultId}/escalationQueue/${queueId}`,
      );
      await updateDoc(queueRef, {
        attempts: increment(1),
        lastAttemptAt: Date.now(),
        deliveryLog: arrayUnion({
          method,
          timestamp: Date.now(),
          status: 'failed',
          error,
        }),
        // If 3+ failures, mark as failed (stop retrying, escalate to admin)
        .../* attempts would be >= 3 */ {
          status: 'failed',
        },
      });
    } catch (err) {
      console.error(`Error marking escalation failed:`, err);
    }
  }
}

**Function 2: sendCriticoNotification**

- Input: { labId: string, criticoId: string, operator: { id, phone?, whatsappId? }, method: 'sms' | 'whatsapp' | 'email', template: NotificationTemplate }
- Call Twilio SDK:
  - SMS: `twilio.messages.create({ to, from, body })`
  - WhatsApp: `twilio.messages.create({ to, from, body, contentVariables, mediaUrl })`
  - Email: Use Twilio SendGrid integration OR Firebase email (simpler for this phase)
- On success:
  - Return: { sent: true, messageId, sentAt }
  - Call EscalationService.markEscalationSent()
- On failure:
  - Return: { sent: false, error: string }
  - Call EscalationService.markEscalationFailed()
- Retry logic: max 3 attempts, exponential backoff (handled by scheduled function in Task 3)

```typescript
// functions/src/callables/criticos/sendCriticoNotification.ts
import * as twilio from 'twilio';
import * as admin from 'firebase-admin';
import { EscalationService } from '../../shared/escalation-service';

const twilioClient = twilio(process.env.TWILIO_ACCOUNT_SID, process.env.TWILIO_AUTH_TOKEN);

export const sendCriticoNotification = onCall({ region: 'southamerica-east1' }, async (request) => {
  const { labId, criticoId, operator, method, template, queueId } = request.data;

  const criticoRef = admin.firestore().doc(`labs/${labId}/criticos/${criticoId}`);
  const criticoSnap = await criticoRef.get();

  if (!criticoSnap.exists) {
    throw new HttpsError('not-found', `Critical result ${criticoId} not found`);
  }

  const critico = criticoSnap.data();
  let messageId: string | undefined;
  let sent = false;
  let error: string | undefined;

  try {
    if (method === 'sms' && operator.phone) {
      const message = await twilioClient.messages.create({
        to: operator.phone,
        from: process.env.TWILIO_PHONE_NUMBER,
        body: template.sms,
      });
      messageId = message.sid;
      sent = true;
    } else if (method === 'whatsapp' && operator.whatsappId) {
      const message = await twilioClient.messages.create({
        to: `whatsapp:${operator.whatsappId}`,
        from: `whatsapp:${process.env.TWILIO_WHATSAPP_NUMBER}`,
        body: template.whatsapp,
      });
      messageId = message.sid;
      sent = true;
    } else if (method === 'email') {
      // Use Firebase email or SendGrid
      // For now, assume Firebase email is configured
      await admin
        .firestore()
        .collection('mail')
        .add({
          to: operator.email || 'supervisor@lab.com',
          message: {
            subject: template.email.subject,
            html: template.email.body,
          },
        });
      messageId = `email-${Date.now()}`;
      sent = true;
    } else {
      throw new Error(`Unsupported method ${method} or missing contact info`);
    }

    // Mark as sent in Firestore
    if (queueId) {
      await EscalationService.markEscalationSent(labId, criticoId, queueId, method, messageId);
    }

    return { sent: true, messageId, sentAt: Date.now() };
  } catch (err: any) {
    error = err.message || 'Unknown error';
    console.error(`Notification send failed:`, err);

    // Mark as failed in Firestore (will be retried by scheduled function)
    if (queueId) {
      await EscalationService.markEscalationFailed(labId, criticoId, queueId, method, error);
    }

    throw new HttpsError('internal', `Failed to send ${method} notification: ${error}`);
  }
});
```

- Both functions in `southamerica-east1` region

- Idempotency: messageId stored, allows re-sending safely (Twilio deduplicates by message content hash)
- Error backoff: scheduled function retries up to 3 times with exponential delay

**Function: monitorCriticoSLA**

- Trigger: Cloud Scheduler, runs every 1 minute
- Logic:
  1. Query all criticos with state='ALERTADO' AND escalation.triggeredAt < (now - 5 minutes)
  2. For each SLA breach:
     - Fetch lab settings (clinical supervisor email)
     - Send email: "SLA BREACH: Result still unacknowledged after 5 min"
     - Log to criticos/{resultId}/slaBreachLog
     - Update escalationQueue status: 'sla_breach'
  3. Return: { checked: number, breached: number, notified: number, errors: string[] }

```typescript
// functions/src/scheduled/monitorCriticoSLA.ts
import * as admin from 'firebase-admin';
import * as functions from 'firebase-functions';

export const monitorCriticoSLA = functions
  .region('southamerica-east1')
  .pubsub
  .schedule('every 1 minutes')
  .onRun(async (context) => {
    const db = admin.firestore();
    const now = Date.now();
    const slaTargetMs = 5 * 60 * 1000; // 5 minutes

    try {
      // Get all labs
      const labsSnap = await db.collection('labs').get();
      let totalChecked = 0;
      let totalBreached = 0;
      let totalNotified = 0;

      for (const labDoc of labsSnap.docs) {
        const labId = labDoc.id;
        const labSettings = labDoc.data().settings || {};

        // Query ALERTADO criticos
        const criticosSnap = await db
          .collection(`labs/${labId}/criticos`)
          .where('state', '==', 'ALERTADO')
          .where('escalation.triggeredAt', '<', now - slaTargetMs)
          .get();

        totalChecked += criticosSnap.size;

        for (const criticoDoc of criticosSnap.docs) {
          const critico = criticoDoc.data();

          // Skip if already breached (logged in slaBreachLog)
          if (critico.slaBreach === true) continue;

          totalBreached++;

          // Send email to clinical supervisor
          const supervisorEmail = labSettings.adminEmails?.[0] || 'admin@lab.com';
          await db.collection('mail').add({
            to: supervisorEmail,
            message: {
              subject: `[SLA BREACH] Resultado crítico não reconhecido - ${critico.paciente.nome}`,
              html: `
<html>
<body>
  <h2 style="color: red;">⚠️ SLA BREACH - RESULTADO NÃO RECONHECIDO</h2>
  <p><strong>Paciente:</strong> ${critico.paciente.nome}</p>
  <p><strong>Análise:</strong> ${critico.analyte.nome}</p>
  <p><strong>Valor:</strong> ${critico.analyte.valor} ${critico.analyte.unidade}</p>
  <p><strong>Tempo sem resposta:</strong> ${Math.round((now - critico.escalation.triggeredAt) / 1000 / 60)} minutos</p>
  <p><strong>Objetivo de SLA:</strong> 5 minutos</p>
  <p>Verifique se o operador on-call recebeu o SMS/WhatsApp ou tome medidas imediatas.</p>
</body>
</html>
              `,
            },
          });

          // Log to slaBreachLog subcollection
          await criticoDoc.ref.collection('slaBreachLog').add({
            breachedAt: now,
            elapsedMs: now - critico.escalation.triggeredAt,
            targetMs: slaTargetMs,
            notifiedEmail: supervisorEmail,
          });

          // Mark critico as SLA breached
          await criticoDoc.ref.update({
            slaBreach: true,
            slaBreach At: now,
          });

          totalNotified++;
        }
      }

      console.log(`SLA Monitor: checked ${totalChecked}, breached ${totalBreached}, notified ${totalNotified}`);
      return { checked: totalChecked, breached: totalBreached, notified: totalNotified };
    } catch (err: any) {
      console.error('SLA Monitor error:', err);
      return { error: err.message };
    }
  });
```

import { CriticoDocument } from '../types/machine';
import { useCriticosStateMachine } from '../hooks/useCriticosStateMachine';
import { CriticosAcknowledgmentModal } from './CriticosAcknowledgmentModal';

interface CriticosAlertPanelProps {
  labId: string;
  operatorId: string;
  operatorRole: 'lab-tech' | 'qa-manager' | 'clinical' | 'admin';
}

export const CriticosAlertPanel: React.FC<CriticosAlertPanelProps> = ({
  labId,
  operatorId,
  operatorRole,
}) => {
  const [alertados, setAlertados] = useState<CriticoDocument[]>([]);
  const [selectedCritico, setSelectedCritico] = useState<CriticoDocument | null>(null);
  const [showAckModal, setShowAckModal] = useState(false);

  // Subscribe to ALERTADO criticos
  useEffect(() => {

      collection(getFirestore(), `labs/${labId}/criticos`),
      where('state', '==', 'ALERTADO'),
      orderBy('escalation.triggeredAt', 'desc'),
      limit(10)
    );

    const unsubscribe = onSnapshot(q, (snap) => {
      const docs = snap.docs.map(doc => ({
        id: doc.id,
        ...doc.data(),
      })) as CriticoDocument[];
      setAlertados(docs);

    setSelectedCritico(critico);
    setShowAckModal(true);

    if (seconds < 60) return `${seconds}s`;
    const minutes = Math.floor(seconds / 60);
    return `${minutes}m ${seconds % 60}s`;

    if (elapsed < sla) return 'met';
    if (elapsed < sla + 2 * 60 * 1000) return 'warning'; // 5–7 min
    return 'breach';

  if (alertados.length === 0) {
    return null; // Don't show panel if no alerts

            {alertados.length} RESULTADOS CRÍTICOS
          </h3>
          <button
            className="text-white hover:bg-red-700 rounded p-1"
            aria-label="Fechar painel de alertas"
          >
            ✕
          </button>
        </div>

        <div
          className="space-y-2 max-h-96 overflow-y-auto"
          role="region"
          aria-label="Resultados críticos pendentes"
          aria-live="polite"
        >
          {alertados.map(critico => {
            const slaStatus = getSLAStatus(critico);
            const timeAgo = critico.escalation
              ? formatTimeAgo(critico.escalation.triggeredAt)
              : '—';

            return (
              <div
                key={critico.id}
                className={`p-3 rounded-lg bg-opacity-20 border border-white border-opacity-30 ${
                  slaStatus === 'breach' ? 'bg-white' : 'bg-white'
                }`}
              >
                <div className="flex justify-between items-start mb-2">
                  <div className="flex-1">
                    <p className="font-semibold text-white">
                      {critico.paciente.nome}
                    </p>
                    <p className="text-white text-sm opacity-90">
                      {critico.analyte.nome}: {critico.analyte.valor} {critico.analyte.unidade}
                    </p>
                    <p className="text-white text-xs opacity-75">
                      Intervalo crítico: {critico.threshold.criticalMin || '—'} a {critico.threshold.criticalMax || '—'}
                    </p>
                  </div>
                  <div className="text-right">
                    <p
                      className={`text-xs font-bold ${
                        slaStatus === 'met'
                          ? 'text-green-300'
                          : slaStatus === 'warning'
                          ? 'text-yellow-300'
                          : 'text-red-200'
                      }`}
                    >
                      {timeAgo}
                    </p>
                  </div>
                </div>

                <button
                  onClick={() => handleAcknowledge(critico)}
                  className="w-full bg-white text-red-600 hover:bg-gray-100 font-bold py-2 px-3 rounded transition-colors"
                >
                  ✓ Confirmar Recebimento
                </button>
              </div>
            );
          })}
        </div>
      </div>

      {showAckModal && selectedCritico && (
        <CriticosAcknowledgmentModal
          critico={selectedCritico}
          operatorId={operatorId}
          operatorRole={operatorRole}
          onClose={() => setShowAckModal(false)}
          onSuccess={() => {
            setShowAckModal(false);
            setSelectedCritico(null);
          }}
        />
      )}
    </div>
  );
};
