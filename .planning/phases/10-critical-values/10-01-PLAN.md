```typescript
/**
 * Critical Value State Machine
 * States: NORMAL → CRITICO (auto) → ALERTADO (escalation) → RESOLVIDO (acknowledged)
 * RDC 978 Arts. 184–191 compliance
 */

// ============ States ============
export type CriticoState = 'NORMAL' | 'CRITICO' | 'ALERTADO' | 'RESOLVIDO';

export interface CriticoStateInfo {
  state: CriticoState;
  enteredAt: number; // Unix timestamp
  operatorId?: string; // Who triggered transition
  reason?: string; // Why transition happened
}

// ============ Events (Transition Triggers) ============
export type CriticoEvent =
  | { type: 'EVALUATE_THRESHOLD'; value: number; analyte: string; labId: string }
  | { type: 'ESCALATE'; operatorId: string; reason: string }
  | { type: 'ACKNOWLEDGE'; operatorId: string; acknowledgedAt: number }
  | { type: 'RESET'; operatorId: string; reason: string }; // Admin-only revert (rare)

// ============ Thresholds ============
export interface AnalyteThreshold {
  analyteId: string;
  analyteName: string;
  unit: string;
  normalMin: number;
  normalMax: number;
  criticalMin?: number; // Below this = critical low
  criticalMax?: number; // Above this = critical high
  severity: 'info' | 'warning' | 'critical'; // Always 'critical' for this module
  source: 'lab-setting' | 'default'; // Lab-customized or default reference range
}

export interface ThresholdEvaluationResult {
  isOutOfRange: boolean; // true if value < criticalMin OR > criticalMax
  violation: 'low' | 'high' | null; // Which bound violated
  normalRange: { min: number; max: number };
  criticalRange: { min?: number; max?: number };
  evaluatedAt: number;
}

// ============ Critico Document (Firestore) ============
export interface CriticoDocument {
  id: string; // resultId
  labId: string;
  resultId: string; // Back-ref to bioquimica result
  paciente: {
    cpf: string; // Hashed in queries
    nome: string;
  };
  analyte: {
    id: string;
    nome: string;
    valor: number; // The measured value
    unidade: string;
  };
  threshold: AnalyteThreshold; // Snapshot of threshold at time of evaluation
  state: CriticoState;
  stateHistory: CriticoStateInfo[]; // Immutable log of all transitions
  
  // Escalation metadata (populated when state=ALERTADO)
  escalation?: {
    triggeredAt: number;
    operatorId: string;
    method: 'sms' | 'whatsapp' | 'email'; // Phase 10 Plan 02
    messageId?: string; // For delivery tracking
  };
  
  // Resolution metadata (populated when state=RESOLVIDO)
  resolution?: {
    acknowledgedAt: number;
    acknowledgedBy: string; // Operator ID
    slaMetMs?: number; // Time from CRITICO→ALERTADO to RESOLVIDO (for SLA tracking)
  };

  criadoEm: number;
  deletadoEm?: number; // Soft delete only (RN-06)
}

// ============ Transition Guards + Actions ============
export interface TransitionResult {
  allowed: boolean;
  reason?: string; // Why transition was rejected
  newState?: CriticoState;
  sideEffects?: Array<{
    type: 'log_audit' | 'queue_escalation' | 'set_sla_timer';
    payload: any;
  }>;
}

export interface TransitionContext {
  currentState: CriticoState;
  event: CriticoEvent;
  operator: { id: string; role: 'lab-tech' | 'qa-manager' | 'clinical' | 'admin' };
  timestamp: number;
}

// ============ Lab Settings Threshold Config ============
export interface AnalyteCriticalConfig {
  analyteId: string;
  analyteName: string;
  unit: string;
  normalMin: number;
  normalMax: number;
  criticalMin?: number; // Lab-specific override
  criticalMax?: number; // Lab-specific override
  enabled: boolean; // Can disable critical detection per analyte
}

export interface LabCriticosSettings {
  labId: string;
  enabled: boolean; // Master switch for entire feature
  analyteThresholds: AnalyteCriticalConfig[];
  slaThresholdMs: number; // Target SLA for acknowledgment (5 min = 300000ms)
  defaultNotificationMethods: Array<'sms' | 'whatsapp' | 'email'>;
  adminEmails: string[]; // Who to notify on critical
  onCallOperators: Array<{ operatorId: string; phone?: string; whatsappId?: string }>;
  updatedAt: number;
  updatedBy: string;
}
```

```typescript
import { CriticoState, CriticoEvent, TransitionContext, TransitionResult, CriticoDocument } from '../types/machine';

/**
 * Pure state machine logic: no side effects, no async I/O
 * Guards enforce RDC 978 Art. 191 (immutability of critical detection)
 */

export class CriticosMachine {
  /**
   * Main transition function: evaluates whether event is allowed given current state
   * Returns TransitionResult with new state (if allowed) + side effects to queue
   */
  static async transition(
    context: TransitionContext,
    current: CriticoDocument
  ): Promise<TransitionResult> {
    const { event, currentState, operator, timestamp } = context;

    // ========== NORMAL State ==========
    if (current.state === 'NORMAL') {
      if (event.type === 'EVALUATE_THRESHOLD') {
        const { value, analyte, labId } = event;
        // If threshold breach detected, auto-transition to CRITICO
        // (threshold evaluation done by criticosThresholdService)
        // This event is only fired if breach detected, so:
        return {
          allowed: true,
          newState: 'CRITICO',
          sideEffects: [
            {
              type: 'log_audit',
              payload: {
                from: 'NORMAL',
                to: 'CRITICO',
                value,
                analyte,
                timestamp,
                reason: 'automatic_threshold_breach',
              },
            },
          ],
        };
      }
    }

    // ========== CRITICO State (Immutable per RDC 978 Art. 191) ==========
    if (current.state === 'CRITICO') {
      // CRITICAL: Cannot revert to NORMAL (even if result edited)
      if (event.type === 'EVALUATE_THRESHOLD') {
        return {
          allowed: false,
          reason: 'RDC 978 Art. 191: Critical detection is immutable. Once marked CRITICO, cannot revert.',
        };
      }

      // Can only escalate to ALERTADO (human acknowledges need for escalation)
      if (event.type === 'ESCALATE') {
        // Guard: only QA manager or clinical role can escalate
        if (!['qa-manager', 'clinical', 'admin'].includes(operator.role)) {
          return {
            allowed: false,
            reason: 'Only QA manager, clinical staff, or admin can escalate critical results.',
          };
        }

        // Guard: escalate reason must be non-empty
        if (!event.reason || event.reason.trim().length === 0) {
          return {
            allowed: false,
            reason: 'Escalation reason required (RDC 978 Art. 184 communication trail).',
          };
        }

        return {
          allowed: true,
          newState: 'ALERTADO',
          sideEffects: [
            {
              type: 'log_audit',
              payload: {
                from: 'CRITICO',
                to: 'ALERTADO',
                operator: operator.id,
                reason: event.reason,
                timestamp,
              },
            },
            {
              type: 'queue_escalation',
              payload: {
                resultId: current.id,
                labId: current.labId,
                analyte: current.analyte.nome,
                value: current.analyte.valor,
                methods: ['sms', 'whatsapp', 'email'], // Phase 10 Plan 02 configures
              },
            },
            {
              type: 'set_sla_timer',
              payload: {
                resultId: current.id,
                targetMs: 300000, // 5 minutes per lab setting
                startTime: timestamp,
              },
            },
          ],
        };
      }
    }

    // ========== ALERTADO State ==========
    if (current.state === 'ALERTADO') {
      // Can transition to RESOLVIDO (operator acknowledges)
      if (event.type === 'ACKNOWLEDGE') {
        // Guard: must be an operator (not just any user)
        if (!['lab-tech', 'qa-manager', 'clinical', 'admin'].includes(operator.role)) {
          return {
            allowed: false,
            reason: 'Only lab staff can acknowledge critical results.',
          };
        }

        const slaMetMs = event.acknowledgedAt - current.escalation!.triggeredAt;
        const slaThreshold = 300000; // 5 minutes (Phase 5.1 setting)

        return {
          allowed: true,
          newState: 'RESOLVIDO',
          sideEffects: [
            {
              type: 'log_audit',
              payload: {
                from: 'ALERTADO',
                to: 'RESOLVIDO',
                operator: event.operatorId,
                acknowledgedAt: event.acknowledgedAt,
                slaMetMs,
                slaTarget: slaThreshold,
                slaBreach: slaMetMs > slaThreshold,
                timestamp,
              },
            },
          ],
        };
      }

      // Cannot go back to CRITICO
      if (event.type === 'EVALUATE_THRESHOLD') {
        return {
          allowed: false,
          reason: 'Already escalated. Escalation cannot be cancelled by re-evaluation.',
        };
      }
    }

    // ========== RESOLVIDO State (Terminal) ==========
    if (current.state === 'RESOLVIDO') {
      // No transitions allowed from RESOLVIDO (unless admin RESET, which is rare)
      if (event.type === 'RESET' && operator.role === 'admin') {
        return {
          allowed: true,
          newState: 'NORMAL',
          sideEffects: [
            {
              type: 'log_audit',
              payload: {
                from: 'RESOLVIDO',
                to: 'NORMAL',
                operator: operator.id,
                reason: event.reason,
                timestamp,
                warning: 'ADMIN RESET: Manual override for data correction only',
              },
            },
          ],
        };
      }

      return {
        allowed: false,
        reason: 'Result is resolved. No further state changes allowed.',
      };
    }

    // Fallback: unknown state
    return {
      allowed: false,
      reason: `Unknown state: ${current.state}`,
    };
  }

  /**
   * Helper: Check if CRITICO→ALERTADO transition happened within SLA window
   */
  static checkSLA(critico: CriticoDocument, targetMs: number): {
    slaStatus: 'met' | 'breached' | 'pending';
    elapsedMs?: number;
    message: string;
  } {
    if (critico.state === 'NORMAL' || critico.state === 'CRITICO') {
      return { slaStatus: 'pending', message: 'Not yet escalated' };
    }

    if (!critico.escalation || !critico.resolution) {
      return { slaStatus: 'pending', message: 'Escalation or resolution incomplete' };
    }

    const elapsedMs = critico.resolution.acknowledgedAt - critico.escalation.triggeredAt;
    const breached = elapsedMs > targetMs;

    return {
      slaStatus: breached ? 'breached' : 'met',
      elapsedMs,
      message: `Acknowledged in ${Math.round(elapsedMs / 1000)}s (target: ${Math.round(targetMs / 1000)}s)`,
    };
  }

  /**
   * Helper: Retrieve full transition history (audit trail)
   */
  static getTransitionHistory(critico: CriticoDocument): Array<{
    from: CriticoState;
    to: CriticoState;
    at: number;
    by: string;
    reason: string;
  }> {
    return critico.stateHistory.map((info, idx) => ({
      from: idx === 0 ? 'NORMAL' : critico.stateHistory[idx - 1].state,
      to: info.state,
      at: info.enteredAt,
      by: info.operatorId || 'system',
      reason: info.reason || 'automatic',
    }));
  }
}
```

```typescript
/**
 * Threshold Service: Looks up lab-specific critical ranges + evaluates result values
 * Per-analyte ranges stored in labSettings, fallback to defaults
 */

export class CriticosThresholdService {
  /**
   * Get threshold config for a single analyte (lab-specific or default)
   */
  static async getAnalyteThreshold(
    labId: string,
    analyteId: string
  ): Promise<AnalyteThreshold | null> {
    try {
      // Fetch lab settings
      const labSettingsRef = doc(getFirestore(), `labs/${labId}/labSettings`, 'config');
      const labSettingsSnap = await getDoc(labSettingsRef);

      if (!labSettingsSnap.exists()) {
        console.warn(`No labSettings found for lab ${labId}. Using defaults.`);
        return this.getDefaultThreshold(analyteId);
      }

      const labSettings = labSettingsSnap.data();
      const analyteConfig = labSettings.analyteThresholds?.find(
        (a: AnalyteCriticalConfig) => a.analyteId === analyteId
      );

      if (!analyteConfig) {
        return this.getDefaultThreshold(analyteId);
      }

      // Build threshold from lab config
      const threshold: AnalyteThreshold = {
        analyteId: analyteConfig.analyteId,
        analyteName: analyteConfig.analyteName,
        unit: analyteConfig.unit,
        normalMin: analyteConfig.normalMin,
        normalMax: analyteConfig.normalMax,
        criticalMin: analyteConfig.criticalMin,
        criticalMax: analyteConfig.criticalMax,
        severity: 'critical',
        source: 'lab-setting',
      };

      return threshold;
    } catch (err: any) {
      console.error(`Error fetching threshold for analyte ${analyteId}:`, err);
      return null;
    }
  }

  /**
   * Evaluate if a result value is out of critical range
   * Returns null if no critical thresholds defined (feature disabled for this analyte)
   */
  static evaluateThreshold(
    value: number,
    threshold: AnalyteThreshold
  ): ThresholdEvaluationResult {
    const isLowBreach = threshold.criticalMin !== undefined && value < threshold.criticalMin;
    const isHighBreach = threshold.criticalMax !== undefined && value > threshold.criticalMax;
    const isOutOfRange = isLowBreach || isHighBreach;

    return {
      isOutOfRange,
      violation: isLowBreach ? 'low' : isHighBreach ? 'high' : null,
      normalRange: { min: threshold.normalMin, max: threshold.normalMax },
      criticalRange: {
        min: threshold.criticalMin,
        max: threshold.criticalMax,
      },
      evaluatedAt: Date.now(),
    };
  }

   * Helper: Get all thresholds for a lab (for UI configuration screens)
   */

    try {

    }
  }
```

```typescript
// functions/src/callables/criticos/transitionCritico.ts
import * as admin from 'firebase-admin';
import { CriticosMachine } from '../../shared/criticos-machine';
import { CriticoEvent } from '../../shared/types';

export const transitionCritico = onCall(
  { region: 'southamerica-east1' },
  async (request) => {
    const { resultId, labId, event, operator } = request.data;

    // Auth: operator.id must match request.auth.uid
    if (operator.id !== request.auth?.uid) {
      throw new HttpsError('permission-denied', 'Operator ID mismatch');
    }

    // Load current critico
    const criticoRef = admin.firestore()
      .doc(`labs/${labId}/criticos/${resultId}`);
    const criticoSnap = await criticoRef.get();

    if (!criticoSnap.exists) {
      throw new HttpsError('not-found', `Critical result ${resultId} not found`);
    }

    const current = criticoSnap.data();

    // Evaluate transition
    const transitionResult = await CriticosMachine.transition(
      { event, currentState: current.state, operator, timestamp: Date.now() },
      current
    );

    if (!transitionResult.allowed) {
      throw new HttpsError('failed-precondition', transitionResult.reason);
    }

    // Write atomic batch
    const batch = admin.firestore().batch();
    const timestamp = Date.now();

    // 1. Update critico doc
    batch.update(criticoRef, {
      state: transitionResult.newState,
      stateHistory: admin.firestore.FieldValue.arrayUnion({
        state: transitionResult.newState,
        enteredAt: timestamp,
        operatorId: operator.id,
        reason: event.reason || 'automatic',
      }),
      ...(transitionResult.newState === 'ALERTADO' && {
        escalation: {
          triggeredAt: timestamp,
          operatorId: operator.id,
          method: event.method || 'sms',
        },
      }),
      ...(transitionResult.newState === 'RESOLVIDO' && {
        resolution: {
          acknowledgedAt: timestamp,
          acknowledgedBy: operator.id,
        },
      }),
    });

    // 2. Append to audit log
    batch.set(
      criticoRef.collection('transitionLog').doc(),
      {
        from: current.state,
        to: transitionResult.newState,
        operator: operator.id,
        reason: event.reason || 'automatic',
        timestamp,
        sideEffects: transitionResult.sideEffects || [],
      },
      { merge: true }
    );

    // 3. If escalation, queue notifications (Phase 10 Plan 02 consumes)
    if (transitionResult.sideEffects?.some(s => s.type === 'queue_escalation')) {
      batch.set(
        criticoRef.collection('escalationQueue').doc(),
        {
          status: 'pending',
          createdAt: timestamp,
          payload: transitionResult.sideEffects[0].payload,
        }
      );
    }

    await batch.commit();

    return {
      allowed: true,
      newState: transitionResult.newState,
      appliedAt: timestamp,
    };
  }
);
```

```typescript
import { useState, useCallback } from 'react';
import { httpsCallable } from 'firebase/functions';
import { getFirestore } from 'firebase/firestore';
import { CriticoDocument, CriticoEvent, CriticoState } from '../types/machine';

export interface UseCriticosStateMachineReturn {
  critico: CriticoDocument | null;
  state: CriticoState | null;
  loading: boolean;
  error: string | null;

  // Actions
  transitionTo: (event: CriticoEvent) => Promise<void>;
  escalate: (reason: string) => Promise<void>;
  acknowledge: () => Promise<void>;
  getStateInfo: () => { state: CriticoState; enteredAt: number; reason?: string } | null;
}

export function useCriticosStateMachine(
  initialCritico: CriticoDocument | null,
  operatorId: string,
  operatorRole: 'lab-tech' | 'qa-manager' | 'clinical' | 'admin'
): UseCriticosStateMachineReturn {
  const [critico, setCritico] = useState<CriticoDocument | null>(initialCritico);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const transitionTo = useCallback(
    async (event: CriticoEvent) => {
      if (!critico) {
        setError('No critico loaded');
        return;
      }

      setLoading(true);
      setError(null);

      try {
        const transitionCriticoCallable = httpsCallable(
          getFirestore(),
          'transitionCritico'
        );

        const result = await transitionCrititicoCallable({
          resultId: critico.id,
          labId: critico.labId,
          event,
          operator: { id: operatorId, role: operatorRole },
        });

        // Update local state with new state
        setCritico(prev => ({
          ...prev!,
          state: result.data.newState,
          stateHistory: [
            ...prev!.stateHistory,
            {
              state: result.data.newState,
              enteredAt: result.data.appliedAt,
              operatorId,
            },
          ],
        }));
      } catch (err: any) {
        const message = err.code === 'failed-precondition'
          ? `Transition not allowed: ${err.message}`
          : err.message || 'Transition failed';
        setError(message);
      } finally {
        setLoading(false);
      }
    },
    [critico, operatorId, operatorRole]
  );

  const escalate = useCallback(
    async (reason: string) => {
      await transitionTo({
        type: 'ESCALATE',
        operatorId,
        reason,
      });
    },
    [transitionTo, operatorId]
  );

  const acknowledge = useCallback(
    async () => {
      await transitionTo({
        type: 'ACKNOWLEDGE',
        operatorId,
        acknowledgedAt: Date.now(),
      });
    },
    [transitionTo, operatorId]
  );

  const getStateInfo = useCallback(() => {
    if (!critico || critico.stateHistory.length === 0) return null;
    const latest = critico.stateHistory[critico.stateHistory.length - 1];
    return {
      state: latest.state,
      enteredAt: latest.enteredAt,
      reason: latest.reason,
    };
  }, [critico]);

  return {
    critico,
    state: critico?.state || null,
    loading,
    error,
    transitionTo,
    escalate,
    acknowledge,
    getStateInfo,
  };
}
```
