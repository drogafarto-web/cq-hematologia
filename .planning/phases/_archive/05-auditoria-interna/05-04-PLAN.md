### PDF Generation Stack

**Decision:** Use `jsPDF` + `html2canvas` on client-side OR `puppeteer` + `pdfkit` server-side.

**Client-side (jsPDF):**

- Pros: Faster, simpler, no puppeteer overhead
- Cons: Limited styling, images must be embedded as base64, layout control harder

**Server-side (puppeteer):**

- Pros: Full HTML/CSS control, professional layout, signatures rendered
- Cons: Slower, puppeteer ~50MB, cold start time

**Recommendation for MVP:** Use client-side `jsPDF` + `html2canvas` (keeps bundle reasonable per performance.md). Phase 05-05+ can upgrade to puppeteer if PDF quality needs improvement.

Actually, check `.claude/rules/performance.md` — puppeteer is functions-only (never client). So use server-side with puppeteer.

Revised: **Server-side puppeteer** (correct per project rules). Cold start acceptable for report generation (not user-blocking, <10s typical).

import { onCall, HttpsError } from 'firebase-functions/v2/https';
import * as admin from 'firebase-admin';
import * as puppeteer from 'puppeteer';
import type { Auditoria, Sessao, ChecklistItem, Achado } from './types';

const db = admin.firestore();
const storage = admin.storage();

export const generateAuditReportPDF = onCall(
  { region: 'southamerica-east1', timeoutSeconds: 540, memory: '2GB' },
  async (request: any) => {
    if (!request.auth) throw new HttpsError('unauthenticated', 'Auth required');

    const { labId, auditoriaId, sessaoId } = request.data;

    if (!labId || !auditoriaId || !sessaoId) {
      throw new HttpsError('invalid-argument', 'labId, auditoriaId, sessaoId required');
    }

    try {
      // 1. Fetch data
      const audSnap = await db
        .collection(`labs/${labId}/auditorias-internas`)
        .doc(auditoriaId)
        .get();

      if (!audSnap.exists) {
        throw new HttpsError('not-found', 'Auditoria not found');
      }

      const auditoria = audSnap.data() as Auditoria;

      const sessaoSnap = await db
        .collection(`labs/${labId}/auditorias-internas/${auditoriaId}/sessoes`)
        .doc(sessaoId)
        .get();

      if (!sessaoSnap.exists) {
        throw new HttpsError('not-found', 'Sessão not found');
      }

      const sessao = sessaoSnap.data() as Sessao;

      // Fetch all checklist items
      const itemsSnap = await db
        .collection(
          `labs/${labId}/auditorias-internas/${auditoriaId}/sessoes/${sessaoId}/checklist-items`,
        )
        .orderBy('numeroDICQ', 'asc')
        .get();

      const items = itemsSnap.docs.map((doc) => doc.data() as ChecklistItem);

      // Fetch all achados
      const achadosSnap = await db
        .collection(`labs/${labId}/auditorias-internas/${auditoriaId}/sessoes/${sessaoId}/achados`)
        .get();

      const achados = achadosSnap.docs.map((doc) => doc.data() as Achado);

      // Fetch lab info for header
      const labSnap = await db.collection('labs').doc(labId).get();
      const lab = labSnap.data() || {};

      // 2. Generate HTML content
      const htmlContent = generatePDFHTML({
        lab,
        auditoria,
        sessao,
        items,
        achados,
      });

      // 3. Render to PDF using puppeteer
      const browser = await puppeteer.launch({
        headless: 'new',
        args: ['--no-sandbox', '--disable-setuid-sandbox'],
      });

      const page = await browser.newPage();
      await page.setContent(htmlContent, { waitUntil: 'networkidle0' });

      const pdfBuffer = await page.pdf({
        format: 'A4',
        margin: { top: 20, right: 20, bottom: 20, left: 20 },
        printBackground: true,
      });

      await browser.close();

      // 4. Upload to Cloud Storage
      const filename = `audit-${sessaoId}-${Date.now()}.pdf`;
      const filepath = `labs/${labId}/audit-reports/${new Date().getFullYear()}/${filename}`;

      const bucket = storage.bucket();
      const file = bucket.file(filepath);

      await file.save(pdfBuffer, {
        metadata: {
          contentType: 'application/pdf',
          metadata: {
            auditoriaId,
            sessaoId,
            labId,
            generatedAt: new Date().toISOString(),
            generatedBy: request.auth.uid,
          },
          cacheControl: 'public, max-age=3600', // 1 hour
        },
      });

      // 5. Generate signed URL (7-day expiry)
      const [signedUrl] = await file.getSignedUrl({
        version: 'v4',
        action: 'read',
        expires: Date.now() + 7 * 24 * 60 * 60 * 1000, // 7 days
      });

      return {
        success: true,
        url: signedUrl,
        filename,
        filesize: pdfBuffer.length,
      };
    } catch (error: any) {
      if (error instanceof HttpsError) throw error;
      throw new HttpsError('internal', error.message || 'Error generating PDF');
    }
  },
);

```typescript
import { useState } from 'react';
import { functions } from 'src/shared/services/firebase';
import { httpsCallable } from 'firebase/functions';

interface UseAuditReportExportOptions {
  labId: string;
  auditoriaId: string;
  sessaoId: string;
}

export function useAuditReportExport(options: UseAuditReportExportOptions) {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  const generateReport = async () => {
    setLoading(true);
    setError(null);

    try {
      const callable = httpsCallable(functions, 'generateAuditReportPDF');
      const result = await callable({
        labId: options.labId,
        auditoriaId: options.auditoriaId,
        sessaoId: options.sessaoId,
      });

      const data = result.data as { url: string; filename: string; filesize: number };

      // Trigger download
      const a = document.createElement('a');
      a.href = data.url;
      a.download = data.filename || 'audit-report.pdf';
      a.target = '_blank';
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);

      // Toast: "Relatório exportado"
      return { success: true, filename: data.filename };
    } catch (err: any) {
      const errorMsg = err.message || 'Erro ao gerar relatório';
      setError(new Error(errorMsg));
      throw err;
    } finally {
      setLoading(false);
    }
  };

  return {
    generateReport,
    loading,
    error,
  };
}

import { describe, it, expect, beforeAll, afterAll } from '@jest/globals';
import { initializeApp, deleteApp } from 'firebase/app';
import { connectAuthEmulator, getAuth, signInAnonymously } from 'firebase/auth';
import {
  connectFirestoreEmulator,
  getFirestore,
  collection,
  addDoc,
  doc,
  getDoc,
} from 'firebase/firestore';
import { connectFunctionsEmulator, getFunctions, httpsCallable } from 'firebase/functions';

describe('Auditoria Interna E2E', () => {
  let app: any, auth: any, db: any, functions: any;
  let labId: string, auditoriaId: string, sessaoId: string;

  beforeAll(async () => {
    // Initialize emulator
    app = initializeApp({
      apiKey: 'test',
      projectId: 'hmatologia2-test',
    });

    auth = getAuth(app);
    connectAuthEmulator(auth, 'http://localhost:9099');

    db = getFirestore(app);
    connectFirestoreEmulator(db, 'localhost', 8080);

    functions = getFunctions(app, 'southamerica-east1');
    connectFunctionsEmulator(functions, 'localhost', 5001);

    // Sign in test user
    const userCred = await signInAnonymously(auth);
    console.log('Signed in as:', userCred.user.uid);
  });

  afterAll(async () => {
    await deleteApp(app);
  });

  it('should create annual audit plan', async () => {
    labId = 'lab-test-' + Date.now();

    const createAuditoria = httpsCallable(functions, 'createAuditoria');
    const result = await createAuditoria({
      labId,
      codigo: 'AUD-2026-001',
      titulo: 'Auditoria Interna DICQ 2026',
      tipo: 'interna',
      escopo: 'Todos os processos DICQ 4.x + 5.x',
      agendadaPara: new Date('2026-06-01').toISOString(),
    });

    expect(result.data).toHaveProperty('auditoriaId');
    auditoriaId = (result.data as any).auditoriaId;
    console.log('Created auditoria:', auditoriaId);
  });

  it('should install checklist template', async () => {
    const installTemplate = httpsCallable(functions, 'installChecklistTemplate');
    const result = await installTemplate({
      labId,
      auditoriaId,
      templateId: 'dicq-4-3-rdc-978-v1',
    });

    expect(result.data).toHaveProperty('sessaoId');
    expect((result.data as any).itemsCreated).toBeGreaterThan(100);
    sessaoId = (result.data as any).sessaoId;
    console.log(
      'Installed template:',
      sessaoId,
      'with',
      (result.data as any).itemsCreated,
      'items',
    );
  });

  it('should execute checklist items (mark conforme/não conforme)', async () => {
    // Fetch checklist items
    const itemsSnap = await getDocs(
      collection(
        db,
        `labs/${labId}/auditorias-internas/${auditoriaId}/sessoes/${sessaoId}/checklist-items`,
      ),
    );

    expect(itemsSnap.size).toBeGreaterThan(100);

    // Prepare responses (first 10 conforme, next 5 não conforme)
    const responses: any[] = [];
    itemsSnap.docs.forEach((doc, idx) => {
      if (idx < 10) {
        responses.push({
          itemId: doc.id,
          resposta: 'conforme',
        });
      } else if (idx < 15) {
        responses.push({
          itemId: doc.id,
          resposta: 'não-conforme',
          severidade: idx === 13 || idx === 14 ? 'grave' : 'leve',
        });
      }
    });

    const updateResponses = httpsCallable(functions, 'updateChecklistResponses');
    const result = await updateResponses({
      labId,
      auditoriaId,
      sessaoId,
      responses,
    });

    expect(result.data).toHaveProperty('itemsUpdated');
    expect((result.data as any).stats.conforme).toEqual(10);
    expect((result.data as any).stats.naoConforme).toEqual(5);
    console.log('Updated responses:', result.data);
  });

  it('should register achado for critical finding', async () => {
    // Get first "não conforme" item
    const itemsSnap = await getDocs(
      query(
        collection(
          db,
          `labs/${labId}/auditorias-internas/${auditoriaId}/sessoes/${sessaoId}/checklist-items`,
        ),
        where('resposta', '==', 'não-conforme'),
        limit(1),
      ),
    );

    if (itemsSnap.empty) {
      console.log('No não conforme items, skipping achado test');
      return;
    }

    const itemId = itemsSnap.docs[0].id;

    const registerAchado = httpsCallable(functions, 'registerAchado');
    const result = await registerAchado({
      labId,
      auditoriaId,
      sessaoId,
      checklistItemId: itemId,
      descricao: 'Procedimento POP-001 não está vigente, versão anterior em uso',
      severidade: 'grave',
      evidencia: 'data:image/png;base64,iVBORw0KGgoAAAANS...',
    });

    expect(result.data).toHaveProperty('achadoId');
    expect((result.data as any).ncCreated).toBe(true);
    console.log('Created achado:', result.data);
  });

  it('should generate PDF report', async () => {
    const generatePDF = httpsCallable(functions, 'generateAuditReportPDF');
    const result = await generatePDF({
      labId,
      auditoriaId,
      sessaoId,
    });

    expect(result.data).toHaveProperty('url');
    expect(result.data).toHaveProperty('filename');
    expect((result.data as any).filesize).toBeLessThan(10 * 1024 * 1024); // <10MB

    console.log(
      'Generated PDF:',
      (result.data as any).filename,
      'Size:',
      (result.data as any).filesize,
      'bytes',
    );
  });
});
```
