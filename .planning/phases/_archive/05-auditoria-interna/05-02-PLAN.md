```typescript
interface AchadoFormProps {
  itemId: string;
  severity: string;
  onSave: () => void;
}

export function AchadoForm({ itemId, severity, onSave }: AchadoFormProps) {
  const labId = useActiveLabId();
  const { sessao } = useSessao(/* from parent context */);
  const { registerAchado } = useAchadoMutation();
  const [descricao, setDescricao] = useState('');
  const [evidencias, setEvidencias] = useState<string[]>([]);
  const [isSaving, setIsSaving] = useState(false);

  const handleFileUpload = async (file: File) => {
    // Upload to Firebase Storage: `labs/{labId}/auditorias-internas/{auditoriaId}/{sessaoId}/{itemId}/{file.name}`
    // For MVP: convert to base64 and store inline (if <5MB)
    // Alternatively: use uploadBytes to Storage + return signed URL
    const reader = new FileReader();
    reader.onload = (e) => {
      const data = e.target?.result as string;
      setEvidencias(prev => [...prev, data]);
    };
    reader.readAsDataURL(file);
  };

  const handleSave = async () => {
    if (!descricao.trim()) {
      alert('Descrição do achado é obrigatória');
      return;
    }

    setIsSaving(true);
    try {
      await registerAchado({
        labId,
        auditoriaId: sessao?.auditoriaId || '',
        sessaoId: sessao?.id || '',
        checklistItemId: itemId,
        descricao,
        severidade,
        evidencia: evidencias[0] || '', // First image only for MVP
      });
      onSave();
    } catch (err) {
      console.error('Error saving achado:', err);
      alert('Erro ao salvar achado');
    } finally {
      setIsSaving(false);
    }
  };

  return (
    <div className="bg-red-900/20 border border-red-500/30 rounded-lg p-4 space-y-3">
      <h4 className="text-sm font-semibold text-red-400">Registrar Achado</h4>

      {/* Description textarea */}
      <textarea
        placeholder="Descreva o achado encontrado..."
        value={descricao}
        onChange={e => setDescricao(e.target.value)}
        className="w-full h-24 bg-white/5 border border-white/10 rounded p-2 text-white placeholder-white/40 focus:outline-none focus:ring-2 focus:ring-red-500"
      />

      {/* Evidence upload */}
      <div className="space-y-2">
        <label className="block text-sm text-white/70">Evidência (foto/documento)</label>
        <input
          type="file"
          accept="image/*,.pdf"
          onChange={e => e.target.files?.[0] && handleFileUpload(e.target.files[0])}
          className="w-full"
        />
        {evidencias.length > 0 && (
          <div className="space-y-2">
            {evidencias.map((ev, i) => (
              <div key={i} className="flex items-center space-x-2">
                {ev.startsWith('data:image') && (
                  <img src={ev} alt={`Evidence ${i}`} className="w-12 h-12 rounded object-cover" />
                )}
                <span className="text-sm text-white/60">Evidência {i + 1}</span>
                <button
                  onClick={() => setEvidencias(prev => prev.filter((_, idx) => idx !== i))}
                  className="text-red-500 hover:text-red-400 text-sm"
                >
                  Remover
                </button>
              </div>
            ))}
          </div>
        )}
      </div>

      {/* Buttons */}
      <div className="flex gap-2">
        <Button
          onClick={handleSave}
          disabled={isSaving || !descricao.trim()}
          className="flex-1 bg-red-600 hover:bg-red-700 h-10"
        >
          {isSaving ? '⏳ Salvando...' : 'Salvar achado'}
        </Button>
        <Button
          onClick={onSave}
          variant="outline"
          className="flex-1 h-10"
        >
          Cancelar
        </Button>
      </div>
    </div>
  );
}
```
