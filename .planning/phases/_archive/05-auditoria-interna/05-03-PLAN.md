### Audit Trail Signing (RDC 978 requirement)

Per root CLAUDE.md + `.claude/rules/firestore-security.md`:

```typescript
interface LogicalSignature {
  hash: string; // SHA-256 of canonical payload (64 chars)
  operatorId: string; // request.auth.uid
  ts: Timestamp; // server-side
}
```

`hash` is computed server-side from canonical JSON (deterministic field order).

Existing helper: `utils/logicalSignature.ts` (from educacao-continuada pattern).

### Multi-tenant Enforcement

Every callable must:

1. Receive `labId` as explicit parameter (never from token)
2. Validate `labId` matches request.auth claim (if present)
3. Check `isActiveMemberOfLab(labId, uid)`

### NC Creation Workflow (AUDI-03)

When `registerAchado` severity >= grave:

1. Call `createNCFromAchado(labId, achado, auditoriaId, sessaoId)`
2. Create NC doc in `/labs/{labId}/naoConformidades/{ncId}`
3. Link achado: set `achado.ncId = ncId` + `achado.statusNC = 'criada'`
4. Link NC: set `nc.achadoId = achadoId` (back-reference)
5. Atomic batch: both writes together

### Callable Errors

Align with Firebase Cloud Function error codes:

- `unauthenticated` — no auth token
- `permission-denied` — auth valid but no permission
- `invalid-argument` — input validation failed
- `not-found` — resource doesn't exist
- `internal` — server error (don't expose details to client)
- `failed-precondition` — blocking condition (e.g., lab has critical NC blocking operations)

```typescript
export const registerAchado = onCall(
  { region: 'southamerica-east1' },
  async (request: CallableRequest<any>) => {
    // 1. Validate auth
    if (!request.auth) throw new HttpsError('unauthenticated', 'Auth token missing');

    // 2. Validate input
    const input = RegisterAchadoInput.parse(request.data);

    // 3. Check lab membership
    const isMember = await isActiveMemberOfLab(input.labId, request.auth.uid);
    if (!isMember) throw new HttpsError('permission-denied', 'Not a lab member');

    // 4. Check blocking NCs (per ADR 0003)
    const ncCheck = await checkNCs(input.labId, 'auditoria');
    if (ncCheck.blocked) {
      throw new HttpsError('failed-precondition', ncCheck.message);
    }

    try {
      // 5. Generate logical signature
      const canonicalPayload = {
        auditoriaId: input.auditoriaId,
        sessaoId: input.sessaoId,
        checklistItemId: input.checklistItemId,
        descricao: input.descricao,
        severidade: input.severidade,
      };
      const hash = crypto
        .createHash('sha256')
        .update(JSON.stringify(canonicalPayload, Object.keys(canonicalPayload).sort()))
        .digest('hex');

      const assinatura: LogicalSignature = {
        hash,
        operatorId: request.auth.uid,
        ts: admin.firestore.Timestamp.now(),
      };

      // 6. Create achado document
      const achadoRef = db
        .collection(
          `labs/${input.labId}/auditorias-internas/${input.auditoriaId}/sessoes/${input.sessaoId}/achados`,
        )
        .doc();

      const achado: Achado = {
        id: achadoRef.id,
        sessaoId: input.sessaoId,
        labId: input.labId,
        checklistItemId: input.checklistItemId,
        descricao: input.descricao,
        evidencia: input.evidencia || '',
        severidade: input.severidade,
        statusNC:
          input.severidade === 'critica' || input.severidade === 'grave'
            ? 'pendente'
            : 'nao_aplicavel',
        assinatura,
        criadoEm: assinatura.ts,
        criadoPor: request.auth.uid,
        deletadoEm: null,
      };

      // 7. Start batch
      const batch = db.batch();
      batch.set(achadoRef, achado);

      // 8. If severity >= grave, auto-create NC
      if (input.severidade === 'critica' || input.severidade === 'grave') {
        const { ncId } = await createNCFromAchado(
          input.labId,
          achado,
          input.auditoriaId,
          input.sessaoId,
        );

        // Update achado to link NC
        batch.update(achadoRef, {
          ncId,
          statusNC: 'criada',
        });
      }

      await batch.commit();

      return {
        success: true,
        achadoId: achadoRef.id,
        severidade: input.severidade,
        ncCreated: input.severidade === 'critica' || input.severidade === 'grave',
      };
    } catch (error: any) {
      if (error instanceof HttpsError) throw error;
      throw new HttpsError('internal', error.message || 'Error registering achado');
    }
  },
);
```

```typescript
import * as admin from 'firebase-admin';
import type { Achado, LogicalSignature } from './types';

const db = admin.firestore();

export async function createNCFromAchado(
  labId: string,
  achado: Achado,
  auditoriaId: string,
  sessaoId: string,
): Promise<{ ncId: string }> {
  // Map severity → prioridade
  const severidadeMap: Record<string, string> = {
    critica: 'critica',
    grave: 'grave',
    moderada: 'moderada',
  };

  const prioridade = severidadeMap[achado.severidade] || 'leve';

  // Create NC document
  const ncRef = db.collection(`labs/${labId}/naoConformidades`).doc();

  const nc = {
    id: ncRef.id,
    labId,
    titulo: `Achado auditoria: ${achado.descricao.substring(0, 50)}...`,
    descricao: achado.descricao,
    origem: 'auditoria-interna',
    achadoId: achado.id, // back-reference
    auditoriaId, // reference for tracing
    severidade: prioridade,
    status: 'aberta',
    criadoEm: admin.firestore.FieldValue.serverTimestamp(),
    criadoPor: achado.criadoPor,
    deletadoEm: null,
  };

  await ncRef.set(nc);

  return { ncId: ncRef.id };
}
```

```typescript
import { onCall, HttpsError } from 'firebase-functions/v2/https';
import * as admin from 'firebase-admin';
import CHECKLIST_TEMPLATES from '../seeds/checklistTemplates.json';
import type { ChecklistItem, TemplateChecklist } from '../../features/auditoria-interna/types';

const db = admin.firestore();

const InstallTemplateInput = z.object({
  labId: z.string().min(1),
  auditoriaId: z.string().min(1),
  templateId: z.string().min(1),
});

export const installChecklistTemplate = onCall(
  { region: 'southamerica-east1' },
  async (request: CallableRequest<any>) => {
    if (!request.auth) throw new HttpsError('unauthenticated', 'Auth required');

    const input = InstallTemplateInput.parse(request.data);

    // Check membership
    const isMember = await isActiveMemberOfLab(input.labId, request.auth.uid);
    if (!isMember) throw new HttpsError('permission-denied', 'Not a lab member');

    try {
      // Get template
      const template = (CHECKLIST_TEMPLATES as Record<string, TemplateChecklist>)[input.templateId];
      if (!template) {
        throw new HttpsError('not-found', `Template ${input.templateId} not found`);
      }

      // Create sessão document
      const sessaoRef = db
        .collection(`labs/${input.labId}/auditorias-internas/${input.auditoriaId}/sessoes`)
        .doc();

      const sessao = {
        id: sessaoRef.id,
        auditoriaId: input.auditoriaId,
        labId: input.labId,
        auditor: request.auth.uid,
        dataInicio: admin.firestore.FieldValue.serverTimestamp(),
        dataFim: null,
        status: 'planejada',
        totalItens: template.itens.length,
        itensConforme: 0,
        itensNãoConforme: 0,
        itensNA: 0,
        criadoEm: admin.firestore.FieldValue.serverTimestamp(),
        criadoPor: request.auth.uid,
        deletadoEm: null,
      };

      // Batch create: sessão + checklist items (max 500 writes per batch)
      const batch = db.batch();
      batch.set(sessaoRef, sessao);

      let itemsCount = 0;
      for (const templateItem of template.itens) {
        const itemRef = sessaoRef.collection('checklist-items').doc();
        const item: ChecklistItem = {
          id: itemRef.id,
          sessaoId: sessaoRef.id,
          labId: input.labId,
          numeroDICQ: templateItem.numeroDICQ,
          descricao: templateItem.descricao,
          categoria: templateItem.categoria,
          bloco: templateItem.bloco,
          isApplicable: templateItem.isApplicable,
          resposta: null,
          severidade: null,
          observacoes: '',
          criadoEm: admin.firestore.FieldValue.serverTimestamp(),
          criadoPor: request.auth.uid,
        };
        batch.set(itemRef, item);
        itemsCount++;

        // Commit every 400 items (safe margin below 500 limit)
        if (itemsCount % 400 === 0) {
          await batch.commit();
          batch = db.batch();
        }
      }

      // Final commit
      await batch.commit();

      return {
        success: true,
        sessaoId: sessaoRef.id,
        itemsCreated: itemsCount,
      };
    } catch (error: any) {
      if (error instanceof HttpsError) throw error;
      throw new HttpsError('internal', error.message || 'Error installing template');
    }
  },
);
```

```typescript
const UpdateChecklistResponseInput = z.object({
  labId: z.string(),
  auditoriaId: z.string(),
  sessaoId: z.string(),
  responses: z.array(
    z.object({
      itemId: z.string(),
      resposta: z.enum(['conforme', 'não-conforme', 'N/A']),
      severidade: z.enum(['critica', 'grave', 'moderada', 'leve', 'observacao']).optional(),
      observacoes: z.string().optional(),
    }),
  ),
});

export const updateChecklistResponses = onCall(
  { region: 'southamerica-east1' },
  async (request: CallableRequest<any>) => {
    if (!request.auth) throw new HttpsError('unauthenticated', 'Auth required');

    const input = UpdateChecklistResponseInput.parse(request.data);
    const isMember = await isActiveMemberOfLab(input.labId, request.auth.uid);
    if (!isMember) throw new HttpsError('permission-denied', 'Not a lab member');

    try {
      const batch = db.batch();
      let conforme = 0,
        naoConforme = 0,
        na = 0;

      for (const resp of input.responses) {
        const itemRef = db
          .collection(
            `labs/${input.labId}/auditorias-internas/${input.auditoriaId}/sessoes/${input.sessaoId}/checklist-items`,
          )
          .doc(resp.itemId);

        const update: any = {
          resposta: resp.resposta,
          observacoes: resp.observacoes || '',
        };

        if (resp.resposta === 'não-conforme') {
          update.severidade = resp.severidade || 'observacao';
        }

        batch.update(itemRef, update);

        // Count for sessão stats
        if (resp.resposta === 'conforme') conforme++;
        else if (resp.resposta === 'não-conforme') naoConforme++;
        else na++;
      }

      // Update sessão stats
      const sessaoRef = db
        .collection(`labs/${input.labId}/auditorias-internas/${input.auditoriaId}/sessoes`)
        .doc(input.sessaoId);

      batch.update(sessaoRef, {
        itensConforme: conforme,
        itensNãoConforme: naoConforme,
        itensNA: na,
        dataFim: input.responses.length > 0 ? admin.firestore.FieldValue.serverTimestamp() : null,
      });

      await batch.commit();

      return {
        success: true,
        itemsUpdated: input.responses.length,
        stats: { conforme, naoConforme, na },
      };
    } catch (error: any) {
      throw new HttpsError('internal', error.message);
    }
  },
);
```

This callable is invoked when auditor clicks "Finalizar sessão" in SessaoExecucaoPanel.

Per offline-friendly requirement: Offline responses saved in localStorage, batch-synced when network returns.
