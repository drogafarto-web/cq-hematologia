**Create hook `src/hooks/useCIQRuns.ts`:**
Subscribe to Firestore collection `/labs/{labId}/runs` with onSnapshot. Cache in Zustand per Phase 3.2 RESEARCH pattern 2.

```typescript
// hc-quality-mobile/src/hooks/useCIQRuns.ts
import { useEffect, useState } from 'react';
import { collection, onSnapshot, query, where, orderBy, Firestore } from 'firebase/firestore';
import { db } from '../core/firebase';
import { useAuthStore } from '../store/useAuthStore';

export interface CIQRun {
  id: string;
  labId: string;
  equipmentId: string;
  runType: 'CIQ' | 'VALIDATION';
  startedAt: number; // timestamp
  status: 'valid' | 'invalid' | 'pending';
  comments: string;
  operatorId: string;
  createdAt: number;
}

export function useCIQRuns() {
  const labId = useAuthStore((s) => s.activeLabId);
  const [runs, setRuns] = useState<CIQRun[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!labId) {
      setLoading(false);
      return;
    }

    const q = query(
      collection(db, `labs/${labId}/runs`),
      where('deletadoEm', '==', null), // soft delete check (RN-06)
      orderBy('startedAt', 'desc'),
    );

    const unsubscribe = onSnapshot(
      q,
      (snapshot) => {
        const data = snapshot.docs.map((doc) => ({
          ...doc.data(),
          id: doc.id,
        })) as CIQRun[];
        setRuns(data);
        setLoading(false);
        setError(null);
      },
      (err) => {
        setError(err.message);
        setLoading(false);
      },
    );

    return unsubscribe;
  }, [labId]);

  return { runs, loading, error };
}
```

```typescript
  View,
  Text,
  ScrollView,
  ActivityIndicator,
  RefreshControl,
  StyleSheet,
} from 'react-native';
import { useCIQRuns } from '../hooks/useCIQRuns';
import { CIQCard } from '../components/CIQCard';
import { OfflineIndicator } from '../components/OfflineIndicator';

export function CIQScreen({ navigation }: any) {
  const { runs, loading, error } = useCIQRuns();
  const [refreshing, setRefreshing] = React.useState(false);

  const handleRefresh = () => {
    setRefreshing(true);
    // Firestore listener auto-refreshes; simulate refresh UI
    setTimeout(() => setRefreshing(false), 500);
  };

  if (error) {
    return (
      <View style={styles.container}>
        <OfflineIndicator />
        <View style={styles.errorContainer}>
          <Text style={styles.errorText}>Erro ao carregar corridas: {error}</Text>
        </View>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <OfflineIndicator />
      <ScrollView
        refreshControl={
          <RefreshControl refreshing={refreshing} onRefresh={handleRefresh} />
        }
      >
        <View style={styles.content}>
          <Text style={styles.title}>Corridas CIQ</Text>
```

```typescript
// hc-quality-mobile/src/screens/CIQDetailScreen.tsx
import React, { useState } from 'react';
import {
  View,
  Text,
  TextInput,
  TouchableOpacity,
  ActivityIndicator,
  StyleSheet,
  ScrollView,
  Alert,
} from 'react-native';
import { useMobileCallables } from '../hooks/useMobileCallables';
import { useAuthStore } from '../store/useAuthStore';

export function CIQDetailScreen({ route, navigation }: any) {
  const { runId } = route.params as { runId: string };
  const labId = useAuthStore((s) => s.activeLabId);
  const { callFunction } = useMobileCallables();
  const [comments, setComments] = useState('');
  const [loading, setLoading] = useState(false);

  const handleSubmit = async () => {
    if (!comments.trim()) {
      Alert.alert('Erro', 'Adicione um comentário antes de enviar');
      return;
    }

    setLoading(true);
    try {
      await callFunction('submitCIQComment', {
        labId,
        runId,
        comments: comments.trim(),
      });
      Alert.alert('Sucesso', 'Comentário salvo');
      navigation.goBack();
    } catch (err: any) {
      Alert.alert('Erro', err.message || 'Falha ao enviar comentário');
    } finally {
      setLoading(false);
    }
  };

  return (
    <ScrollView style={styles.container}>
      <View style={styles.content}>
        <Text style={styles.label}>Comentários</Text>
        <TextInput
          style={styles.input}
          placeholder="Adicione observações sobre esta corrida..."
          placeholderTextColor="#666"
          multiline
          numberOfLines={6}
          value={comments}
          onChangeText={setComments}
          editable={!loading}
        />
        <TouchableOpacity
          style={[styles.button, loading && styles.buttonDisabled]}
          onPress={handleSubmit}
          disabled={loading}
        >
          {loading ? (
            <ActivityIndicator color="#fff" />
          ) : (
            <Text style={styles.buttonText}>Salvar Comentário</Text>
          )}
        </TouchableOpacity>
      </View>
    </ScrollView>
  );
}
```

```typescript
// hc-quality-mobile/src/hooks/useOpenNCs.ts
import { useEffect, useState } from 'react';
import { collection, onSnapshot, query, where, orderBy, Firestore } from 'firebase/firestore';
import { db } from '../core/firebase';
import { useAuthStore } from '../store/useAuthStore';

export interface NC {
  id: string;
  labId: string;
  status: 'open' | 'investigating' | 'resolved' | 'closed';
  description: string;
  detectadoEm: number; // timestamp
  equipmentId: string;
  moduleId: string;
  createdAt: number;
  updatedAt: number;
  deletadoEm?: number | null;
}

export function useOpenNCs() {
  const labId = useAuthStore((s) => s.activeLabId);
  const [ncs, setNCs] = useState<NC[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!labId) {
      setLoading(false);
      return;
    }

    const q = query(
      collection(db, `labs/${labId}/nc`),
      where('status', 'in', ['open', 'investigating']),
      where('deletadoEm', '==', null),
      orderBy('detectadoEm', 'asc'),
    );

    const unsubscribe = onSnapshot(
      q,
      (snapshot) => {
        const data = snapshot.docs.map((doc) => ({
          ...doc.data(),
          id: doc.id,
        })) as NC[];
        setNCs(data);
        setLoading(false);
      },
      (err) => {
        setError(err.message);
        setLoading(false);
      },
    );

    return unsubscribe;
  }, [labId]);

  return { ncs, loading, error };
}
```

**Create service `src/services/offlineQueueService.ts`:**
AsyncStorage CRUD for offline action queue (enqueue, dequeue, retry logic).

```typescript
// hc-quality-mobile/src/services/offlineQueueService.ts
import AsyncStorage from '@react-native-async-storage/async-storage';
import { v4 as uuid } from 'uuid';

export interface QueuedAction {
  id: string;
  action: 'submitCIQComment' | 'updateNCStatus' | 'submitReading' | 'submitTrainingSignature';
  labId: string;
  data: Record<string, any>;
  timestamp: number;
  retryCount: number;
  lastError?: string;
}

const QUEUE_KEY = 'offlineQueue';
const MAX_RETRIES = 5;

export async function enqueueAction(
  action: QueuedAction['action'],
  labId: string,
  data: Record<string, any>,
): Promise<string> {
  try {
    const existing = await AsyncStorage.getItem(QUEUE_KEY);
    const queue = existing ? JSON.parse(existing) : [];

    const newAction: QueuedAction = {
      id: uuid(),
      action,
      labId,
      data,
      timestamp: Date.now(),
      retryCount: 0,
    };

    queue.push(newAction);
    await AsyncStorage.setItem(QUEUE_KEY, JSON.stringify(queue));
    console.log(`[OfflineQueue] Enqueued ${action}:${newAction.id}`);
    return newAction.id;
  } catch (err) {
    console.error('[OfflineQueue] Enqueue error:', err);
    throw err;
  }
}

export async function getQueue(): Promise<QueuedAction[]> {
  try {
    const existing = await AsyncStorage.getItem(QUEUE_KEY);
    return existing ? JSON.parse(existing) : [];
  } catch (err) {
    console.error('[OfflineQueue] GetQueue error:', err);
    return [];
  }
}

export async function dequeueAction(id: string): Promise<void> {
  try {
    const existing = await AsyncStorage.getItem(QUEUE_KEY);
    if (!existing) return;

    const queue = JSON.parse(existing) as QueuedAction[];
    const filtered = queue.filter((a) => a.id !== id);

    await AsyncStorage.setItem(QUEUE_KEY, JSON.stringify(filtered));
    console.log(`[OfflineQueue] Dequeued ${id}`);
  } catch (err) {
    console.error('[OfflineQueue] Dequeue error:', err);
  }
}

export async function markRetry(id: string, error: string): Promise<void> {
  try {
    const existing = await AsyncStorage.getItem(QUEUE_KEY);
    if (!existing) return;

    const queue = JSON.parse(existing) as QueuedAction[];
    const updated = queue.map((a) =>
      a.id === id ? { ...a, retryCount: a.retryCount + 1, lastError: error } : a,
    );

    await AsyncStorage.setItem(QUEUE_KEY, JSON.stringify(updated));
  } catch (err) {
    console.error('[OfflineQueue] MarkRetry error:', err);
  }
}

export async function clearQueue(): Promise<void> {
  try {
    await AsyncStorage.removeItem(QUEUE_KEY);
    console.log('[OfflineQueue] Queue cleared');
  } catch (err) {
    console.error('[OfflineQueue] ClearQueue error:', err);
  }
}
```

**Create hook `src/hooks/useMobileCallables.ts`:**
Wrapper around Firebase Cloud Callables that gracefully queues on failure.

```typescript
// hc-quality-mobile/src/hooks/useMobileCallables.ts
import { useCallback } from 'react';
import { httpsCallable, HttpsCallableResult } from 'firebase/functions';
import { functions } from '../core/firebase';
import { enqueueAction } from '../services/offlineQueueService';

export function useMobileCallables() {
  const callFunction = useCallback(
    async (functionName: string, data: Record<string, any>): Promise<any> => {
      try {
        const fn = httpsCallable(functions, functionName);
        const result: HttpsCallableResult<any> = await fn(data);
        return result.data;
      } catch (err: any) {
        console.warn(`[Callable] ${functionName} failed:`, err.message);

        // Enqueue for retry if not a validation error
        if (err.code !== 'invalid-argument') {
          const actionType = functionName as any;
          await enqueueAction(actionType, data.labId, data);
        }

        throw err;
      }
    },
    [],
  );

  return { callFunction };
}
```

```typescript
import { useEffect, useState } from 'react';
import { NetInfo } from '@react-native-community/netinfo';

export function useNetInfo() {
  const [isOnline, setIsOnline] = useState(true);

  useEffect(() => {
    let unsubscribe: (() => void) | null = null;

    // Check initial state
    NetInfo.fetch().then((state) => {
      setIsOnline(state.isConnected ?? true);
    });

    // Subscribe to changes
    unsubscribe = NetInfo.addEventListener((state) => {
      setIsOnline(state.isConnected ?? true);
    });

    return () => {
      if (unsubscribe) unsubscribe();
    };
  }, []);

  return { isOnline };
}
```

```typescript
import { getQueue, dequeueAction, markRetry } from '../services/offlineQueueService';
import { useMobileCallables } from './useMobileCallables';

export function useOfflineSync() {
  const { isOnline } = useNetInfo();
  const { callFunction } = useMobileCallables();

  const processSyncQueue = useCallback(async () => {
    if (!isOnline) return;

    const queue = await getQueue();
    if (queue.length === 0) return;

    console.log(`[OfflineSync] Processing ${queue.length} queued actions`);

    for (const action of queue) {
      if (action.retryCount >= 5) {
        console.warn(`[OfflineSync] Action ${action.id} max retries reached`);
        continue;
      }

      try {
        await callFunction(action.action, {
          labId: action.labId,
          ...action.data,
        });

        await dequeueAction(action.id);
        console.log(`[OfflineSync] Action ${action.id} synced successfully`);
      } catch (err) {
        await markRetry(action.id, (err as Error).message);
        console.warn(`[OfflineSync] Action ${action.id} retry ${action.retryCount + 1}`);
      }
    }
  }, [isOnline, callFunction]);

  useEffect(() => {
    if (isOnline) {
      processSyncQueue();
    }
  }, [isOnline, processSyncQueue]);

  return { isOnline };
}
```

```typescript
export function OfflineIndicator() {
  const { isOnline } = useNetInfo();
  const [queueLength, setQueueLength] = useState(0);

  useEffect(() => {
    getQueue().then((q) => setQueueLength(q.length));
```

| ----------------------------- | ----------------------------------------------------------------------------------------------- |
| Client → Cloud Callable       | Mobile app sends action payload; Firebase validates auth token + labId matching                 |
| AsyncStorage → Firestore Sync | Queued action replayed on reconnect; server-wins conflict resolution prevents dupe/merge issues |
| User PIN → Signature          | PIN entered locally; transmitted as part of callable (TLS encrypted), never stored unencrypted  |

## STRIDE Threat Register

| Threat ID | Category               | Component                           | Disposition | Mitigation Plan                                                                                                                                       |
| --------- | ---------------------- | ----------------------------------- | ----------- | ----------------------------------------------------------------------------------------------------------------------------------------------------- |
| T-03.2-01 | Tampering              | AsyncStorage Queue                  | mitigate    | Queue stores minimal refs only (docId, fields-changed), not full documents. Server re-fetches current state before sync (prevents old data overwrite) |
| T-03.2-02 | Information Disclosure | OfflineIndicator shows queue length | accept      | Queue length is UI transparency, not sensitive data. User needs visibility into pending actions                                                       |
| T-03.2-03 | Spoofing               | Mobile user falsifies operatorId    | mitigate    | Cloud Callable validates `request.auth.uid` matches stored operatorId; Firestore rules enforce user-labId binding                                     |

| T-03.2-05 | Repudiation            | Training signature PIN              | mitigate    | PIN combined with timestamp + user identity + module context; audit log records callable invocation atomically                                        |
| T-03.2-06 | Elevation              | NC status update without permission | mitigate    | Cloud Callable checks `labMember` role before write; Firestore rules enforce read-only for non-lab users                                              |
