```typescript
// src/features/exports/hooks/useExportWizardState.ts
import { create } from 'zustand';
import { ExportFormat, ExportWizardState } from '../types/Exports';

interface WizardStore extends ExportWizardState {
  // Actions
  setFormat: (format: ExportFormat) => void;
  setDateRange: (start: string, end: string) => void;
  nextStep: () => void;
  prevStep: () => void;
  reset: () => void;
  open: () => void;
  close: () => void;
}

export const useExportWizardStore = create<WizardStore>((set) => ({
  step: 1,
  format: null,
  dateRangeStart: null,
  dateRangeEnd: null,
  isOpen: false,

  setFormat: (format) => set({ format }),
  setDateRange: (start, end) => set({ dateRangeStart: start, dateRangeEnd: end }),

  nextStep: () =>
    set((s) => ({
      step: Math.min(s.step + 1, 3) as 1 | 2 | 3,
    })),

  prevStep: () =>
    set((s) => ({
      step: Math.max(s.step - 1, 1) as 1 | 2 | 3,
    })),

  reset: () =>
    set({
      step: 1,
      format: null,
      dateRangeStart: null,
      dateRangeEnd: null,
    }),

  open: () => set({ isOpen: true }),
  close: () => set({ isOpen: false }),
}));
```

```typescript
// src/features/exports/components/ExportStep2.tsx
import React from 'react';
import { useExportWizardStore } from '../hooks/useExportWizardState';

export function ExportStep2() {
  const dateRangeStart = useExportWizardStore((s) => s.dateRangeStart);
  const dateRangeEnd = useExportWizardStore((s) => s.dateRangeEnd);
  const setDateRange = useExportWizardStore((s) => s.setDateRange);

  // Default to last 30 days
  const today = new Date().toISOString().split('T')[0];
  const thirtyDaysAgo = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000)
    .toISOString()
    .split('T')[0];

  const handleStartChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    setDateRange(e.target.value, dateRangeEnd || today);
  };

  const handleEndChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    setDateRange(dateRangeStart || thirtyDaysAgo, e.target.value);
  };

  return (
    <div className="space-y-4">
      <div>
        <label className="block text-sm font-medium text-slate-700 dark:text-slate-300 mb-1">
          Data Inicial
        </label>
        <input
          type="date"
          value={dateRangeStart || thirtyDaysAgo}
          onChange={handleStartChange}
```

```typescript
// src/features/exports/components/ExportWizard.tsx
import React from 'react';
import { useExportWizardStore } from '../hooks/useExportWizardState';
import { useExportInitiate } from '../hooks/useExportInitiate';
import { useAuthStore } from '../../../shared/hooks/useAuthStore';
import { ExportStep1 } from './ExportStep1';
import { ExportStep2 } from './ExportStep2';
import { ExportStep3 } from './ExportStep3';

export function ExportWizard() {
  const wizard = useExportWizardStore();
  const { labId } = useAuthStore();
  const { initiateExport, isLoading, error } = useExportInitiate();

  const handleSubmit = async () => {
    if (!wizard.format || !wizard.dateRangeStart || !wizard.dateRangeEnd) return;

    try {
      await initiateExport({
        labId: labId!,
        format: wizard.format,
        dateRangeStart: wizard.dateRangeStart,
        dateRangeEnd: wizard.dateRangeEnd,
      });

      wizard.close();
      wizard.reset();
    } catch (err) {
      // Error already in state
    }
  };

  if (!wizard.isOpen) return null;

  return (
    <div className="fixed inset-0 bg-black/50 dark:bg-black/70 flex items-center justify-center z-50">
      <div className="bg-white dark:bg-white/[0.03] border border-slate-200 dark:border-white/[0.06] rounded-xl shadow-lg w-[500px] max-h-[80vh] overflow-auto">
        {/* Step indicator */}
        <div className="px-6 py-4 border-b border-slate-100 dark:border-white/[0.06]">
          <div className="flex items-center gap-0 mb-6">
            {[
              { label: 'Formato', num: 1 },
              { label: 'Período', num: 2 },
              { label: 'Revisar', num: 3 },
            ].map((step, i) => (
              <React.Fragment key={step.num}>
                <div className="flex items-center gap-2.5">
                  <div
                    className={`w-[22px] h-[22px] rounded-full flex items-center justify-center text-[11px] font-semibold ${
                      wizard.step > step.num
                        ? 'bg-emerald-500 text-white'
                        : wizard.step === step.num
                        ? 'bg-blue-600 text-white'
                        : 'bg-white dark:bg-white/[0.05] border border-slate-200 dark:border-white/[0.06] text-slate-400'
                    }`}
                  >
                    {wizard.step > step.num ? '✓' : step.num}
                  </div>
                  <span
                    className={`text-sm ${
                      wizard.step >= step.num
                        ? 'text-slate-900 dark:text-white font-medium'
                        : 'text-slate-400'
                    }`}
                  >
                    {step.label}
                  </span>
                </div>
                {i < 2 && (
                  <div
                    className={`flex-1 h-px mx-3 ${
                      wizard.step > step.num ? 'bg-emerald-300' : 'bg-slate-200'
                    }`}
                  />
                )}
              </React.Fragment>
            ))}
          </div>
        </div>

        {/* Content */}
        <div className="p-6">
          {wizard.step === 1 && <ExportStep1 />}
          {wizard.step === 2 && <ExportStep2 />}
          {wizard.step === 3 && <ExportStep3 />}

          {error && (
            <div className="mt-4 p-3 bg-red-50 dark:bg-red-500/10 border border-red-200 dark:border-red-500/20 rounded-lg text-sm text-red-700 dark:text-red-400">
              {error}
            </div>
          )}
        </div>

        {/* Actions */}
        <div className="px-6 py-4 border-t border-slate-100 dark:border-white/[0.06] flex gap-3 justify-end">
          <button
            onClick={() => {
              if (wizard.step === 1) wizard.close();
              else wizard.prevStep();
            }}
            className="px-3.5 h-9 rounded-lg text-sm font-medium bg-white dark:bg-white/[0.05] border border-slate-200 dark:border-white/10 text-slate-700 dark:text-slate-300 hover:bg-slate-50 dark:hover:bg-white/[0.08] transition-colors"
          >
            {wizard.step === 1 ? 'Cancelar' : 'Voltar'}
          </button>
          <button
            onClick={() => {
              if (wizard.step === 3) handleSubmit();
              else wizard.nextStep();
            }}
            disabled={
              isLoading ||
              (wizard.step === 1 && !wizard.format) ||
              (wizard.step === 2 && (!wizard.dateRangeStart || !wizard.dateRangeEnd))
            }
            className="px-3.5 h-9 rounded-lg text-sm font-medium bg-blue-600 text-white hover:bg-blue-700 disabled:opacity-50 disabled:cursor-not-allowed transition-colors"
          >
            {isLoading ? '⟳' : wizard.step === 3 ? 'Exportar' : 'Próximo'}
          </button>
        </div>
      </div>
    </div>
  );
}
```

```typescript
import { useCallback, useState } from 'react';
import { initiateExportCallable } from '../services/exportCallables';
import { ExportInitiateRequest } from '../types/Exports';

export function useExportInitiate() {
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const initiateExport = useCallback(async (request: ExportInitiateRequest): Promise<string> => {
    setIsLoading(true);
    setError(null);

    try {
      const response = await initiateExportCallable(request);
      console.log('[Export] Job initiated:', response.jobId);
      return response.jobId;
    } catch (err: any) {
      const message = err.message || 'Falha ao iniciar exportação';
      setError(message);
      throw err;
    } finally {
      setIsLoading(false);
    }
  }, []);

  return { initiateExport, isLoading, error };
}
```

```typescript
// src/features/exports/hooks/useExportJobs.ts
import { useEffect, useState } from 'react';
import { collection, onSnapshot, query, where, orderBy } from 'firebase/firestore';
import { db } from '../../../core/firebase';
import { useAuthStore } from '../../../shared/hooks/useAuthStore';
import { ExportJob } from '../types/Exports';

export function useExportJobs() {
  const labId = useAuthStore((s) => s.activeLabId);
  const [jobs, setJobs] = useState<ExportJob[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!labId) {
      setLoading(false);
      return;
    }

    const q = query(collection(db, `labs/${labId}/export-jobs`), orderBy('createdAt', 'desc'));

    const unsubscribe = onSnapshot(
      q,
      (snapshot) => {
        const data = snapshot.docs.map((doc) => ({
          ...doc.data(),
          id: doc.id,
        })) as ExportJob[];
        setJobs(data);
        setLoading(false);
        setError(null);
      },
      (err) => {
        setError(err.message);
        setLoading(false);
      },
    );

    return unsubscribe;
  }, [labId]);

  return { jobs, loading, error };
}
```

```typescript
// src/features/exports/components/ExportStatusPoller.tsx
import React, { useEffect, useState } from 'react';
import { useExportPolling } from '../hooks/useExportPolling';
import { DownloadButton } from './DownloadButton';
import { JOB_STATUS_DISPLAY } from '../services/jobStatusCodes';

interface ExportStatusPollerProps {
  jobId: string;
}

export function ExportStatusPoller({ jobId }: ExportStatusPollerProps) {
  const { job, loading, isComplete } = useExportPolling(jobId);
  const [expiryCountdown, setExpiryCountdown] = useState<string>('');

  useEffect(() => {
    if (!job?.completedAt) return;

    const interval = setInterval(() => {
      const expiryTime = job.completedAt! + 7 * 24 * 60 * 60 * 1000; // 7 days
      const now = Date.now();
      const remaining = expiryTime - now;

      if (remaining <= 0) {
        setExpiryCountdown('Expirada');
      } else {
        const days = Math.floor(remaining / (24 * 60 * 60 * 1000));
        const hours = Math.floor((remaining % (24 * 60 * 60 * 1000)) / (60 * 60 * 1000));
        setExpiryCountdown(`${days}d ${hours}h`);
      }
    }, 60000); // update every minute

    return () => clearInterval(interval);
  }, [job?.completedAt]);

  if (loading || !job) {
    return <div className="animate-pulse h-10 bg-slate-200 dark:bg-white/[0.05] rounded-lg" />;
  }

  const statusInfo = JOB_STATUS_DISPLAY[job.status];

  return (
    <div
      className="p-3 rounded-lg border flex items-center justify-between"
      style={{
        backgroundColor: `${statusInfo.color}20`,
        borderColor: statusInfo.color,
      }}
    >
      <div className="flex items-center gap-3 flex-1">
        <span style={{ color: statusInfo.color }} className="text-lg">
          {statusInfo.icon}
        </span>
        <div className="flex-1">
          <div style={{ color: statusInfo.color }} className="text-sm font-semibold">
            {statusInfo.label}
          </div>
          {job.error && (
            <div className="text-xs text-slate-600 dark:text-slate-400 mt-0.5">
              {job.error}
            </div>
          )}
          {isComplete && expiryCountdown && (
            <div className="text-xs text-slate-600 dark:text-slate-400 mt-0.5">
              Link expira em: {expiryCountdown}
            </div>
          )}
        </div>
      </div>
      {isComplete && job.signedUrl && (
        <DownloadButton
          signedUrl={job.signedUrl}
          filename={`${job.format}_${job.dateRangeStart}_${job.dateRangeEnd}.xlsx`}
          fileSize={job.fileSize}
        />
      )}
    </div>
  );
}
```
