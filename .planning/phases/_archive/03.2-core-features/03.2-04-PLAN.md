import { render, screen, fireEvent, waitFor } from '@testing-library/react-native';

import { auth } from '../../core/firebase';

describe('Mobile E2E: Auth Flow', () => {
  beforeEach(() => {
    jest.clearAllMocks();
  });

  it('shows auth screen on cold start', async () => {
    (signInWithEmailAndPassword as jest.Mock).mockRejectedValue(
      new Error('Not logged in')
    );

    render(<App />);

    await waitFor(() => {
      expect(screen.queryByText('HC Quality Mobile')).toBeTruthy();
    });
  });

  it('logs in with valid credentials and shows home', async () => {
    const mockUser = {
      uid: 'user-123',
      email: 'test@example.com',
      emailVerified: true,
    };

    (signInWithEmailAndPassword as jest.Mock).mockResolvedValue({
      user: mockUser,
    });

    render(<App />);

    const emailInput = screen.getByPlaceholderText('Email');
    const passwordInput = screen.getByPlaceholderText('Senha');
    const loginButton = screen.getByText('Entrar');

    fireEvent.changeText(emailInput, 'test@example.com');
    fireEvent.changeText(passwordInput, 'password123');
    fireEvent.press(loginButton);

    await waitFor(() => {
      expect(screen.queryByText('Dashboard')).toBeTruthy();
    });
  });

  it('persists session on app restart', async () => {
    const mockUser = {
      uid: 'user-123',
      email: 'test@example.com',
    };

    // Mock Firebase Auth state restoration
    jest.spyOn(auth, 'currentUser', 'get').mockReturnValue(mockUser as any);

    render(<App />);

    await waitFor(() => {
      expect(screen.queryByText('Dashboard')).toBeTruthy();
    });
  });

  it('clears session on logout', async () => {
    // Start logged in
    jest.spyOn(auth, 'currentUser', 'get').mockReturnValue({
      uid: 'user-123',
      email: 'test@example.com',

    const { rerender } = render(<App />);

    jest.spyOn(auth, 'currentUser', 'get').mockReturnValue(null);
    rerender(<App />);

    await waitFor(() => {
      expect(screen.queryByText('HC Quality Mobile')).toBeTruthy();
    });
  });
});
