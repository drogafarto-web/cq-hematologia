**Create `src/hooks/useBiometricAuth.ts`:**

Wraps `react-native-biometrics` with graceful PIN fallback. Responsibilities:

- Check biometric sensor availability on mount
- Prompt biometric on first auth; store preference in AsyncStorage
- Return `promptBiometric()` function for use in AuthScreen
- Expose `isBiometricAvailable`, `isBiometricEnabled` state

Implementation contract:

```typescript
import ReactNativeBiometrics, { BiometryTypes } from 'react-native-biometrics';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { useState, useEffect, useCallback } from 'react';

const rnBiometrics = new ReactNativeBiometrics({ allowDeviceCredentials: false });
const BIOMETRIC_PREF_KEY = 'biometricEnabled';

export interface BiometricAuthResult {
  success: boolean;
  error?: string;
  method: 'biometric' | 'pin' | 'none';
}

export function useBiometricAuth() {
  const [isBiometricAvailable, setIsBiometricAvailable] = useState(false);
  const [isBiometricEnabled, setIsBiometricEnabled] = useState(false);
  const [biometryType, setBiometryType] = useState<string | null>(null);

  useEffect(() => {
    // Check sensor availability
    rnBiometrics.isSensorAvailable().then(({ available, biometryType: type }) => {
      setIsBiometricAvailable(available);
      if (available) setBiometryType(type ?? null);
    });

    // Check user preference
    AsyncStorage.getItem(BIOMETRIC_PREF_KEY).then((val) => {
      setIsBiometricEnabled(val === 'true');
    });
  }, []);

  const promptBiometric = useCallback(
    async (promptMessage = 'Autentique-se para continuar'): Promise<BiometricAuthResult> => {
      if (!isBiometricAvailable || !isBiometricEnabled) {
        return { success: false, method: 'none' };
      }

      try {
        const { success } = await rnBiometrics.simplePrompt({ promptMessage });
        return { success, method: 'biometric' };
      } catch (err: any) {
        // User cancelled or hardware error — fall back to PIN
        return { success: false, error: err.message, method: 'pin' };
      }
    },
    [isBiometricAvailable, isBiometricEnabled],
  );

  const enableBiometric = useCallback(async () => {
    await AsyncStorage.setItem(BIOMETRIC_PREF_KEY, 'true');
    setIsBiometricEnabled(true);
  }, []);

  const disableBiometric = useCallback(async () => {
    await AsyncStorage.setItem(BIOMETRIC_PREF_KEY, 'false');
    setIsBiometricEnabled(false);
  }, []);

  return {
    isBiometricAvailable,
    isBiometricEnabled,
    biometryType, // 'FaceID' | 'TouchID' | 'Biometrics'
    promptBiometric,
    enableBiometric,
    disableBiometric,
  };
}
```
