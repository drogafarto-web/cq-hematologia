import AsyncStorage from '@react-native-async-storage/async-storage';
import { v4 as uuidv4 } from 'uuid';

export interface QueuedWrite {
  id: string;
  operation: 'CREATE' | 'UPDATE' | 'DELETE';
  collection: string;
  docId: string;
  payload: Record<string, any>;
  timestamp: number;
  retryCount: number;
  lastError?: string;
}

const QUEUE_KEY = 'OFFLINE_SYNC_QUEUE';
const MAX_RETRIES = 3;
const RETRY_DELAY_MS = 5000;

export const offlineSyncService = {
  // Queue a write operation for offline sync
  async queueWrite(
    operation: 'CREATE' | 'UPDATE' | 'DELETE',
    collection: string,
    docId: string,
    payload: Record<string, any>,
  ): Promise<QueuedWrite> {
    try {
      const queue = await this.getQueue();
      const write: QueuedWrite = {
        id: uuidv4(),
        operation,
        collection,
        docId,
        payload,
        timestamp: Date.now(),
        retryCount: 0,
      };

      queue.push(write);
      await AsyncStorage.setItem(QUEUE_KEY, JSON.stringify(queue));
      
      console.log(`Queued: ${operation} ${collection}/${docId}`);
      return write;
    } catch (error) {
      console.error('Failed to queue write:', error);
      throw new Error('Falha ao salvar ação offline');
    }
  },

  // Get all queued writes
  async getQueue(): Promise<QueuedWrite[]> {
    try {
      const data = await AsyncStorage.getItem(QUEUE_KEY);
      return data ? JSON.parse(data) : [];
    } catch (error) {
      console.error('Failed to retrieve queue:', error);
      return [];
    }
  },

  // Sync queue to server
  async syncQueue(
    syncCallback: (write: QueuedWrite) => Promise<void>,
  ): Promise<{ success: number; failed: number }> {
    let successCount = 0;
    let failedCount = 0;

    try {
      const queue = await this.getQueue();
      if (queue.length === 0) return { success: 0, failed: 0 };

      console.log(`Syncing ${queue.length} queued writes...`);

      const updatedQueue: QueuedWrite[] = [];

      for (const write of queue) {
        try {
          await syncCallback(write);
          console.log(`Synced: ${write.operation} ${write.collection}/${write.docId}`);
          successCount++;
          // Don't add to updatedQueue — it's synced, remove from queue
        } catch (error) {
          failedCount++;
          write.retryCount++;
          write.lastError = String(error);

          if (write.retryCount < MAX_RETRIES) {
            console.warn(`Retry ${write.retryCount}/${MAX_RETRIES}: ${write.id}`, error);
            updatedQueue.push(write); // Keep for next sync attempt
          } else {
            console.error(`Failed after ${MAX_RETRIES} retries: ${write.id}`, error);
            // Drop from queue, log to analytics for debugging
          }
        }
      }

      // Update queue with remaining items
      if (updatedQueue.length > 0) {
        await AsyncStorage.setItem(QUEUE_KEY, JSON.stringify(updatedQueue));
      } else {
        await AsyncStorage.removeItem(QUEUE_KEY);
      }

      return { success: successCount, failed: failedCount };
    } catch (error) {
      console.error('Sync queue error:', error);
      return { success: successCount, failed: failedCount };
    }
  },

  // Clear entire queue
  async clearQueue(): Promise<void> {
    try {
      await AsyncStorage.removeItem(QUEUE_KEY);
      console.log('Queue cleared');
    } catch (error) {
      console.error('Failed to clear queue:', error);
    }
  },

  // Get queue length
  async getQueueLength(): Promise<number> {
    const queue = await this.getQueue();
    return queue.length;
  },
};
```

export interface OfflineSyncStore {
  queuedWrites: QueuedWrite[];
  isSyncing: boolean;
  lastSyncTime: number | null;
  syncError: string | null;

  // Actions
  setQueuedWrites: (writes: QueuedWrite[]) => void;
  setIsSyncing: (syncing: boolean) => void;
  setSyncError: (error: string | null) => void;
  setLastSyncTime: (time: number) => void;
  incrementQueueLength: () => void;
  decrementQueueLength: () => void;
}

export const useOfflineSyncStore = create<OfflineSyncStore>((set) => ({
  queuedWrites: [],
  isSyncing: false,
  lastSyncTime: null,
  syncError: null,

  setQueuedWrites: (writes) => set({ queuedWrites: writes }),
  setIsSyncing: (syncing) => set({ isSyncing: syncing }),
  setSyncError: (error) => set({ syncError: error }),
  setLastSyncTime: (time) => set({ lastSyncTime: time }),
  incrementQueueLength: () => set((state) => ({
    queuedWrites: [...state.queuedWrites, { id: '', operation: 'CREATE', collection: '', docId: '', payload: {}, timestamp: Date.now(), retryCount: 0 }],
  })),
  decrementQueueLength: () => set((state) => ({
    queuedWrites: state.queuedWrites.slice(0, -1),
  })),
}));

    isOnline: netInfo.isConnected === true,

import { useEffect, useState } from 'react';
import { useNetworkStatus } from './useNetworkStatus';
import { offlineSyncService } from '@/services/offlineSyncService';
import { useOfflineSyncStore } from '@/stores/offlineSyncStore';

export interface OfflineModeState {
  isOnline: boolean;
  isSyncing: boolean;
  queuedWriteCount: number;
  lastSyncTime: number | null;
}

export function useOfflineMode(): OfflineModeState {
  const networkStatus = useNetworkStatus();
  const { queuedWrites, isSyncing, lastSyncTime } = useOfflineSyncStore();

  return {
    isOnline: networkStatus.isOnline,
    isSyncing,
    queuedWriteCount: queuedWrites.length,
    lastSyncTime,
  };
}

// Hook to auto-sync when online
export function useAutoSync(
  onSync?: (result: { success: number; failed: number }) => void,
) {
  const { isOnline } = useOfflineMode();
  const { setIsSyncing, setSyncError } = useOfflineSyncStore();
  const [syncAttempted, setSyncAttempted] = useState(false);

  useEffect(() => {
    if (!isOnline || syncAttempted) return;

    const performSync = async () => {
      try {
        setIsSyncing(true);
        setSyncError(null);

        const result = await offlineSyncService.syncQueue(async (write) => {
          // Call server-side sync callable
          // Example: await callSyncQueuedWrite(write);
          console.log(`Syncing: ${write.operation} ${write.collection}/${write.docId}`);
        });

        setSyncError(null);
        setSyncAttempted(true);
        onSync?.(result);

        console.log(`Sync complete: ${result.success} success, ${result.failed} failed`);
      } catch (error) {
        setSyncError(String(error));
        console.error('Auto-sync error:', error);
      } finally {
        setIsSyncing(false);
      }
    };

    performSync();
  }, [isOnline, syncAttempted]);
}
```

export interface OfflineIndicatorProps {

export function OfflineIndicator({ onSyncPress }: OfflineIndicatorProps) {

          Modo offline • Dados em cache
        </Text>
        <TouchableOpacity onPress={onSyncPress}>
          <Text style={{ fontSize: 11, color: colors.text.accent, fontWeight: '600' }}>
            Tentar conectar
          </Text>
        </TouchableOpacity>
      </View>
    );
  }

  if (isSyncing) {
    return (
      <View
        style={{
          backgroundColor: colors.bg.secondary,
          paddingHorizontal: 16,
          paddingVertical: 12,

          gap: 8,

          Sincronizando {queuedWriteCount} ação{queuedWriteCount !== 1 ? 's' : ''}...

  return null;
}
