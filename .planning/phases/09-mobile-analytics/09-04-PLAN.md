```typescript
import { useState, useEffect, useCallback } from 'react';
import { KPIResponse } from '@/features/analytics/types/analyticsTypes';
import { analyticsService } from '@/features/analytics/services/analyticsService';

export interface AnalyticsState {
  kpiData: KPIResponse | null;
  isLoading: boolean;
  error: string | null;
  dateRange: { startDate: Date; endDate: Date };
}

export function useAnalyticsData(
  labId: string,
  autoRefreshMs: number = 30000, // 30 seconds
) {
  const [state, setState] = useState<AnalyticsState>({
    kpiData: null,
    isLoading: false,
    error: null,
    dateRange: {
      startDate: new Date(Date.now() - 30 * 24 * 60 * 60 * 1000), // 30 days ago
      endDate: new Date(),
    },
  });

  // Fetch analytics data
  const fetchData = useCallback(async () => {
    setState((prev) => ({ ...prev, isLoading: true, error: null }));
    try {
      const data = await analyticsService.callGetKPIs(
        labId,
        state.dateRange.startDate,
        state.dateRange.endDate,
      );
      setState((prev) => ({ ...prev, kpiData: data, isLoading: false }));
    } catch (error) {
      setState((prev) => ({
        ...prev,
        error: String(error),
        isLoading: false,
      }));
    }
  }, [labId, state.dateRange]);

  // Auto-refresh
  useEffect(() => {
    fetchData();
    const interval = setInterval(fetchData, autoRefreshMs);
    return () => clearInterval(interval);
  }, [fetchData, autoRefreshMs]);

  const setDateRange = useCallback((range: { startDate: Date; endDate: Date }) => {
    setState((prev) => ({ ...prev, dateRange: range }));
  }, []);

  const refreshNow = useCallback(() => {
    fetchData();
  }, [fetchData]);

  return {
    ...state,
    setDateRange,
    refreshNow,
  };
}
```

```typescript
import React from 'react';
import { View, Text, TouchableOpacity } from 'react-native';
import { useColors } from '@/hooks/useColors';

export interface KPICardProps {
  label: string;
  value: number;
  unit: string;
  trend: 'up' | 'down' | 'stable';
  onPress: () => void;
  trendData?: { date: string; value: number }[];
}

export function KPICard({
  label,
  value,
  unit,
  trend,
  onPress,
  trendData,
}: KPICardProps) {
  const colors = useColors();

  const trendColor =
    trend === 'up'
      ? colors.status.error // Increasing is bad for retrabalho%, good for conformidade
      : trend === 'down'
        ? colors.status.success
        : colors.text.secondary;

  const trendIcon = trend === 'up' ? '↑' : trend === 'down' ? '↓' : '→';

  return (
    <TouchableOpacity
      onPress={onPress}
      style={{
        backgroundColor: colors.bg.secondary,
        borderRadius: 12,
        padding: 16,
        marginBottom: 12,
        borderLeftWidth: 4,
        borderLeftColor: trendColor,
      }}
    >
      {/* Header: Label + Trend */}
      <View style={{ flexDirection: 'row', justifyContent: 'space-between', alignItems: 'center', marginBottom: 12 }}>
        <Text style={{ fontSize: 14, color: colors.text.secondary, fontWeight: '500' }}>
          {label}
        </Text>
        <Text style={{ fontSize: 16, color: trendColor, fontWeight: '600' }}>
          {trendIcon}
        </Text>
      </View>

      {/* Value */}
      <View style={{ marginBottom: 12 }}>
        <Text style={{ fontSize: 28, color: colors.text.primary, fontWeight: '700' }}>
          {value.toFixed(1)}
          <Text style={{ fontSize: 16, fontWeight: '500' }}>{unit}</Text>
        </Text>
      </View>

      {/* Mini Sparkline (simplified) */}
      {trendData && trendData.length > 0 && (
        <View style={{ height: 30, backgroundColor: colors.bg.tertiary, borderRadius: 4 }}>
          {/* Render mini chart (simplified as dots) */}
          <View style={{ flexDirection: 'row', paddingHorizontal: 4, paddingVertical: 4, gap: 2 }}>
            {trendData.slice(-7).map((point, i) => (
              <View
                key={i}
                style={{
                  width: 4,
                  height: 4,
                  borderRadius: 2,
                  backgroundColor: colors.text.secondary,
                  opacity: 0.5 + (i / 7) * 0.5, // Fade in effect
                }}
              />
            ))}
          </View>
        </View>
      )}
    </TouchableOpacity>
  );
}
```

```typescript
import React from 'react';
import { View, Text, ScrollView } from 'react-native';
import { useColors } from '@/hooks/useColors';
import { SPCResponse } from '@/features/analytics/types/analyticsTypes';

export interface SPCChartProps {
  data: SPCResponse;
}

export function SPCChart({ data }: SPCChartProps) {
  const colors = useColors();

  const { controlLimits, recentResults, outOfControl } = data;
  const yMin = Math.min(controlLimits.lowerControl, ...recentResults.map((r) => r.value));
  const yMax = Math.max(controlLimits.upperControl, ...recentResults.map((r) => r.value));
  const yRange = yMax - yMin;

  // Normalize values to 0-100 scale for plotting
  const normalize = (value: number) => ((value - yMin) / yRange) * 100;

  return (
    <View
      style={{
        backgroundColor: colors.bg.secondary,
        borderRadius: 12,
        padding: 16,
        marginVertical: 12,
      }}
    >
      {/* Header */}
      <View style={{ marginBottom: 16 }}>
        <View style={{ flexDirection: 'row', justifyContent: 'space-between', alignItems: 'center' }}>
          <Text style={{ fontSize: 16, color: colors.text.primary, fontWeight: '600' }}>
            Análise SPC — {data.analyte}
          </Text>
          {outOfControl && (
            <Text style={{ fontSize: 12, color: colors.status.error, fontWeight: '600' }}>
              ⚠ Fora de controle
            </Text>
          )}
        </View>
      </View>

      {/* Chart Area */}
      <View style={{ height: 200, marginBottom: 16, position: 'relative' }}>
        {/* Control Lines */}
        <View
          style={{
            position: 'absolute',
            top: `${100 - normalize(controlLimits.upperControl)}%`,
            left: 0,
            right: 0,
            height: 1,
            backgroundColor: colors.status.error,
            opacity: 0.5,
          }}
        />
        <View
          style={{
            position: 'absolute',
            top: `${100 - normalize(controlLimits.mean)}%`,
            left: 0,
            right: 0,
            height: 2,
            backgroundColor: colors.text.secondary,
            opacity: 0.7,
          }}
        />
        <View
          style={{
            position: 'absolute',
            top: `${100 - normalize(controlLimits.lowerControl)}%`,
            left: 0,
            right: 0,
            height: 1,
            backgroundColor: colors.status.error,
            opacity: 0.5,
          }}
        />

        {/* Data Points */}
        <View style={{ flexDirection: 'row', height: '100%', alignItems: 'flex-end', gap: 4 }}>
          {recentResults.map((result, i) => (
            <View
              key={i}
              style={{
                flex: 1,
                alignItems: 'center',
                justifyContent: 'flex-end',
              }}
            >
              <View
                style={{
                  width: '100%',
                  height: `${normalize(result.value)}%`,
                  backgroundColor: result.flagged ? colors.status.error : colors.text.accent,
                  borderRadius: 2,
                  opacity: result.flagged ? 1 : 0.7,
                }}
              />
            </View>
          ))}
        </View>
      </View>

      {/* Legend */}
      <View style={{ gap: 8 }}>
        <View style={{ flexDirection: 'row', alignItems: 'center', gap: 8 }}>
          <View style={{ width: 12, height: 12, backgroundColor: colors.text.accent, borderRadius: 2 }} />
          <Text style={{ fontSize: 12, color: colors.text.secondary }}>
            Dentro dos limites
          </Text>
        </View>
        <View style={{ flexDirection: 'row', alignItems: 'center', gap: 8 }}>
          <View style={{ width: 12, height: 12, backgroundColor: colors.status.error, borderRadius: 2 }} />
          <Text style={{ fontSize: 12, color: colors.text.secondary }}>
            Violação de regra
          </Text>
        </View>
      </View>

      {/* Westgard Violations */}
      {recentResults.some((r) => r.violations.length > 0) && (
        <View style={{ marginTop: 12, paddingTop: 12, borderTopWidth: 1, borderTopColor: colors.border.primary }}>
          <Text style={{ fontSize: 12, color: colors.text.secondary, fontWeight: '600', marginBottom: 8 }}>
            Violações detectadas:
          </Text>
          {recentResults
            .filter((r) => r.violations.length > 0)
            .map((result, i) => (
              <Text key={i} style={{ fontSize: 11, color: colors.status.error }}>
                {result.date}: {result.violations.join(', ')}
              </Text>
            ))}
        </View>
      )}
    </View>
  );
}
```

```typescript
            <Text style={{ color: colors.text.primary, fontWeight: '600' }}>
              {metric === 'turnaround'

                  : day.conformidade.toFixed(1)}
            </Text>
          </View>
        ))}

    </View>
  );
}
```

```typescript
import React from 'react';
import { View, Text, TouchableOpacity } from 'react-native';
import { useColors } from '@/hooks/useColors';
import DateTimePicker from '@react-native-community/datetimepicker';

export interface AnalyticsFiltersProps {
  onDateRangeChange: (range: { startDate: Date; endDate: Date }) => void;
  currentRange: { startDate: Date; endDate: Date };
}

export function AnalyticsFilters({
  onDateRangeChange,
  currentRange,
}: AnalyticsFiltersProps) {
  const colors = useColors();
  const [showStartPicker, setShowStartPicker] = React.useState(false);
  const [showEndPicker, setShowEndPicker] = React.useState(false);

  const handleStartDateChange = (event, date) => {
    if (date) {
      onDateRangeChange({ startDate: date, endDate: currentRange.endDate });
    }
    setShowStartPicker(false);
  };

  const handleEndDateChange = (event, date) => {
    if (date) {
      onDateRangeChange({ startDate: currentRange.startDate, endDate: date });
    }
    setShowEndPicker(false);
  };

  const formatDate = (date: Date) => {
    return date.toLocaleDateString('pt-BR');
  };

  return (
    <View style={{ paddingHorizontal: 16, paddingVertical: 12, gap: 8 }}>
      <Text style={{ fontSize: 12, color: colors.text.secondary, fontWeight: '600' }}>
        Período
      </Text>
      <View style={{ flexDirection: 'row', gap: 8 }}>
        <TouchableOpacity
          onPress={() => setShowStartPicker(true)}
          style={{
            flex: 1,
            paddingHorizontal: 12,
            paddingVertical: 8,
            backgroundColor: colors.bg.secondary,
            borderRadius: 6,
          }}
        >
          <Text style={{ fontSize: 12, color: colors.text.primary }}>
            {formatDate(currentRange.startDate)}
          </Text>
        </TouchableOpacity>

        <Text style={{ color: colors.text.secondary, paddingVertical: 8 }}>até</Text>

        <TouchableOpacity
          onPress={() => setShowEndPicker(true)}
          style={{
            flex: 1,
            paddingHorizontal: 12,
            paddingVertical: 8,
            backgroundColor: colors.bg.secondary,
            borderRadius: 6,
          }}
        >
          <Text style={{ fontSize: 12, color: colors.text.primary }}>
            {formatDate(currentRange.endDate)}
          </Text>
        </TouchableOpacity>
      </View>

      {showStartPicker && (
        <DateTimePicker
          value={currentRange.startDate}
          mode="date"
          display="default"
          onChange={handleStartDateChange}
        />
      )}

      {showEndPicker && (
        <DateTimePicker
          value={currentRange.endDate}
          mode="date"
          display="default"
          onChange={handleEndDateChange}
        />
      )}
    </View>
  );
}
```
