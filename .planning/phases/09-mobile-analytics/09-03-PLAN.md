```typescript
import * as admin from 'firebase-admin';
import { Timestamp } from 'firebase-admin/firestore';

export interface KPIResponse {
  labId: string;
  period: { startDate: string; endDate: string };
  kpis: {
    turnaroundTime: {
      mean: number; // minutes
      p50: number;
      p95: number;
    };
    retrabalhoRate: {
      percentage: number;
      count: number; // number of retesting items
      total: number; // total items tested
    };
    conformidade: {
      percentage: number;
      metCount: number; // items meeting criteria
      totalCount: number;
    };
  };
  trendData: {
    date: string;
    turnaround: number;
    retrabalho: number;
    conformidade: number;
  }[];
}

export async function aggregateKPIs(
  labId: string,
  startDate: Date,
  endDate: Date,
): Promise<KPIResponse> {
  const db = admin.firestore();

  // 1. Compute Turnaround Time
  const laudosSnapshot = await db
    .collection('labs')
    .doc(labId)
    .collection('laudos')
    .where('criadoEm', '>=', Timestamp.fromDate(startDate))
    .where('criadoEm', '<=', Timestamp.fromDate(endDate))
    .get();

  const turnaroundTimes: number[] = [];
  laudosSnapshot.docs.forEach((doc) => {
    const created = (doc.data().criadoEm as Timestamp).toDate();
    const released = (doc.data().liberadoEm as Timestamp)?.toDate() || new Date();
    const diffMinutes = (released.getTime() - created.getTime()) / (1000 * 60);
    turnaroundTimes.push(diffMinutes);
  });

  const turnaround = {
    mean: turnaroundTimes.reduce((a, b) => a + b, 0) / turnaroundTimes.length || 0,
    p50: percentile(turnaroundTimes, 50),
    p95: percentile(turnaroundTimes, 95),
  };

  // 2. Compute Retrabalho Rate
  const ncSnapshot = await db
    .collection('labs')
    .doc(labId)
    .collection('nonconformities')
    .where('criadoEm', '>=', Timestamp.fromDate(startDate))
    .where('criadoEm', '<=', Timestamp.fromDate(endDate))
    .where('tipo', '==', 'retrabalho')
    .get();

  const retestCount = ncSnapshot.docs.length;
  const totalTestedEstimate = turnaroundTimes.length; // or query actual test count
  const retrabalhoRate = {
    percentage: (retestCount / (totalTestedEstimate || 1)) * 100,
    count: retestCount,
    total: totalTestedEstimate,
  };

  // 3. Compute Conformidade (% of tests meeting QC criteria)
  const ciqSnapshot = await db
    .collection('labs')
    .doc(labId)
    .collection('ciq')
    .where('criadoEm', '>=', Timestamp.fromDate(startDate))
    .where('criadoEm', '<=', Timestamp.fromDate(endDate))
    .get();

  const conformeCount = ciqSnapshot.docs.filter((doc) => {
    const result = doc.data().resultado;
    return result === 'conforme' || result === 'dentro-dos-limites';
  }).length;

  const conformidade = {
    percentage: (conformeCount / (ciqSnapshot.size || 1)) * 100,
    metCount: conformeCount,
    totalCount: ciqSnapshot.size,
  };

  // 4. Build Trend Data (daily aggregates)
  const trendData: KPIResponse['trendData'] = [];
  let current = new Date(startDate);
  while (current < endDate) {
    const dayEnd = new Date(current);
    dayEnd.setDate(dayEnd.getDate() + 1);

    const dayTurnarounds = turnaroundTimes.filter((t) => {
      // Filter turnarounds for this day (simplified)
      return t > 0;
    });
    const dayRetrabalho = (retestCount / (totalTestedEstimate || 1)) * 100;
    const dayConformidade = (conformeCount / (ciqSnapshot.size || 1)) * 100;

    trendData.push({
      date: current.toISOString().split('T')[0],
      turnaround: dayTurnarounds.length > 0 ? dayTurnarounds.reduce((a, b) => a + b) / dayTurnarounds.length : 0,
      retrabalho: dayRetrabalho,
      conformidade: dayConformidade,
    });

    current.setDate(current.getDate() + 1);
  }

  return {
    labId,
    period: {
      startDate: startDate.toISOString().split('T')[0],
      endDate: endDate.toISOString().split('T')[0],
    },
    kpis: {
      turnaroundTime: turnaround,
      retrabalhoRate: retrabalhoRate,
      conformidade: conformidade,
    },
    trendData,
  };
}

// Helper: compute percentile from array
function percentile(arr: number[], p: number): number {
  const sorted = [...arr].sort((a, b) => a - b);
  const index = Math.ceil((p / 100) * sorted.length) - 1;
  return sorted[Math.max(0, index)] || 0;
}
```

```typescript
// functions/src/modules/analytics/predictiveSPC.ts
import * as admin from 'firebase-admin';

export interface ControlLimits {
  mean: number;
  stdDev: number;
  upperControl: number; // mean + 3*SD
  lowerControl: number; // mean - 3*SD
  warning: { upper: number; lower: number }; // mean +/- 2*SD
}

export interface SPCResult {
  value: number;
  flagged: boolean;
  violations: string[]; // e.g., ['2-2s', 'R-4s']
}

export function computeControlLimits(
  values: number[],
): ControlLimits {
  const mean = values.reduce((a, b) => a + b, 0) / values.length;
  const variance = values.reduce((sq, val) => sq + Math.pow(val - mean, 2), 0) / values.length;
  const stdDev = Math.sqrt(variance);

  return {
    mean,
    stdDev,
    upperControl: mean + 3 * stdDev,
    lowerControl: mean - 3 * stdDev,
    warning: {
      upper: mean + 2 * stdDev,
      lower: mean - 2 * stdDev,
    },
  };
}

export function detectWestgardViolations(
  recentResults: number[],
  controlLimits: ControlLimits,
): string[] {
  const violations: string[] = [];
  const { mean, stdDev, upperControl, lowerControl, warning } = controlLimits;

  if (recentResults.length === 0) return violations;

  const latest = recentResults[recentResults.length - 1];

  // 1. 1-3s: Single result beyond 3-sigma
  if (latest > upperControl || latest < lowerControl) {
    violations.push('1-3s');
  }

  // 2. 2-2s: Two consecutive results between 2-sigma and 3-sigma (same side)
  if (recentResults.length >= 2) {
    const prev = recentResults[recentResults.length - 2];
    const latestAbove2s = latest > warning.upper;
    const prevAbove2s = prev > warning.upper;
    const latestBelow2s = latest < warning.lower;
    const prevBelow2s = prev < warning.lower;

    if ((latestAbove2s && prevAbove2s) || (latestBelow2s && prevBelow2s)) {
      violations.push('2-2s');
    }
  }

  // 3. R-4s: Range between consecutive results exceeds 4-sigma
  if (recentResults.length >= 2) {
    const prev = recentResults[recentResults.length - 2];
    const range = Math.abs(latest - prev);
    const limit4s = 4 * stdDev;
    if (range > limit4s) {
      violations.push('R-4s');
    }
  }

  // 4. 10x: Ten consecutive results on one side of mean
  if (recentResults.length >= 10) {
    const last10 = recentResults.slice(-10);
    const allAbove = last10.every((v) => v > mean);
    const allBelow = last10.every((v) => v < mean);
    if (allAbove || allBelow) {
      violations.push('10x');
    }
  }

  return violations;
}

export async function analyzeAnalyteSPC(
  labId: string,
  analyte: string,
  lookbackDays: number = 30,
): Promise<{
  controlLimits: ControlLimits;
  recentResults: { date: string; value: number; flagged: boolean; violations: string[] }[];
  outOfControl: boolean;
}> {
  const db = admin.firestore();

  // Get QC results for analyte over lookback period
  const startDate = new Date();
  startDate.setDate(startDate.getDate() - lookbackDays);

  const resultsSnapshot = await db
    .collection('labs')
    .doc(labId)
    .collection('qc_results')
    .where('analyte', '==', analyte)
    .where('timestamp', '>=', admin.firestore.Timestamp.fromDate(startDate))
    .orderBy('timestamp', 'asc')
    .limit(100)
    .get();

  const values = resultsSnapshot.docs.map((doc) => doc.data().value as number);
  if (values.length < 3) {
    throw new Error(`Insufficient data for analyte ${analyte}`);
  }

  const controlLimits = computeControlLimits(values);
  const violations = detectWestgardViolations(values, controlLimits);

  // Build results with flags
  const recentResults = resultsSnapshot.docs.map((doc) => {
    const value = doc.data().value as number;
    const docViolations = detectWestgardViolations([value], controlLimits);
    return {
      date: new Date(doc.data().timestamp.toDate()).toISOString().split('T')[0],
      value,
      flagged: docViolations.length > 0,
      violations: docViolations,
    };
  });

  // Out of control if latest result has violations
  const outOfControl = recentResults.length > 0 && recentResults[recentResults.length - 1].flagged;

  return {
    controlLimits,
    recentResults,
    outOfControl,
  };
}
```

**Westgard Rules Explained:**
- **1-3s:** Single result beyond 3 standard deviations (immediate halt)
- **2-2s:** Two consecutive results >2 SD from mean (same direction)
- **R-4s:** Range between consecutive results >4 SD
- **10x:** Ten consecutive results all on one side of mean

**Verification:**
- Control limits computed correctly from data
- Westgard violations detected accurately
- Out-of-control flag triggers on rule violations
- Results flagged and stored for trending

```typescript
// src/features/analytics/services/analyticsService.ts
import { httpsCallable } from 'firebase/functions';
import { functions } from '@/firebase/config';
import { KPIResponse, SPCResponse, BenchmarkResponse } from '@/features/analytics/types/analyticsTypes';

const CACHE_TTL_MS = 30 * 1000; // 30 seconds

interface CacheEntry {
  data: any;
  timestamp: number;
}

const cache = new Map<string, CacheEntry>();

export const analyticsService = {
  async callGetKPIs(
    labId: string,
    startDate: Date,
    endDate: Date,
  ): Promise<KPIResponse> {
    const cacheKey = `kpi-${labId}-${startDate}-${endDate}`;
    const cached = cache.get(cacheKey);

    if (cached && Date.now() - cached.timestamp < CACHE_TTL_MS) {
      console.log('KPI cache hit');
      return cached.data;
    }

    try {
      const callGetKPIs = httpsCallable(functions, 'analyticsGetKPIs');
      const response = await callGetKPIs({
        labId,
        startDate: startDate.toISOString(),
        endDate: endDate.toISOString(),
      });

      const data = response.data as KPIResponse;
      cache.set(cacheKey, { data, timestamp: Date.now() });
      return data;
    } catch (error) {
      console.error('KPI fetch error:', error);
      throw new Error('Falha ao buscar KPIs');
    }
  },

  async callPredictiveSPC(
    labId: string,
    analyte: string,
  ): Promise<SPCResponse> {
    const cacheKey = `spc-${labId}-${analyte}`;
    const cached = cache.get(cacheKey);

    if (cached && Date.now() - cached.timestamp < CACHE_TTL_MS) {
      console.log('SPC cache hit');
      return cached.data;
    }

    try {
      const callSPC = httpsCallable(functions, 'analyticsPredictiveSPC');
      const response = await callSPC({ labId, analyte });

      const data = response.data as SPCResponse;
      cache.set(cacheKey, { data, timestamp: Date.now() });
      return data;
    } catch (error) {
      console.error('SPC fetch error:', error);
      throw new Error('Falha ao buscar análise SPC');
    }
  },

  async callBenchmark(
    labId: string,
    metric: 'turnaround' | 'retrabalho' | 'conformidade',
    yourScore: number,
  ): Promise<BenchmarkResponse> {
    const cacheKey = `benchmark-${labId}-${metric}`;
    const cached = cache.get(cacheKey);

    if (cached && Date.now() - cached.timestamp < CACHE_TTL_MS) {
      console.log('Benchmark cache hit');
      return cached.data;
    }

    try {
      const callBenchmark = httpsCallable(functions, 'analyticsBenchmark');
      const response = await callBenchmark({ labId, metric, yourScore });

      const data = response.data as BenchmarkResponse;
      cache.set(cacheKey, { data, timestamp: Date.now() });
      return data;
    } catch (error) {
      console.error('Benchmark fetch error:', error);
      throw new Error('Falha ao buscar benchmarks');
    }
  },

  clearCache(): void {
    cache.clear();
  },
};
```
