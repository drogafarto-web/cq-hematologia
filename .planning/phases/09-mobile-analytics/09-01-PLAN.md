   ```typescript
   import { useMemo, useCallback, useState, useEffect } from 'react';
   import AsyncStorage from '@react-native-async-storage/async-storage';
   import * as BiometricLib from 'react-native-biometrics';

   export interface BiometricState {
     available: boolean;
     biometricType: 'faceID' | 'fingerprint' | 'none';
     isLocked: boolean;
     lockoutTimeRemaining: number; // seconds
     failedAttempts: number;
   }

   export function useBiometricAuth() {
     const [state, setState] = useState<BiometricState>({
       available: false,
       biometricType: 'none',
       isLocked: false,
       lockoutTimeRemaining: 0,
       failedAttempts: 0,
     });

     // Initialize: detect biometric type on mount
     useEffect(() => {
       const detectBiometric = async () => {
         try {
           const compatible = await BiometricLib.isSensorAvailable();
           if (!compatible) {
             setState((prev) => ({ ...prev, available: false }));
             return;
           }

           const biometryType = await BiometricLib.biometricKeysExist();
           const type = biometryType.includes('FaceID')
             ? 'faceID'
             : biometryType.includes('TouchID')
               ? 'fingerprint'
               : 'none';

           setState((prev) => ({
             ...prev,
             available: type !== 'none',
             biometricType: type,
           }));
         } catch (error) {
           console.error('Biometric detection error:', error);
           setState((prev) => ({ ...prev, available: false }));
         }
       };

       detectBiometric();
       loadLockoutState();
     }, []);

     // Load lockout state from AsyncStorage
     const loadLockoutState = useCallback(async () => {
       try {
         const lockoutData = await AsyncStorage.getItem('PIN_LOCKOUT');
         if (!lockoutData) return;

         const { lockedUntil, failedAttempts } = JSON.parse(lockoutData);
         const now = Date.now();
         const remainingMs = lockedUntil - now;

         if (remainingMs > 0) {
           setState((prev) => ({
             ...prev,
             isLocked: true,
             lockoutTimeRemaining: Math.ceil(remainingMs / 1000),
             failedAttempts,
           }));

           // Start countdown timer
           const interval = setInterval(async () => {
             setState((prev) => {
               const newRemaining = prev.lockoutTimeRemaining - 1;
               if (newRemaining <= 0) {
                 AsyncStorage.removeItem('PIN_LOCKOUT').catch(console.error);
                 clearInterval(interval);
                 return { ...prev, isLocked: false, lockoutTimeRemaining: 0, failedAttempts: 0 };
               }
               return { ...prev, lockoutTimeRemaining: newRemaining };
             });
           }, 1000);

           return () => clearInterval(interval);
         }
       } catch (error) {
         console.error('Failed to load lockout state:', error);
       }
     }, []);

     // Prompt biometric
     const promptBiometric = useCallback(async () => {
       if (!state.available) return 'pin' as const;

       try {
         await BiometricLib.simplePrompt({
           promptMessage:
             'Autenticar com ' +
             (state.biometricType === 'faceID' ? 'Face ID' : 'Impressão digital'),
           fallbackLabel: 'Use PIN',
         });
         return state.biometricType as 'faceID' | 'fingerprint';
       } catch (error) {
         // Biometric failed or user cancelled
         return 'pin' as const;
       }
     }, [state]);

     // Record failed PIN attempt
     const recordFailedAttempt = useCallback(async () => {
       try {
         const newFailures = state.failedAttempts + 1;

         if (newFailures >= 5) {
           // Lock device for 5 minutes
           const lockedUntil = Date.now() + 5 * 60 * 1000;
           await AsyncStorage.setItem(
             'PIN_LOCKOUT',
             JSON.stringify({ lockedUntil, failedAttempts: newFailures }),
           );
           setState((prev) => ({
             ...prev,
             isLocked: true,
             lockoutTimeRemaining: 300,
             failedAttempts: newFailures,
           }));
         } else {
           setState((prev) => ({ ...prev, failedAttempts: newFailures }));
         }
       } catch (error) {
         console.error('Failed to record lockout:', error);
       }
     }, [state.failedAttempts]);

     // Clear failed attempts on success
     const clearFailedAttempts = useCallback(async () => {
       try {
         await AsyncStorage.removeItem('PIN_LOCKOUT');
         setState((prev) => ({
           ...prev,
           failedAttempts: 0,
           isLocked: false,
           lockoutTimeRemaining: 0,
         }));
       } catch (error) {
         console.error('Failed to clear lockout:', error);
       }
     }, []);

     return {
       ...state,
       promptBiometric,
       recordFailedAttempt,
       clearFailedAttempts,
     };
   }
   ```

   ```typescript
   import * as SecureStore from 'react-native-keychain';
   import { Platform } from 'react-native';

   export const secureStorageService = {
     async saveFirebaseToken(token: string): Promise<void> {
       try {
         await SecureStore.setGenericPassword('firebase_token', token, {
           storage:
             Platform.OS === 'ios' ? SecureStore.STORAGE_TYPE.KC : SecureStore.STORAGE_TYPE.FB,
           accessible: Platform.OS === 'ios' ? SecureStore.ACCESSIBLE.WHEN_UNLOCKED : undefined,
         });
       } catch (error) {
         console.error('Failed to save token to Keychain:', error);
         throw new Error('Falha ao salvar token');
       }
     },

     async retrieveFirebaseToken(): Promise<string | null> {
       try {
         const credentials = await SecureStore.getGenericPassword();
         if (credentials && credentials.password) {
           return credentials.password;
         }
         return null;
       } catch (error) {
         console.error('Failed to retrieve token from Keychain:', error);
         return null;
       }
     },

     async clearAll(): Promise<void> {
       try {
         await SecureStore.resetGenericPassword();
       } catch (error) {
         console.error('Failed to clear Keychain:', error);
         throw new Error('Falha ao limpar dados');
       }
     },

     async tokenExists(): Promise<boolean> {
       const token = await this.retrieveFirebaseToken();
       return token !== null;
     },
   };
   ```

   ```typescript
   import { useEffect } from 'react';
   import { useAuthStore } from '@/stores/authStore';
   import { secureStorageService } from '@/services/secureStorageService';
   import { validateFirebaseToken } from '@/services/authService';

   export function useAuthPersistence() {
     const { setAuthToken, clearAuth } = useAuthStore();

     useEffect(() => {
       const restoreAuth = async () => {
         try {
           const token = await secureStorageService.retrieveFirebaseToken();
           if (!token) return;

           // Validate token is not expired
           const isValid = await validateFirebaseToken(token);
           if (isValid) {
             setAuthToken(token);
           } else {
             // Token expired, clear it
             await secureStorageService.clearAll();
             clearAuth();
           }
         } catch (error) {
           console.error('Auth restore error:', error);
           clearAuth();
         }
       };

       restoreAuth();
     }, []);
   }
   ```

```typescript
// hc-quality-mobile/src/components/PINForm.tsx
import React, { useState, useCallback } from 'react';
import { View, Text, TouchableOpacity, TextInput, ActivityIndicator } from 'react-native';
import { useColors } from '@/hooks/useColors';

export interface PINFormProps {
  isLocked: boolean;
  lockoutTimeRemaining: number; // seconds
  failedAttempts: number;
  isLoading: boolean;
  error?: string | null;
  onPINSubmit: (pin: string) => Promise<void>;
  onBiometricFallback?: () => void;
}

export function PINForm({
  isLocked,
  lockoutTimeRemaining,
  failedAttempts,
  isLoading,
  error,
  onPINSubmit,
  onBiometricFallback,
}: PINFormProps) {
  const [pin, setPin] = useState('');
  const colors = useColors();

  const handlePINChange = useCallback((text: string) => {
    // Only allow digits, max 6
    const digits = text.replace(/\D/g, '').slice(0, 6);
    setPin(digits);

    // Auto-submit when 6 digits entered
    if (digits.length === 6) {
      onPINSubmit(digits);
    }
  }, [onPINSubmit]);

  if (isLocked) {
    return (
      <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center', gap: 16 }}>
        <Text style={{ fontSize: 24, fontWeight: '700', color: colors.status.error }}>
          Dispositivo bloqueado
        </Text>
        <Text style={{ fontSize: 16, color: colors.text.secondary }}>
          Tente novamente em {lockoutTimeRemaining}s
        </Text>
        <View style={{ width: 200, height: 4, backgroundColor: colors.bg.secondary, borderRadius: 2 }}>
          <View
            style={{
              width: `${(100 * (300 - lockoutTimeRemaining)) / 300}%`,
              height: '100%',
              backgroundColor: colors.status.success,
              borderRadius: 2,
            }}
          />
        </View>
      </View>
    );
  }

  return (
    <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center', gap: 24 }}>
      {/* PIN Display */}
      <View style={{ flexDirection: 'row', gap: 8 }}>
        {Array.from({ length: 6 }).map((_, i) => (
          <View
            key={i}
            style={{
              width: 48,
              height: 48,
              borderRadius: 8,
              backgroundColor: i < pin.length ? colors.bg.secondary : colors.bg.tertiary,
              borderWidth: 2,
              borderColor: colors.border.primary,
              justifyContent: 'center',
              alignItems: 'center',
            }}
          >
            {i < pin.length && (
              <Text style={{ fontSize: 24, color: colors.text.primary }}>●</Text>
            )}
          </View>
        ))}
      </View>

      {/* Hidden TextInput */}
      <TextInput
        value={pin}
        onChangeText={handlePINChange}
        keyboardType="numeric"
        maxLength={6}
        secureTextEntry={false}
        style={{ opacity: 0, width: 0, height: 0 }}
        autoFocus
      />

      {/* Error Message */}
      {error && (
        <Text style={{ fontSize: 14, color: colors.status.error, textAlign: 'center' }}>
          {error}
        </Text>
      )}

      {/* Attempts Counter */}
      {failedAttempts > 0 && (
        <Text style={{ fontSize: 12, color: colors.text.secondary }}>
          {failedAttempts} / 5 tentativas incorretas
        </Text>
      )}

      {/* Loading Indicator */}
      {isLoading && <ActivityIndicator size="large" color={colors.text.primary} />}

      {/* Biometric Fallback Button */}
      {onBiometricFallback && (
        <TouchableOpacity onPress={onBiometricFallback} style={{ paddingVertical: 8 }}>
          <Text style={{ fontSize: 14, color: colors.text.accent }}>
            Usar biometria
          </Text>
        </TouchableOpacity>
      )}
    </View>
  );
}
```

```typescript
// src/features/auth/services/mobileAuthService.ts
import { httpsCallable } from 'firebase/functions';
import { functions } from '@/firebase/config';

export const mobileAuthService = {
  async authenticateWithBiometric(
    labId: string,
    email: string,
    biometricToken: string, // Device-specific token from biometric provider
  ) {
    try {
      const callAuth = httpsCallable(functions, 'mobileAuthenticateWithBiometric');
      const response = await callAuth({ labId, email, biometricToken });
      return response.data as { idToken: string; refreshToken: string };
    } catch (error) {
      console.error('Biometric auth failed:', error);
      throw new Error('Falha na autenticação biométrica');
    }
  },

  async authenticateWithPIN(
    labId: string,
    email: string,
    pin: string, // 6-digit PIN from user
  ) {
    try {
      const callAuth = httpsCallable(functions, 'mobileAuthenticateWithPIN');
      const response = await callAuth({ labId, email, pin });
      return response.data as { idToken: string; refreshToken: string };
    } catch (error) {
      if (error.code === 'invalid-argument') {
        throw new Error('PIN incorreto');
      }
      console.error('PIN auth failed:', error);
      throw new Error('Falha na autenticação');
    }
  },

  async validateToken(token: string): Promise<boolean> {
    try {
      const callValidate = httpsCallable(functions, 'mobileValidateToken');
      await callValidate({ idToken: token });
      return true;
    } catch (error) {
      return false;
    }
  },

  async logout() {
    // Client-side only: clear secure storage
    // Backend cleanup handled by token expiry
  },
};
```
