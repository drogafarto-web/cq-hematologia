### REQ-04.1: Accuracy Calculation Engine

**File:** `functions/src/modules/ciqImuno/accuracyCalculator.ts`

**Logic:**
```typescript
function calculateAccuracy(
  images: ImmunoIAImage[]
): {
  totalImages: number;
  correctMatches: number;
  accuracy: number;
  confusionMatrix: {
    [key: string]: { [key: string]: number };
  };
  confidenceBins: {
    [confidence: string]: {
      count: number;
      accuracy: number;
    };
  };
  perTestKitMetrics: {
    [testKit: string]: {
      accuracy: number;
      count: number;
      confidence_avg: number;
    };
  };
} {
  // 1. Filter images with manual verifications
  const verified = images.filter(img => img.manualVerdict);
  const total = verified.length;
  
  // 2. Build confusion matrix
  const matrix: Record<string, Record<string, number>> = {};
  let correct = 0;
  
  for (const img of verified) {
    const predicted = img.geminiClassification.classification;
    const actual = img.manualVerdict.classification;
    
    if (predicted === actual) correct++;
    
    matrix[actual] = matrix[actual] || {};
    matrix[actual][predicted] = (matrix[actual][predicted] || 0) + 1;
  }
  
  // 3. Calculate per-confidence bins (0.7-0.8, 0.8-0.9, 0.9-1.0)
  const bins: Record<string, { correct: number; total: number }> = {
    '0.7-0.8': { correct: 0, total: 0 },
    '0.8-0.9': { correct: 0, total: 0 },
    '0.9-1.0': { correct: 0, total: 0 },
  };
  
  for (const img of verified) {
    const conf = img.geminiClassification.confidence;
    const isCorrect = img.geminiClassification.classification === img.manualVerdict.classification;
    
    if (conf >= 0.7 && conf < 0.8) {
      bins['0.7-0.8'].total++;
      if (isCorrect) bins['0.7-0.8'].correct++;
    } else if (conf >= 0.8 && conf < 0.9) {
      bins['0.8-0.9'].total++;
      if (isCorrect) bins['0.8-0.9'].correct++;
    } else if (conf >= 0.9) {
      bins['0.9-1.0'].total++;
      if (isCorrect) bins['0.9-1.0'].correct++;
    }
  }
  
  // 4. Per-test-kit metrics
  const perKit: Record<string, any> = {};
  for (const kit of ['HIV', 'Dengue', 'Syphilis', 'COVID', 'HCG']) {
    const kitImages = verified.filter(img => img.testKit === kit);
    const kitCorrect = kitImages.filter(
      img => img.geminiClassification.classification === img.manualVerdict.classification
    ).length;
    const avgConf = kitImages.length > 0
      ? kitImages.reduce((sum, img) => sum + img.geminiClassification.confidence, 0) / kitImages.length
      : 0;
    
    perKit[kit] = {
      accuracy: kitImages.length > 0 ? (kitCorrect / kitImages.length) : 0,
      count: kitImages.length,
      confidence_avg: avgConf,
    };
  }
  
  return {
    totalImages: total,
    correctMatches: correct,
    accuracy: total > 0 ? correct / total : 0,
    confusionMatrix: matrix,
    confidenceBins: {
      '0.7-0.8': {
        count: bins['0.7-0.8'].total,
        accuracy: bins['0.7-0.8'].total > 0 
          ? bins['0.7-0.8'].correct / bins['0.7-0.8'].total 
          : 0,
      },
      '0.8-0.9': {
        count: bins['0.8-0.9'].total,
        accuracy: bins['0.8-0.9'].total > 0 
          ? bins['0.8-0.9'].correct / bins['0.8-0.9'].total 
          : 0,
      },
      '0.9-1.0': {
        count: bins['0.9-1.0'].total,
        accuracy: bins['0.9-1.0'].total > 0 
          ? bins['0.9-1.0'].correct / bins['0.9-1.0'].total 
          : 0,
      },
    },
    perTestKitMetrics: perKit,
  };
}
```

**Acceptance criteria:**
- [ ] Accuracy calculation correct (unit test with known data)
- [ ] Confusion matrix accurate
- [ ] Per-confidence and per-test-kit metrics calculated
- [ ] Handles zero-division edge cases
- [ ] Performance: <5s for 500 images

### REQ-04.4: ADR-0016: IA Strip Classification Approach

**File:** `docs/adr/0016-IA-STRIP-CLASSIFICATION.md`

**Contents:**

#### Decision
Use Gemini 2.5 Flash Vision API for baseline (v1.4) rapid test classification. Fine-tune proprietary model in v1.5 based on collected dataset.

#### Context
- Rapid test strips (HIV, Dengue, Syphilis, COVID, HCG) require fast OCR + classification
- Gemini Vision provides immediate accuracy (~88% on test images)
- Timeline: v1.4 foundation, v1.5 fine-tuning

#### Alternatives Considered
1. **Fine-tuned model from day 1:** Higher accuracy (91-95%) but requires 500+ training samples upfront. Timeline slip: 3-4 weeks. Selected: ❌
2. **Manual RT entry only:** 100% accuracy but no automation. Selected: ❌
3. **Gemini baseline + fine-tune in v1.5:** Balanced approach. Get to market with 88% accuracy, improve to 92%+ with feedback loop. Selected: ✅

#### Consequences
- Positive: Faster v1.4 launch, field validation of dataset quality
- Negative: 12% error rate in v1.4 (manual override always available)
- Mitigation: Confidence threshold 0.85; RT always can override

#### Implementation Details
- Phase 5 Tasks 05-03, 05-04: Baseline + dataset collection
- Phase 11 (v1.5 planning): Fine-tuning pipeline (collect 1000+ images, train model, deploy)

#### Status
Proposed (2026-05-07)
