export interface UseNPSSurveyState {
  token: string;
  resposta: 1 | 2 | 3 | 4 | 5 | null;
  comentario: string;
  consentimentoLGPD: boolean;
  loading: boolean;
  error: string | null;
  tokenValidated: boolean;
  comentarioLength: number;
}

export function useNPSSurvey(token: string, labId: string) {
  const [state, setState] = useState<UseNPSSurveyState>({
    token,
    resposta: null,
    comentario: '',
    consentimentoLGPD: false,
    loading: false,
    error: null,
    tokenValidated: false,
    comentarioLength: 0,
  });

  // Validate token on mount
  useEffect(() => {
    async function validateToken() {
      try {
        // Call submitNPSResposta with token (validation only, no response yet)
        // If 200: tokenValidated = true
        // If 400/403: show error
        setState(prev => ({ ...prev, tokenValidated: true }));
      } catch (err) {
        setState(prev => ({
          ...prev,
          error: 'Link expirado ou inválido',
          tokenValidated: false,
        }));
      }
    }
    validateToken();
  }, [token, labId]);

  async function handleSubmit(): Promise<void> {
    // Validate form
    if (!state.resposta || !state.consentimentoLGPD) return;

    setState(prev => ({ ...prev, loading: true, error: null }));
    try {
      // Call Cloud Callable: submitNPSResposta
      const submitNPSRespostaCallable = httpsCallable(
        getFirestore(),
        'submitNPSResposta'
      );
      const result = await submitNPSRespostaCallable({
        token,
        resposta: state.resposta,
        comentario: state.comentario.trim() || null,
        consentimentoLGPD: true,
      });
      // On success: show "Obrigado!" + redirect after 2s
      setState(prev => ({ ...prev, loading: false }));
    } catch (err: any) {
      setState(prev => ({
        ...prev,
        loading: false,
        error: err.message || 'Erro ao enviar resposta',
      }));
    }
  }

  function selectResposta(value: 1 | 2 | 3 | 4 | 5): void {
    setState(prev => ({ ...prev, resposta: value }));
  }

  function updateComentario(text: string): void {
    setState(prev => ({
      ...prev,
      comentario: text.slice(0, 500),
      comentarioLength: Math.min(text.length, 500),
    }));
  }

  function toggleConsent(): void {
    setState(prev => ({
      ...prev,
      consentimentoLGPD: !prev.consentimentoLGPD,
    }));
  }

  function canSubmit(): boolean {
    return (
      state.resposta !== null &&
      state.consentimentoLGPD &&
      state.tokenValidated &&
      !state.loading
    );
  }

  return {
    state,
    handleSubmit,
    selectResposta,
    updateComentario,
    toggleConsent,
    canSubmit,
  };
}
```

- LGPD consent MUST be checked (no submit without it)
- Scale buttons: Use onClick handlers (not radio inputs, for better mobile UX)
- Success: Show "Obrigado!" message, auto-redirect to portal home after 3s (or show button to return)

    filterBy: 'todas',
    searchTerm: '',
  });

  // Load suggestions + user votes on mount
  useEffect(() => {
    const unsubscribe = subscribeToSugestoes(labId, (sugestoes) => {
      const userVotes: Record<string, 1 | -1 | 0> = {};
      sugestoes.forEach(s => {
        userVotes[s.id] = s.votosPorUsuario[pacienteId] || 0;
      });
      setState(prev => ({
        ...prev,
        sugestoes,
        userVotes,
        loading: false,
      }));
    });
    return () => unsubscribe();
  }, [labId, pacienteId]);

  async function handleVote(sugestaoId: string, voto: 1 | -1 | 0): Promise<void> {
    const votarSugestaoCallable = httpsCallable(
      getFirestore(),
      'votarSugestao'
    );
    try {
      await votarSugestaoCallable({
        labId,
        sugestaoId,
        voto,
      });
      // Update local state with new vote
      setState(prev => ({
        ...prev,
        userVotes: {
          ...prev.userVotes,
          [sugestaoId]: voto,
        },
      }));
    } catch (err: any) {
      setState(prev => ({
        ...prev,
        error: 'Erro ao votar. Tente novamente.',
      }));
    }
  }

  function getFilteredSugestoes(): Sugestao[] {
    let filtered = state.sugestoes;

    // Apply filter
    if (state.filterBy === 'minhas') {
      filtered = filtered.filter(s => s.criadoPorId === pacienteId);
    } else if (state.filterBy === 'populares') {
      filtered = filtered.filter(s => s.votos >= 5);
    } else if (state.filterBy === 'recentes') {
      // Already sorted by recency below
    }

    // Apply search
    if (state.searchTerm) {
      const term = state.searchTerm.toLowerCase();
      filtered = filtered.filter(
        s => s.titulo.toLowerCase().includes(term) ||
             s.descricao.toLowerCase().includes(term)
      );
    }

    // Apply sort
    const sorted = [...filtered];
    if (state.sortBy === 'trending') {
      sorted.sort((a, b) => b.trending - a.trending);
    } else if (state.sortBy === 'recente') {
      sorted.sort((a, b) => b.criadoEm.toMillis() - a.criadoEm.toMillis());
    } else if (state.sortBy === 'upvotadas') {
      sorted.sort((a, b) => b.upvotos - a.upvotos);
    }

    return sorted;
  }

  function setSortBy(sort: 'trending' | 'recente' | 'upvotadas'): void {
    setState(prev => ({ ...prev, sortBy: sort }));
  }

  function setFilterBy(filter: 'todas' | 'minhas' | 'populares' | 'recentes'): void {
    setState(prev => ({ ...prev, filterBy: filter }));
  }

  function setSearchTerm(term: string): void {
    setState(prev => ({ ...prev, searchTerm: term }));
  }

  return {
    state,
    sugestoes: getFilteredSugestoes(),
    userVote: (sugestaoId: string) => state.userVotes[sugestaoId] || 0,
    handleVote,
    setSortBy,
    setFilterBy,
    setSearchTerm,
  };
}
```
