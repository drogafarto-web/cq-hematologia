```firestore-rules
match /labs/{labId}/reclamacoes/{reclamacaoId} {
  allow read: if
    request.auth != null &&
    (
      (isMemberOfLab(labId) &&
       (isQualityManager(labId) || isRT(labId))) ||
      (getPatientSession(labId) != null &&
       resource.data.pacienteId == getPatientSession(labId).pacienteId)
    );

  allow create: if
    false; // Callable only (criarReclamacao)

  allow update: if
    false; // Callable only (transitarReclamacao)

  allow delete: if
    false; // Soft-delete only (RN-06)
}

// RCA subcollection — only RT/Qualidade can read/write
match /labs/{labId}/reclamacoes/{reclamacaoId}/rca/{rcaId} {
  allow read, write: if
    request.auth != null &&
    isMemberOfLab(labId) &&
    (isQualityManager(labId) || isRT(labId));

  allow delete: if false;
}

// Versions (immutable audit trail) — read-only
match /labs/{labId}/reclamacoes-versions/{versionId} {
  allow read: if
    request.auth != null &&
    isMemberOfLab(labId);

  allow create, update, delete: if false;
}

// LGPD audit log — immutable
match /labs/{labId}/lgpd-audit/{logId} {
  allow read: if
    request.auth != null &&
    (isAdminOrOwner(labId) || isPrivacyOfficer(labId));

  allow create: if
    false; // Cloud Function only

  allow update, delete: if false;
}

// Helper functions for complaint RLS
function isRT(labId) {
  return get(/databases/$(database)/documents/labs/$(labId)).data.members[request.auth.uid].roles contains 'RT';
}

function isQualityManager(labId) {
  return get(/databases/$(database)/documents/labs/$(labId)).data.members[request.auth.uid].roles contains 'Qualidade';
}

function isPrivacyOfficer(labId) {
  return get(/databases/$(database)/documents/labs/$(labId)).data.members[request.auth.uid].roles contains 'PrivacyOfficer';
}

function getPatientSession(labId) {
  return get(/databases/$(database)/documents/labs/$(labId)/patientSessions/$(request.auth.uid)).data;
}
```

```typescript
export const FormReclamacaoSchema = z.object({
  motivo: MotivoEnum,
  descricao: z.string()

    type: z.enum(['image', 'pdf', 'documento']),
  })).optional(),
  consentimentoLGPD: z.boolean().refine(v => v === true, 'Consentimento obrigatório'),
});

export type FormReclamacaoInput = z.infer<typeof FormReclamacaoSchema>;

// RCA validation
export const RCANivelSchema = z.object({
  nivel: z.union([z.literal(1), z.literal(2), z.literal(3), z.literal(4), z.literal(5)]),
  pergunta: z.string().min(10, 'Pergunta é obrigatória'),
  resposta: z.string().min(20, 'Resposta detalhada obrigatória'),
});

export const RCASchema = z.object({
  niveis: z.array(RCANivelSchema)
    .refine(arr => arr.length >= 4, 'RCA deve ter no mínimo 4 níveis completos'),
  causaRaiz: z.string().min(30, 'Causa raiz deve ser descritiva (min 30 chars)'),
  acaoCorretivaPropos ta: z.string().min(20, 'Ação corretiva obrigatória'),
});

// Severity classification via keyword heuristics
export function classificarSeveridadeHeuristica(
  descricao: string,
  motivo: string
): { severidade: Severidade; confianca: number; palavrasChave: string[] } {
  const palavrasAlta = [
    'óbito', 'morte', 'crítico', 'crítica', 'risco de vida',
    'falha na comunicação resultado crítico', 'erro de metodologia grave',
    'reagente vencido usado', 'contaminação', 'falha no resultado',
    'paciente prejudicado', 'internação por resultado errado',
  ];

  const palavrasMedia = [
    'atraso na entrega', 'resultado incompleto', 'dúvida no resultado',
    'discrepância em controle', 'reagente', 'equipamento com problema',
    'treinamento deficiente', 'comunicação confusa',
  ];

  const descricaoLower = descricao.toLowerCase();
  let palavrasEncontradas: string[] = [];
  let pontuacao = 0;

  // Check for high-severity keywords
  for (const palavra of palavrasAlta) {
    if (descricaoLower.includes(palavra.toLowerCase())) {
      palavrasEncontradas.push(palavra);
      pontuacao += 3;
    }
  }

  // Check for medium-severity keywords
  for (const palavra of palavrasMedia) {
    if (descricaoLower.includes(palavra.toLowerCase())) {
      palavrasEncontradas.push(palavra);
      pontuacao += 1;
    }
  }

  // Boost if motivo is 'resultado' or 'metodologia'
  if (motivo === 'resultado' || motivo === 'metodologia') pontuacao += 1;

  // Description length as proxy for severity (longer = more serious)
  if (descricao.length > 300) pontuacao += 0.5;

  // Classify
  let severidade = Severidade.BAIXA;
  let confianca = 0.5;

  if (pontuacao >= 3) {
    severidade = Severidade.ALTA;
    confianca = Math.min(0.9, 0.6 + (pontuacao - 3) * 0.1);
  } else if (pontuacao >= 1) {
    severidade = Severidade.MEDIA;
    confianca = Math.min(0.8, 0.5 + pontuacao * 0.15);
  } else {
    severidade = Severidade.BAIXA;
    confianca = Math.max(0.3, 0.8 - descricao.length / 1000);
  }

  return {
    severidade,
    confianca: Math.round(confianca * 100) / 100,
    palavrasChave: [...new Set(palavrasEncontradas)],
  };
}

// Validate complaint before Firestore write
export function validateReclamacao(data: any) {
  try {
    FormReclamacaoSchema.parse(data);
    return { valid: true, errors: [] };
  } catch (err: any) {
    if (err instanceof z.ZodError) {
      return { valid: false, errors: err.errors.map(e => e.message) };
    }
    return { valid: false, errors: [err.message || 'Validação falhou'] };
  }
}

// Validate RCA before Resolvida transition
export function validateRCA(rca: any) {
  try {
    RCASchema.parse(rca);
    return { valid: true, errors: [] };
  } catch (err: any) {
    if (err instanceof z.ZodError) {
      return { valid: false, errors: err.errors.map(e => e.message) };
    }
    return { valid: false, errors: [err.message || 'RCA inválida'] };
  }
}

// Should complaint auto-trigger NC creation?
export function deveriacriarNCAutomatico(
  severidade: Severidade,
  descricao: string
): boolean {
  return severidade === Severidade.ALTA && descricao.length >= 100;
}
```

Input:

```typescript

  anexoURLs?: string[]; // Pre-signed URLs from client file upload
```

Logic:

1. Verify reCAPTCHA v3 token (call reCAPTCHA API, score >= 0.5)
2. Validate input via FormReclamacaoSchema
3. Get patient session from context (request.auth.uid must be patient with valid session)
4. Classify severity via `classificarSeveridadeHeuristica(descricao)`
5. Generate IDs: reclamacaoId = UUID, chainHash = sha256(prev_chainHash + reclamacaoId + timestamp)
6. Create Reclamacao doc in `/labs/{labId}/reclamacoes/{reclamacaoId}`:
   - Set: motivo, descricao, severity, classificacaoConfianca, status='Nova', consentimentoLGPD=true
   - Set: pacienteId = request.auth.uid, pacienteCPF = sha256(patient.cpf)
   - Set: assinatura = { hash: sha256(doc), operatorId: 'patient-system', ts: now }
   - Set: chainHash, dataVencimentoSLA = now + 30 days, criadoEm = now
7. Create audit log in `/labs/{labId}/lgpd-audit/{auditId}`:
   - action: 'reclamacao-criada', recursoId: reclamacaoId, piiFields: ['descricao', 'pacienteId'], consentimento: true, timestamp
8. Return: { id: reclamacaoId, status: 'Nova', severidade, success: true }

Error cases:

- reCAPTCHA score < 0.5 → 403 "Please try again"
- Validation fails → 400 with error list
- No patient session → 401 "Unauthorized"
- Patient has >10 reclamacoes in last 24h → 429 "Rate limited"

**Callable 2: transitarReclamacao**

Input:

```typescript
{
  labId: string;
  reclamacaoId: string;
  novoStatus: ReclamacaoStatus;
  rca?: RCA; // Provided if status='RCA'
  justificativa?: string; // For rejection
}
```

Logic:

1. Load current Reclamacao from Firestore
2. Validate transition is allowed (state machine rules):
   - Nova → Analisando: any RT/Qualidade can do
   - Analisando → RCA: RCA object required, validate via validateRCA()
   - RCA → Resolvida: RCA must be complete (4+ levels), only if severity='alta'
   - Resolvida → Comunicada: status change only (no additional validation)
   - Comunicada → Fechada: allowed after 14 days OR NPS response received
   - Other transitions: rejected (400)
3. Verify user has permission (isMemberOfLab + isRT/isQualityManager)
4. Create version snapshot (immutable copy of old doc in `/reclamacoes-versions/{versionId}`)
5. Update Reclamacao doc:
   - status = novoStatus
   - if novoStatus='RCA': rca = validated RCA object
   - if novoStatus='Resolvida': atualizadoEm = now, (don't set resolvida timestamp yet — Plan 11-02 does it via trigger)
   - assinatura = new signature { hash: sha256(updated doc), operatorId: request.auth.uid, ts: now }
   - chainHash = new chain hash
6. Create audit log entry
7. Return: { id: reclamacaoId, novoStatus, success: true }

Error cases:

- Invalid state transition → 400 "Transição não permitida"
- Validation fails (RCA incomplete) → 400 with details
- User lacks permission → 403
- Reclamacao not found → 404

- Batch write: version snapshot + main doc update + audit log (atomic)
- Chain hash: computeChainHash(prevChainHash, reclamacaoId, operation, timestamp, uid)
- Audit log: immutable + indexed by reclamacaoId for fast audit queries
- Error messages: vague for security (don't leak validation details to clients who shouldn't see)
- Region: southamerica-east1
