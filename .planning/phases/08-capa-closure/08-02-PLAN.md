```typescript
  statusUpdatedAt: number;
  
  alertsSent: { days: number; sentAt: number }[];  // e.g., [{ days: 30, ... }, { days: 7, ... }, { days: 0 (overdue), ... }]

export interface CalibracaoAlert {
  id: string;
  labId: string;
  calibracaoId: string;
  equipamentoName: string;
  daysUntilOverdue: number;
  alertType: 'warning-30d' | 'warning-7d' | 'overdue';
  emailSentAt: number;
  recipients: string[];  // Manager email addresses
}

export function calculateStatus(nextDueDate: number): CalibracaoRecord['status'] {
  const now = Date.now();
  const MS_PER_DAY = 1000 * 60 * 60 * 24;
  const daysRemaining = (nextDueDate - now) / MS_PER_DAY;
  
  if (daysRemaining < 0) return 'overdue';
  if (daysRemaining < 7) return 'warning-7d';
  if (daysRemaining < 30) return 'warning-30d';
  return 'in-date';
}
```
