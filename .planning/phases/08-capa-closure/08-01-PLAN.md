```typescript
// Helper: Calculate days remaining
export function daysRemaining(deadlineDate: number): number {
  const now = Date.now();
  const MS_PER_DAY = 1000 * 60 * 60 * 24;
  return Math.ceil((deadlineDate - now) / MS_PER_DAY);
}

// Helper: Validate state transition
export function isValidStateTransition(from: CapaState, to: CapaState): boolean {
  const validTransitions: Record<CapaState, CapaState[]> = {
    'open': ['in-progress'],
    'in-progress': ['evidence-submitted'],
    'evidence-submitted': ['auditor-reviewing'],
    'auditor-reviewing': ['closed', 'in-progress'], // auditor can request more evidence
    'closed': [], // final state, no transitions
  };
  return validTransitions[from]?.includes(to) ?? false;
}

// Helper: Calculate state color for UI
export function getStateColor(state: CapaState): string {
  const colors: Record<CapaState, string> = {
    'open': 'bg-yellow-100 text-yellow-800',
    'in-progress': 'bg-blue-100 text-blue-800',
    'evidence-submitted': 'bg-purple-100 text-purple-800',
    'auditor-reviewing': 'bg-orange-100 text-orange-800',
    'closed': 'bg-green-100 text-green-800',
  };
  return colors[state];
}
```

```firestore-rules
// CAPA Tracking RLS
match /labs/{labId}/capa-tracking/{capaId} {
  // Creation: QA manager or admin
  allow create: if
    request.auth != null &&
    (isQualityManager(labId) || isAdminOrOwner(labId)) &&
    request.resource.data.state == 'open' &&
    request.resource.data.createdAt == request.time.toMillis() &&
    request.resource.data.evidence == [] &&
    request.resource.data.rfiLog == [] &&
    request.resource.data.auditLog == [];

  // Read: QA manager, auditor-assigned staff, lab admin
  allow read: if
    request.auth != null &&
    (isMemberOfLab(labId) || isAuditorStaff(labId));

  // State transitions: only via callable (server-side validation)
  // Client update is not allowed; use updateCapaState callable
  allow update: if false;

  // Soft delete: QA manager (marks deletedAt)
  allow delete: if
    request.auth != null &&
    (isQualityManager(labId) || isAdminOrOwner(labId));

  // Audit log (sub-collection, immutable append)
  match /audit-log/{logId} {
    allow create: if
      request.auth != null &&
      request.resource.data.timestamp == request.time.toMillis();
    allow read: if
      request.auth != null &&
      isMemberOfLab(labId);
    allow write, delete: if false; // immutable
  }

  // RFI log (sub-collection, append-only)
  match /rfi-log/{rfiId} {
    allow create: if
      request.auth != null &&
      (isQualityManager(labId) || isAdminOrOwner(labId)) &&
      request.resource.data.askedAt == request.time.toMillis();
    allow read: if
      request.auth != null &&
      isMemberOfLab(labId);
    allow update: if
      request.auth != null &&
      (isQualityManager(labId) || isAdminOrOwner(labId)) &&
      request.resource.data.respondedAt == request.time.toMillis();
    allow delete: if false; // immutable (soft-delete via soft-delete pattern)
  }

  // Evidence uploads (immutable, append-only)
  match /evidence/{evidenceId} {
    allow create: if
      request.auth != null &&
      (isQualityManager(labId) || isAdminOrOwner(labId)) &&
      request.resource.data.hash.size() == 64 &&
      request.resource.data.signature.operatorId == request.auth.uid &&
      request.resource.data.uploadedAt == request.time.toMillis();
    allow read: if
      request.auth != null &&
      isMemberOfLab(labId);
    allow write, delete: if false; // immutable
  }
}

// Helper: Check if user is QA manager
function isQualityManager(labId) {
  return get(/databases/$(database)/documents/labs/$(labId)/personnel/designacoes)
    .data.gerente_qualidade == request.auth.uid;
}

// Helper: Check if user is auditor staff (allow read-only access to CAPA docs)
function isAuditorStaff(labId) {
  return get(/databases/$(database)/documents/labs/$(labId)/config)
    .data.auditor_emails[request.auth.email] == true;
}
```

5. **Soft delete**: deletedAt marker (never hard delete)
6. **Audit trail**: every RFI + evidence upload creates audit log entry

```typescript
import {
  collection,
  doc,
  query,
  where,
  orderBy,
  limit,
  onSnapshot,
  getDoc,
  getDocs,
  Timestamp,
  Unsubscribe,
} from 'firebase/firestore';
import { getFirestore } from 'firebase/firestore';
import { CapaDocument, daysRemaining } from '../types';

export class CapaService {
  private db = getFirestore();

  // Read single CAPA
  async getCapaById(labId: string, capaId: string): Promise<CapaDocument | null> {
    const docRef = doc(this.db, `labs/${labId}/capa-tracking`, capaId);
    const snap = await getDoc(docRef);
    if (!snap.exists()) return null;
    
    const data = snap.data();
    // Calculate daysRemaining on every read (never cached)
    return {
      id: snap.id,
      ...data,
      daysRemaining: daysRemaining(data.deadlineDate),
    } as CapaDocument;
  }

  // List all CAPAs for lab (with optional state filter)
  async listCapas(
    labId: string,
    filterState?: string,
    sortBy: 'daysRemaining' | 'createdAt' | 'severity' = 'daysRemaining'
  ): Promise<CapaDocument[]> {
    const q = filterState && filterState !== 'all'
      ? query(
          collection(this.db, `labs/${labId}/capa-tracking`),
          where('state', '==', filterState),
          where('deletedAt', '==', undefined),
          orderBy(sortBy === 'daysRemaining' ? 'deadlineDate' : sortBy === 'createdAt' ? 'createdAt' : 'severity'),
          limit(100)
        )
      : query(
          collection(this.db, `labs/${labId}/capa-tracking`),
          where('deletedAt', '==', undefined),
          orderBy(sortBy === 'daysRemaining' ? 'deadlineDate' : sortBy === 'createdAt' ? 'createdAt' : 'severity'),
          limit(100)
        );

    const snap = await getDocs(q);
    return snap.docs.map(doc => {
      const data = doc.data();
      return {
        id: doc.id,
        ...data,
        daysRemaining: daysRemaining(data.deadlineDate),
      } as CapaDocument;
    });
  }

  // Subscribe to CAPA list (realtime)
  subscribeToCapas(
    labId: string,
    onUpdate: (capaList: CapaDocument[]) => void,
    filterState?: string
  ): Unsubscribe {
    const q = filterState && filterState !== 'all'
      ? query(
          collection(this.db, `labs/${labId}/capa-tracking`),
          where('state', '==', filterState),
          where('deletedAt', '==', undefined),
          orderBy('deadlineDate')
        )
      : query(
          collection(this.db, `labs/${labId}/capa-tracking`),
          where('deletedAt', '==', undefined),
          orderBy('deadlineDate')
        );

    return onSnapshot(q, (snap) => {
      const capaList = snap.docs.map(doc => {
        const data = doc.data();
        return {
          id: doc.id,
          ...data,
          daysRemaining: daysRemaining(data.deadlineDate),
        } as CapaDocument;
      });
      onUpdate(capaList);
    });
  }

  // Get evidence list for CAPA
  async getCapaEvidence(labId: string, capaId: string) {
    const capaRef = doc(this.db, `labs/${labId}/capa-tracking`, capaId);
    const snap = await getDoc(capaRef);
    return snap.exists() ? (snap.data() as CapaDocument).evidence : [];
  }

  // Get RFI log for CAPA
  async getCapaRFILog(labId: string, capaId: string) {
    const capaRef = doc(this.db, `labs/${labId}/capa-tracking`, capaId);
    const snap = await getDoc(capaRef);
    return snap.exists() ? (snap.data() as CapaDocument).rfiLog : [];
  }
}

export const capaService = new CapaService();
```

```typescript
import { useState } from 'react';
import { uploadBytes, ref } from 'firebase/storage';
import { getStorage } from 'firebase/storage';

export function CapaEvidenceUpload({ labId, capaId, onUploadSuccess }) {
  const [file, setFile] = useState<File | null>(null);
  const [description, setDescription] = useState('');
  const [loading, setLoading] = useState(false);
  const [hash, setHash] = useState<string | null>(null);
  const [error, setError] = useState<string | null>(null);

  const computeHash = async (file: File): Promise<string> => {
    const buffer = await file.arrayBuffer();
    const hashBuffer = await crypto.subtle.digest('SHA-256', buffer);
    const hashArray = Array.from(new Uint8Array(hashBuffer));
    return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
  };

  const handleFileChange = async (selectedFile: File) => {
    setFile(selectedFile);
    setError(null);
    
    if (selectedFile.size > 10 * 1024 * 1024) {
      setError('File size exceeds 10MB limit');
      setFile(null);
      return;
    }

    try {
      const computedHash = await computeHash(selectedFile);
      setHash(computedHash);
    } catch (err: any) {
      setError('Failed to compute file hash');
    }
  };

  const handleUpload = async () => {
    if (!file || !hash || !description) {
      setError('File, hash, and description required');
      return;
    }

    setLoading(true);
    try {
      // 1. Upload to Cloud Storage
      const storage = getStorage();
      const storageRef = ref(
        storage,
        `capa-evidence/${labId}/${capaId}/${Date.now()}_${file.name}`
      );
      await uploadBytes(storageRef, file);

      // 2. Create signature
      const signature = {
        hash: hash,
        operatorId: /* from auth state */,
        ts: Date.now(),
      };

      // 3. Call Cloud Function to log in Firestore
      const uploadEvidenceCallable = httpsCallable(
        getFirestore(),
        'uploadCapaEvidence'
      );
      const result = await uploadEvidenceCallable({
        labId,
        capaId,
        fileName: file.name,
        fileSize: file.size,
        mimeType: file.type,
        storagePath: storageRef.fullPath,
        description,
        hash,
        signature,
      });

      onUploadSuccess(result.data);
      setFile(null);
      setDescription('');
      setHash(null);
    } catch (err: any) {
      setError(err.message || 'Upload failed');
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="bg-[#141417] border border-gray-700 rounded-lg p-6">
      <h3 className="text-white text-lg font-semibold mb-4">Upload Evidence</h3>
      
      {/* Drag-drop zone */}
      <div
        className="border-2 border-dashed border-gray-600 rounded-lg p-8 text-center cursor-pointer hover:border-gray-400 transition"
        onDrop={(e) => {
          e.preventDefault();
          const files = e.dataTransfer.files;
          if (files.length > 0) handleFileChange(files[0]);
        }}
        onDragOver={(e) => e.preventDefault()}
      >
        {file ? (
          <p className="text-white">{file.name} ({(file.size / 1024 / 1024).toFixed(2)} MB)</p>
        ) : (
          <>
            <p className="text-gray-400">Drag and drop your file here</p>
            <p className="text-gray-500 text-sm mt-2">or use file picker</p>
            <input
              type="file"
              className="hidden"
              onChange={(e) => {
                if (e.target.files?.[0]) handleFileChange(e.target.files[0]);
              }}
            />
          </>
        )}
      </div>

      {/* Hash display */}
      {hash && (
        <div className="mt-4 p-3 bg-green-900 bg-opacity-30 rounded border border-green-700">
          <p className="text-green-400 text-sm font-mono truncate">Hash: {hash.substring(0, 32)}...</p>
        </div>
      )}

      {/* Description */}
      <textarea
        placeholder="Why is this evidence important? (e.g., 'Calibration certificate for equipment XYZ')"
        value={description}
        onChange={(e) => setDescription(e.target.value)}
        className="w-full mt-4 p-3 bg-gray-800 text-white rounded border border-gray-600 focus:border-blue-500 outline-none"
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| **QA Manager → CAPA Firestore** | QA manager creates CAPA with root cause analysis; Rules enforce create-only on 'open' state |
| **Client → Cloud Storage** | QA manager uploads evidence file; client computes hash, Cloud Function verifies hash match before accepting |
| **Evidence File → Audit Trail** | File immutable after upload; hash + signature stored in Firestore (prevents tampering) |
| **CAPA → Auditor Email** | State transition to 'auditor-reviewing' triggers email; auditor receives RFI + link to dashboard |
| **Auditor RFI Response** | Auditor responds via email or form; Cloud Function validates response + logs in rfiLog (immutable) |
| **State Transition → CAPA Doc** | Only valid state transitions allowed (server-side validation); invalid transitions blocked by Rules + callable |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-01 | Spoofing | QA Manager auth | **Mitigate** | User ID from request.auth.uid (Firebase Auth), Rules check isQualityManager via personnel/designacoes lookup |
| T-08-02 | Spoofing | Auditor email | **Mitigate** | Auditor email hard-coded in lab config (admin-set), no user-supplied email in RFI form |
| T-08-03 | Tampering | Evidence file | **Mitigate** | SHA-256 hash computed client-side, verified server-side before storage; hash stored immutable in Firestore |
| T-08-04 | Tampering | State transition | **Mitigate** | Valid transitions hard-coded in type helpers; Rules enforce isValidStateTransition before accepting update callable |
| T-08-05 | Tampering | CAPA doc | **Mitigate** | Firestore soft-delete only (no hard delete); audit log append-only (no edit/delete) |
