   export function useDPIA(labId: string) {
     const [dpia, setDpia] = useState<DPIA | null>(null);
     const [loading, setLoading] = useState(true);
     const [error, setError] = useState<Error | null>(null);

     useEffect(() => {
       const unsubscribe = subscribeDPIA(
         labId,
         (data) => {
           setDpia(data);
           setLoading(false);
         },
         (err) => {
           setError(err);
           setLoading(false);
         },
       );
       return unsubscribe;
     }, [labId]);

     return { dpia, loading, error };
   }
   ```

3. **Hook: usePrivacyPolicy** (`src/features/lgpd/hooks/usePrivacyPolicy.ts`):

   export function usePrivacyPolicy(labId: string) {
     const [policy, setPolicy] = useState<PolicyVersion | null>(null);
     const [loading, setLoading] = useState(true);

     useEffect(() => {
       const unsubscribe = subscribeCurrentPolicy(labId, (data) => {
         setPolicy(data);
         setLoading(false);
       });
       return unsubscribe;
     }, [labId]);

     return { policy, loading };
   }
   ```

1. **OTP utility** (`functions/src/utils/otp.ts`):

   - Export `sendOTPEmail(email: string, otp: string, labName: string): Promise<void>` — uses SendGrid (existing integration)

2. **Function** (`functions/src/modules/lgpd/deleteTitularData.ts`):

   ```typescript
   export const deleteTitularData = onCall<
     {
       cpf: string;
       otp: string;
       otpToken: string;
       motivo: string;
     },
     {
       deletedDocCount: number;
       auditRecordId: string;
     }
   >(async (request) => {
     // 1. Validate OTP
     const otpDoc = await admin.firestore().doc(`/otps/${request.data.otpToken}`).get();
     if (
       !otpDoc.exists ||
       otpDoc.data()!.otp !== request.data.otp ||
       otpDoc.data()!.expiresAt < admin.firestore.Timestamp.now()
     ) {
       throw new HttpsError('invalid-argument', 'OTP inválido ou expirado');
     }

     // 2. Find all docs with matching cpfHash
     const cpfHash = hashCPF(request.data.cpf); // or lookup from request.auth context if available
     const query = admin.firestore().collection('users').where('cpfHash', '==', cpfHash);
     const users = await query.get();

     if (users.empty) {
       throw new HttpsError('not-found', 'CPF não encontrado no sistema');
     }

     // 3. For each matching doc, zero PII fields
     const batch = admin.firestore().batch();
     let count = 0;

     users.forEach((userDoc) => {
       batch.update(userDoc.ref, {
         nome: '',
         email: '',
         telefone: '',
         endereco: '',
         // Keep: cpfHash (for dedup), id, criadoEm, deletadoEm, chain-hash fields
         atualizadoEm: admin.firestore.FieldValue.serverTimestamp(),
         piiZeradoEm: admin.firestore.FieldValue.serverTimestamp(),
         piiZeradoPor: request.auth?.uid || 'system',
       });
       count++;
     });

     // 4. Commit batch (preserves existing LogicalSignature fields)
     await batch.commit();

     // 5. Log audit event (write to /auditLogs with LogicalSignature)
     const auditRecord = {
       id: admin.firestore.randomId(),
       tipo: 'pii_deletion_request',
       cpfHash: cpfHash,
       docCount: count,
       motivo: request.data.motivo,
       requestedBy: request.auth?.uid || 'anonymous',
       requestedAt: admin.firestore.Timestamp.now(),
       status: 'completed',
       logicalSignature: {
         hash: computeHash({ cpfHash, count, timestamp: Date.now() }),
         operatorId: request.auth?.uid || 'system',
         ts: admin.firestore.Timestamp.now(),
       },
     };
     await admin.firestore().collection('auditLogs').doc(auditRecord.id).set(auditRecord);

     // 6. Clean up OTP
     await admin.firestore().doc(`/otps/${request.data.otpToken}`).delete();

     return {
       deletedDocCount: count,
       auditRecordId: auditRecord.id,
     };
   });
   ```
