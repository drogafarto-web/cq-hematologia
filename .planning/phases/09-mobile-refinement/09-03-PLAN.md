```typescript
/**
 * Performance measurement utilities for HC Quality mobile
 * Tracks: startup time, screen transitions, API latency, memory
 */

interface PerformanceMetrics {
  startupTime?: number; // ms from app init to home screen visible
  lcpTime?: number; // ms to first contentful paint
  screenTransitions: Map<string, number>; // screen name -> latency
  callableLatencies: Map<string, number>; // function name -> latency
  memoryUsage: {
    atStartup: number; // bytes
    current: number;
    peak: number;
  };
}

let metrics: PerformanceMetrics = {
  screenTransitions: new Map(),
  callableLatencies: new Map(),
  memoryUsage: { atStartup: 0, current: 0, peak: 0 },
};

/**
 * Mark app startup beginning (call in App.tsx useEffect)
 */
export function startAppStartupTimer() {
  performance.mark('app-startup-start');
}

/**
 * Mark when home screen is first painted (call in HomeScreen useEffect)
 */
export function markHomeScreenReady() {
  performance.mark('app-startup-end');
  const measure = performance.measure('app-startup', 'app-startup-start', 'app-startup-end');
  metrics.startupTime = measure.duration;
  console.log(`[Performance] Startup time: ${measure.duration.toFixed(2)}ms`);
  return measure.duration;
}

/**
 * Measure screen navigation latency
 * Call when user taps a navigation button, and again when new screen renders
 */
export function measureScreenTransition(screenName: string, duration: number) {
  metrics.screenTransitions.set(screenName, duration);
  console.log(`[Performance] Screen transition [${screenName}]: ${duration.toFixed(2)}ms`);
}

/**
 * Measure Cloud Function callable latency
 */
export async function measureCallableLatency<T>(
  functionName: string,
  fn: () => Promise<T>,
): Promise<T> {
  const start = performance.now();
  try {
    const result = await fn();
    const duration = performance.now() - start;
    metrics.callableLatencies.set(functionName, duration);
    console.log(`[Performance] Callable [${functionName}]: ${duration.toFixed(2)}ms`);
    return result;
  } catch (err) {
    const duration = performance.now() - start;
    console.error(
      `[Performance] Callable [${functionName}] failed after ${duration.toFixed(2)}ms:`,
      err,
    );
    throw err;
  }
}

/**
 * Get current memory usage (bytes)
 * Note: Requires native bridge on Android; returns estimate on JS thread
 */
export function getMemoryUsage(): number {
  // Rough estimate: React Native DevTools can provide actual values
  // For now, return placeholder
  return Math.random() * 200_000_000; // Placeholder: 0-200 MB
}

/**
 * Record memory snapshot
 */
export function recordMemorySnapshot() {
  const current = getMemoryUsage();
  if (metrics.memoryUsage.atStartup === 0) {
    metrics.memoryUsage.atStartup = current;
  }
  metrics.memoryUsage.current = current;
  if (current > metrics.memoryUsage.peak) {
    metrics.memoryUsage.peak = current;
  }
}

/**
 * Get all metrics (for logging or export)
 */
export function getMetrics(): PerformanceMetrics {
  recordMemorySnapshot();
  return { ...metrics };
}

/**
 * Reset metrics (for new session)
 */
export function resetMetrics() {
  metrics = {
    screenTransitions: new Map(),
    callableLatencies: new Map(),
    memoryUsage: { atStartup: 0, current: 0, peak: 0 },
  };
}

/**
 * Log metrics to console (for debugging)
 */
export function logMetrics() {
  const current = getMetrics();
  console.log('=== Performance Metrics ===');
  console.log(`Startup: ${current.startupTime?.toFixed(2)}ms`);
  console.log(
    `Memory: ${(current.memoryUsage.atStartup / 1_000_000).toFixed(1)}MB → ${(current.memoryUsage.current / 1_000_000).toFixed(1)}MB`,
  );
  console.log(`Screen transitions:`, Object.fromEntries(current.screenTransitions));
  console.log(`Callable latencies:`, Object.fromEntries(current.callableLatencies));
}
```

   ```typescript
   interface CIQCardProps {
     imageUrl?: string;
     title: string;
   }

   export function CIQCard({ imageUrl, title }: CIQCardProps) {
     const [isImageLoading, setIsImageLoading] = useState(!!imageUrl);

     return (
       <View className="dark:bg-gray-800 rounded-lg p-4 mb-4">
         {imageUrl && (
           <View className="mb-3">
             {isImageLoading && (
               <View className="w-full h-32 justify-center items-center dark:bg-gray-700 rounded">
                 <ActivityIndicator color="#f5f5f7" />
               </View>
             )}
             <Image
               source={{ uri: imageUrl }}
               style={{
                 width: '100%',
                 height: 128,
                 borderRadius: 8,
                 display: isImageLoading ? 'none' : 'flex',
               }}
               onLoadStart={() => setIsImageLoading(true)}
               onLoadEnd={() => setIsImageLoading(false)}
               onError={() => setIsImageLoading(false)}
             />
           </View>
         )}
         <Text className="text-white font-semibold">{title}</Text>
       </View>
     );
   }
   ```
