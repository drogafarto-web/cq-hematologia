1. Create `src/hooks/useOfflineMode.ts`:

   ```typescript
   import { useEffect, useState } from 'react';
   import { useNetInfo } from '@react-native-community/netinfo';
   import AsyncStorage from '@react-native-async-storage/async-storage';

   interface OfflineModeState {
     isOnline: boolean;
     isSyncing: boolean;
     queuedWrites: number;
   }

   export function useOfflineMode(): OfflineModeState {
     const netInfo = useNetInfo();
     const [isSyncing, setIsSyncing] = useState(false);
     const [queuedWrites, setQueuedWrites] = useState(0);

     const isOnline = netInfo.isConnected === true && netInfo.isInternetReachable === true;

     // Load queued write count on mount
     useEffect(() => {
       const loadQueuedCount = async () => {
         try {
           const queue = await AsyncStorage.getItem('@hcquality/offlineQueue');
           const items = queue ? JSON.parse(queue) : [];
           setQueuedWrites(items.length);
         } catch (err) {
           console.warn('[useOfflineMode] Failed to load queue:', err);
         }
       };
       loadQueuedCount();
     }, []);

     // When coming online, trigger sync
     useEffect(() => {
       if (isOnline && queuedWrites > 0) {
         setIsSyncing(true);
         // Sync will be handled by useOfflineSync hook
       } else if (isOnline) {
         setIsSyncing(false);
       }
     }, [isOnline, queuedWrites]);

     return {
       isOnline,
       isSyncing,
       queuedWrites,
     };
   }
   ```

2. Create `src/hooks/useOfflineSync.ts`:

   ```typescript
   import { useEffect } from 'react';
   import AsyncStorage from '@react-native-async-storage/async-storage';
   import { useOfflineMode } from './useOfflineMode';

   export function useOfflineSync() {
     const { isOnline, queuedWrites } = useOfflineMode();

     useEffect(() => {
       if (!isOnline || queuedWrites === 0) return;

       const syncQueue = async () => {
         try {
           const queueStr = await AsyncStorage.getItem('@hcquality/offlineQueue');
           const items = queueStr ? JSON.parse(queueStr) : [];

           for (const item of items) {
             try {
               // Call Firebase callable to process write
               const callable = firebase.functions().httpsCallable(item.functionName);
               await callable(item.payload);
             } catch (err) {
               console.error(`[OfflineSync] Failed to sync ${item.functionName}:`, err);
               // Keep in queue, retry on next connection
               throw err;
             }
           }

           // Clear queue on success
           await AsyncStorage.setItem('@hcquality/offlineQueue', '[]');
           console.log('[OfflineSync] Queue synced successfully');
         } catch (err) {
           console.error('[OfflineSync] Sync failed, will retry:', err);
         }
       };

       syncQueue();
     }, [isOnline, queuedWrites]);
   }
   ```
