   ```typescript
   // Add to existing hook
   const LOCKOUT_ATTEMPTS_KEY = '@hcquality/pinAttempts';
   const LOCKOUT_TIME_KEY = '@hcquality/lockoutUntil';
   const MAX_ATTEMPTS = 5;
   const LOCKOUT_DURATION_MS = 5 * 60 * 1000; // 5 minutes

   export interface BiometricAuthState {
     isBiometricAvailable: boolean;
     isBiometricEnabled: boolean;
     biometryType: string | null;
     isLoading: boolean;
     // NEW: Lockout tracking
     isLockedOut: boolean;
     lockoutRemainingMs: number; // milliseconds until unlock
     failedAttempts: number;
   }

   export function useBiometricAuth() {
     // Existing state...
     const [isLockedOut, setIsLockedOut] = useState(false);
     const [lockoutRemainingMs, setLockoutRemainingMs] = useState(0);
     const [failedAttempts, setFailedAttempts] = useState(0);

     // NEW: Initialize lockout state on mount
     useEffect(() => {
       let cancelled = false;
       const initLockout = async () => {
         try {
           const [attemptsStr, lockoutUntilStr] = await Promise.all([
             AsyncStorage.getItem(LOCKOUT_ATTEMPTS_KEY),
             AsyncStorage.getItem(LOCKOUT_TIME_KEY),
           ]);
           if (cancelled) return;

           const attempts = parseInt(attemptsStr ?? '0', 10);
           const lockoutUntil = lockoutUntilStr ? parseInt(lockoutUntilStr, 10) : null;

           setFailedAttempts(attempts);

           if (lockoutUntil && Date.now() < lockoutUntil) {
             setIsLockedOut(true);
             setLockoutRemainingMs(lockoutUntil - Date.now());
           } else {
             setIsLockedOut(false);
             setLockoutRemainingMs(0);
           }
         } catch (err) {
           console.warn('[useBiometricAuth] lockout init error:', err);
         }
       };
       initLockout();
       return () => {
         cancelled = true;
       };
     }, []);

     // NEW: Lockout timer countdown
     useEffect(() => {
       if (!isLockedOut || lockoutRemainingMs <= 0) return;
       const interval = setInterval(() => {
         setLockoutRemainingMs((prev) => {
           if (prev <= 1000) {
             setIsLockedOut(false);
             AsyncStorage.removeItem(LOCKOUT_TIME_KEY);
             return 0;
           }
           return prev - 1000;
         });
       }, 1000);
       return () => clearInterval(interval);
     }, [isLockedOut, lockoutRemainingMs]);

     // NEW: Track failed PIN attempt
     const recordFailedAttempt = async () => {
       const newAttempts = failedAttempts + 1;
       setFailedAttempts(newAttempts);
       await AsyncStorage.setItem(LOCKOUT_ATTEMPTS_KEY, newAttempts.toString());

       if (newAttempts >= MAX_ATTEMPTS) {
         const lockoutUntil = Date.now() + LOCKOUT_DURATION_MS;
         await AsyncStorage.setItem(LOCKOUT_TIME_KEY, lockoutUntil.toString());
         setIsLockedOut(true);
         setLockoutRemainingMs(LOCKOUT_DURATION_MS);
       }
     };

     // NEW: Clear lockout on successful auth
     const clearLockout = async () => {
       setFailedAttempts(0);
       setIsLockedOut(false);
       setLockoutRemainingMs(0);
       await AsyncStorage.multiSet([
         [LOCKOUT_ATTEMPTS_KEY, '0'],
         [LOCKOUT_TIME_KEY, ''],
       ]);
     };

     return {
       // Existing returns...
       isBiometricAvailable,
       isBiometricEnabled,
       biometryType,
       isLoading,
       promptBiometric,
       enableBiometric,
       disableBiometric,
       biometricLabel,
       // NEW:
       isLockedOut,
       lockoutRemainingMs,
       failedAttempts,
       recordFailedAttempt,
       clearLockout,
     };
   }
   ```

   ```typescript
   import React, { useState, useRef } from 'react';
   import { View, TextInput, TouchableOpacity, Text } from 'react-native';
   import { twMerge } from 'tailwind-merge';

   interface PINFormProps {
     onSubmit: (pin: string) => void;
     isLoading?: boolean;
     error?: string;
     maxAttempts?: number;
     attemptsRemaining?: number;
     biometricLabel?: string;
     onBiometricPress?: () => void;
   }

   export function PINForm({
     onSubmit,
     isLoading = false,
     error = '',
     maxAttempts = 5,
     attemptsRemaining = maxAttempts,
     biometricLabel = 'Biometria',
     onBiometricPress,
   }: PINFormProps) {
     const [pin, setPin] = useState('');
     const inputRefs = useRef<(TextInput | null)[]>([]);

     const handleDigitChange = (index: number, value: string) => {
       // Only accept digits
       if (!/^\d*$/.test(value)) return;

       const newPin = pin.split('');
       newPin[index] = value.slice(-1); // Take only last digit if paste
       const newPinStr = newPin.join('').slice(0, 6);
       setPin(newPinStr);

       // Auto-focus next input
       if (value && index < 5) {
         inputRefs.current[index + 1]?.focus();
       }

       // Auto-submit if all 6 digits entered
       if (newPinStr.length === 6) {
         onSubmit(newPinStr);
       }
     };

     const handleBackspace = (index: number) => {
       if (index === 0 && pin.length === 0) return;

       const newPin = pin.split('');
       newPin[index] = '';
       const newPinStr = newPin.join('').slice(0, index);
       setPin(newPinStr);

       // Focus previous input if deleting
       if (index > 0) {
         inputRefs.current[index - 1]?.focus();
       }
     };

     return (
       <View className="flex-1 justify-center px-6">
         <Text className="text-center text-lg font-semibold text-white mb-8">
           Digite seu PIN de 6 dígitos
         </Text>

         {/* PIN Digit Inputs */}
         <View className="flex-row justify-between gap-2 mb-6">
           {[0, 1, 2, 3, 4, 5].map((index) => (
             <TextInput
               key={index}
               ref={(ref) => (inputRefs.current[index] = ref)}
               value={pin[index] ?? ''}
               onChangeText={(value) => handleDigitChange(index, value)}
               onKeyPress={({ nativeEvent: { key } }) => {
                 if (key === 'Backspace') handleBackspace(index);
               }}
               keyboardType="numeric"
               maxLength={1}
               secureTextEntry={false} // Masked via display trick below
               className={twMerge(
                 'w-12 h-12 text-center text-2xl font-bold border-2 rounded',
                 error ? 'border-red-500' : 'border-gray-600',
                 'dark:bg-gray-800 dark:text-white'
               )}
               editable={!isLoading}
             />
           ))}
         </View>

         {/* Error Message */}
         {error && (
           <Text className="text-center text-red-500 text-sm mb-4">{error}</Text>
         )}

         {/* Attempts Remaining */}
         {attemptsRemaining < maxAttempts && (
           <Text className="text-center text-yellow-500 text-sm mb-4">
             {attemptsRemaining} tentativa(s) restante(s)
           </Text>
         )}

         {/* Submit Button */}
         <TouchableOpacity
           disabled={pin.length !== 6 || isLoading}
           onPress={() => onSubmit(pin)}
           className={twMerge(
             'py-3 px-6 rounded-lg mb-4',
             pin.length === 6 && !isLoading
               ? 'dark:bg-blue-600'
               : 'dark:bg-gray-700'
           )}
         >
           <Text className="text-center text-white font-semibold">
             {isLoading ? 'Autenticando...' : 'Entrar'}
           </Text>
         </TouchableOpacity>

         {/* Biometric Fallback Button */}
         {onBiometricPress && (
           <TouchableOpacity
             onPress={onBiometricPress}
             className="py-2 px-6 rounded-lg border border-gray-600"
           >
             <Text className="text-center text-gray-400 text-sm">
               Usar {biometricLabel}
             </Text>
           </TouchableOpacity>
         )}
       </View>
     );
   }
   ```

   ```typescript
   import * as Keychain from 'react-native-keychain';

   const SERVICE_NAME = 'com.hcquality.mobile';
   const TOKEN_KEY = 'firebaseToken';
   const USER_ID_KEY = 'userId';

   export interface SecureStorageError {
     message: string;
     code: string;
   }

   /**
    * Save Firebase JWT to device Keychain (iOS) or Credential Manager (Android)
    */
   export async function saveFirebaseToken(token: string, userId: string): Promise<boolean> {
     try {
       const result = await Keychain.setGenericPassword(
         USER_ID_KEY,
         JSON.stringify({ token, userId, savedAt: Date.now() }),
         {
           service: SERVICE_NAME,
           accessible: Keychain.ACCESSIBLE.WHEN_UNLOCKED, // iOS only; requires device unlock
           storage: Keychain.STORAGE_TYPE.KC, // iOS Keychain (default)
         },
       );
       console.log('[SecureStorage] Token saved:', result);
       return result;
     } catch (error: any) {
       console.error('[SecureStorage] saveFirebaseToken failed:', error);
       return false;
     }
   }

   /**
    * Retrieve Firebase JWT from Keychain
    * Returns null if not found or error
    */
   export async function retrieveFirebaseToken(): Promise<{
     token: string;
     userId: string;
   } | null> {
     try {
       const credentials = await Keychain.getGenericPassword({
         service: SERVICE_NAME,
         storage: Keychain.STORAGE_TYPE.KC,
       });

       if (!credentials) {
         console.log('[SecureStorage] No credentials found');
         return null;
       }

       const parsed = JSON.parse(credentials.password);
       console.log('[SecureStorage] Token retrieved:', parsed);
       return parsed;
     } catch (error: any) {
       console.error('[SecureStorage] retrieveFirebaseToken failed:', error);
       return null;
     }
   }

   /**
    * Clear all credentials (logout)
    */
   export async function clearAll(): Promise<boolean> {
     try {
       const result = await Keychain.resetGenericPassword({
         service: SERVICE_NAME,
       });
       console.log('[SecureStorage] All credentials cleared:', result);
       return result;
     } catch (error: any) {
       console.error('[SecureStorage] clearAll failed:', error);
       return false;
     }
   }

   /**
    * Check if token exists (without retrieving it)
    */
   export async function hasCredentials(): Promise<boolean> {
     try {
       const credentials = await Keychain.canImplyCredentialsForService(SERVICE_NAME);
       return credentials;
     } catch (error) {
       return false;
     }
   }
   ```
