```typescript
/**
 * Calculate compliance percentage: (valid runs / total runs) * 100
 * Mirrors Cloud Function formula
 */
export function calculateCompliance(
  totalRunsValid: number,
  totalRuns: number
): number {
  if (totalRuns === 0) return 0;
  return (totalRunsValid / totalRuns) * 100;
}

/**
 * Classify compliance status by threshold
 */
export function getComplianceStatus(compliancePercent: number): {
  trend: string;
  accent: 'emerald' | 'amber' | 'red';
} {
  if (compliancePercent >= 95) {
    return { trend: '↑ Excelente', accent: 'emerald' };
  } else if (compliancePercent >= 85) {
    return { trend: '→ Satisfatório', accent: 'amber' };
  } else {
    return { trend: '↓ Crítico', accent: 'red' };
  }
}
```

```typescript
// src/features/analytics/hooks/useAnalyticsCache.ts
import { create } from 'zustand';
import { AnalyticsAggregate, AnalyticsMeta } from '../types/Analytics';

interface AnalyticsState {
  // Latest aggregate data
  latestAggregate: AnalyticsAggregate | null;
  
  // Metadata (lightweight, ~1KB)
  latestAggregateId: string | null;
  freshness: 'current' | 'stale' | 'unknown';
  nextUpdateAt: Date | null;
  lastFetchedAt: number | null;
  
  // UI state
  isLoading: boolean;
  error: string | null;
  
  // Actions
  setAggregate: (data: AnalyticsAggregate) => void;
  setMetadata: (meta: AnalyticsMeta) => void;
  setLoading: (loading: boolean) => void;
  setError: (error: string | null) => void;
  clear: () => void;
}

export const useAnalyticsStore = create<AnalyticsState>((set) => ({
  latestAggregate: null,
  latestAggregateId: null,
  freshness: 'unknown',
  nextUpdateAt: null,
  lastFetchedAt: null,
  isLoading: false,
  error: null,
  
  setAggregate: (data) => set({
    latestAggregate: data,
    lastFetchedAt: Date.now(),
    error: null,
  }),
  
  setMetadata: (meta) => set({
    latestAggregateId: meta.latestAggregateId,
    freshness: meta.freshness,
    nextUpdateAt: meta.nextUpdateAt ? new Date(meta.nextUpdateAt) : null,
  }),
  
  setLoading: (loading) => set({ isLoading: loading }),
  setError: (error) => set({ error }),
  
  clear: () => set({
    latestAggregate: null,
    latestAggregateId: null,
    freshness: 'unknown',
    nextUpdateAt: null,
    lastFetchedAt: null,
    error: null,
  }),
}));
```

**Create `src/features/analytics/hooks/useAnalyticsAggregates.ts`:**
Subscribe to metadata, auto-fetch aggregate on update.

```typescript
// src/features/analytics/hooks/useAnalyticsAggregates.ts
import { useEffect } from 'react';
import {
  collection,
  onSnapshot,
  doc,
  getDoc,
  query,
  where,
  Firestore,
} from 'firebase/firestore';
import { db } from '../../../core/firebase';
import { useAuthStore } from '../../../shared/hooks/useAuthStore';
import { useAnalyticsStore } from './useAnalyticsCache';
import { AnalyticsAggregate } from '../types/Analytics';

export function useAnalyticsAggregates() {
  const labId = useAuthStore((s) => s.activeLabId);
  const store = useAnalyticsStore();

  useEffect(() => {
    if (!labId) {
      store.setLoading(false);
      return;
    }

    // Subscribe to lightweight metadata
    const metaRef = doc(db, `labs/${labId}/analytics-meta/current`);
    const unsubscribeMeta = onSnapshot(
      metaRef,
      async (metaSnap) => {
        if (!metaSnap.exists()) {
          store.setError('Analytics metadata not found. Aggregation runs hourly.');
          store.setLoading(false);
          return;
        }

        const meta = metaSnap.data();
        store.setMetadata(meta);

        // Auto-fetch full aggregate when metadata changes
        if (meta.latestAggregateId) {
          try {
            store.setLoading(true);
            const aggRef = doc(
              db,
              `labs/${labId}/analytics-agg/${meta.latestAggregateId}`
            );
            const aggSnap = await getDoc(aggRef);

            if (aggSnap.exists()) {
              const aggregateData = aggSnap.data() as AnalyticsAggregate;
              store.setAggregate(aggregateData);
              store.setError(null);
            } else {
              store.setError('Aggregate document not found');
            }
          } catch (err) {
            store.setError((err as Error).message);
          } finally {
            store.setLoading(false);
          }
        }
      },
      (err) => {
        store.setError((err as Error).message);
        store.setLoading(false);
      }
    );

    return unsubscribeMeta;
  }, [labId, store]);

  return {
    aggregate: store.latestAggregate,
    freshness: store.freshness,
    isLoading: store.isLoading,
    error: store.error,
    lastFetchedAt: store.lastFetchedAt,
  };
}
```

```typescript
// src/features/analytics/hooks/useChartData.ts
import { useMemo } from 'react';
import { AnalyticsAggregate, CIQTrendData } from '../types/Analytics';

export interface ChartData {
  day?: string;
  valid?: number;
  invalid?: number;
  compliance?: number;
  value?: number;
  min?: number;
  max?: number;
  module?: string;
  count?: number;
  age?: number;
}

/**
 * Transform CIQ trend data for Recharts ComposedChart
 */
export function useCIQChartData(aggregate: AnalyticsAggregate | null) {
  return useMemo(() => {
    if (!aggregate?.ciqTrendsByDay) return [];

    return aggregate.ciqTrendsByDay.map((trend) => ({
      day: trend.day,
      valid: trend.valid,
      invalid: trend.invalid,
      compliance: trend.valid + trend.invalid > 0
        ? ((trend.valid / (trend.valid + trend.invalid)) * 100).toFixed(1)
        : 0,
      avgValue: trend.avgValue,
    }));
  }, [aggregate]);
}

/**
 * Transform NC trend data for heatmap grid
 */
export function useNCHeatmapData(aggregate: AnalyticsAggregate | null) {
  return useMemo(() => {
    if (!aggregate?.ncTrendsByModule) return [];

    return aggregate.ncTrendsByModule.map((nc) => ({
      module: nc.moduleId,
      count: nc.openCount,
      age: Math.round(nc.ageAverageDays),
      color: nc.ageAverageDays > 30 ? '#ef4444' : nc.ageAverageDays > 7 ? '#f59e0b' : '#10b981',
    }));
  }, [aggregate]);
}

/**
 * Transform training expirations for table
 */
export function useTrainingTableData(aggregate: AnalyticsAggregate | null) {
  return useMemo(() => {
    if (!aggregate?.trainingExpirations) return [];

    return aggregate.trainingExpirations.map((training) => ({
      operatorId: training.operatorId,
      module: training.moduleId,
      expiresAt: new Date(training.expiresAt).toLocaleDateString('pt-BR'),
      daysRemaining: training.daysRemaining,
      status: training.status,
      statusColor: training.status === 'current'
        ? '#10b981'
        : training.status === 'expiring_soon'
        ? '#f59e0b'
        : '#ef4444',
    }));
  }, [aggregate]);
}
```

```typescript
import React, { useState } from 'react';
import { httpsCallable } from 'firebase/functions';
import { functions } from '../../../core/firebase';
import { useAuthStore } from '../../../shared/hooks/useAuthStore';

export function RefreshButton() {
  const labId = useAuthStore((s) => s.activeLabId);
  const [loading, setLoading] = useState(false);

  const handleRefresh = async () => {
    if (!labId) return;
    
    setLoading(true);
    try {
      const fn = httpsCallable(functions, 'triggerAnalyticsRefresh');
      await fn({ labId });
      // Store will auto-update via onSnapshot
    } catch (err) {
      console.error('Refresh failed:', err);
    } finally {
      setLoading(false);
    }
  };

  return (
    <button
      onClick={handleRefresh}
      disabled={loading}
      className="px-3 h-9 rounded-lg text-sm font-medium bg-blue-600 text-white hover:bg-blue-700 disabled:opacity-50 disabled:cursor-not-allowed transition-colors"
    >
      {loading ? '⟳ Atualizando...' : 'Atualizar Agora'}

  );
}
```
