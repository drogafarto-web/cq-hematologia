interface PortalNavbarProps {
  labName: string;
  logoUrl: string;
  patientFirstName: string;

}

export const PortalNavbar: React.FC<PortalNavbarProps> = ({
  labName, logoUrl, patientFirstName, onLogout
}) => {

  return (
    <header className="sticky top-0 z-50 bg-[#141417] border-b border-white/10">
      <nav className="flex items-center justify-between px-4 py-3 sm:px-6">
        {/* Logo + Lab Name */}

            className="h-12 w-auto object-contain"
            loading="eager"
          />
          <span className="hidden sm:inline text-sm text-white/60">{labName}</span>
        </div>
        
        {/* Desktop: Greeting + Logout */}
        <div className="hidden md:flex items-center gap-4">
          <p className="text-sm text-white/90">Bem-vindo, {patientFirstName}</p>
          <button
            onClick={onLogout}
            className="px-4 py-2 text-sm font-medium rounded-md bg-white/10 hover:bg-white/20 transition-colors"
          >
            Sair
          </button>
        </div>
        
        {/* Mobile: Menu toggle */}
        <div className="md:hidden flex items-center gap-2">
          <button
            onClick={() => setMenuOpen(!menuOpen)}
            aria-label="Menu"
            className="p-2 rounded-md hover:bg-white/10"
          >
            {menuOpen ? <XIcon /> : <MenuIcon />}
          </button>
          <button
            onClick={onLogout}
            aria-label="Sair"
            className="p-2 rounded-md hover:bg-white/10"

            <LogoutIcon />
          </button>
        </div>
      </nav>

      {menuOpen && (
        <div className="md:hidden border-t border-white/10 bg-[#1a1a1d] px-4 py-3 space-y-2">
          <p className="text-sm text-white/90">Bem-vindo, {patientFirstName}</p>
          {/* Settings link goes here */}
        </div>
      )}
    </header>
  );
};

interface LaudoCardProps {

export const LaudoCard: React.FC<LaudoCardProps> = ({

### 3. Responsive Breakpoints & Mobile-First

**Tailwind breakpoints (existing, applied consistently):**
- `mobile: 0px` (default)
- `sm: 640px` (tablet portrait)
- `md: 768px` (tablet landscape)
- `lg: 1024px` (desktop)
- `xl: 1280px` (large desktop)

**Portal-specific overrides:**
- LaudoCard: 1 column on mobile, 2 columns on `md:`, 3 columns on `lg:`
- Detail page: full-width mobile, max-w-4xl on desktop
- Navbar: hamburger menu on mobile, full nav on `md:`
- Results table: horizontal scroll on mobile (not truncated), sticky header

### 4. Dark Theme & Accessibility

**Color contrast verification:**
- Text on dark bg: white text on #141417 = 18:1 (exceeds 4.5:1 WCAG AAA) ✓
- Accent buttons: use `primary` color with sufficient contrast
- Links: always underlined (not color-only)
- Focus rings: `outline 2px solid var(--portal-primary)` with 2px offset

**Dark mode best practices:**
- No pure white (#FFFFFF) backgrounds (causes eye strain)
- Use `#141417` for primary dark, `#1a1a1d` for elevated surfaces
- Shadows: use alpha channel (`rgba(0,0,0,0.2)`) on dark backgrounds
- Avoid overly saturated accent colors; dim on dark (`/80` or lower alpha)

**WCAG AA checklist:**
- [ ] All buttons have visible focus state (outline ring)
- [ ] All images have `alt` text (logo alt: lab name)
- [ ] Form inputs have associated `<label>` (if any login form)
- [ ] Keyboard navigation works: Tab through navbar → content → footer
- [ ] Touch target size: min 44×44px (buttons, links)
- [ ] Motion: respect `prefers-reduced-motion` (no auto-play animations)

### 5. Performance Optimizations

**Code splitting:**
- Portal routes lazy-loaded: `React.lazy(() => import('./pages/LaudoDetailPage'))`
- Modal for Terms/Privacy: lazy-load HTML on mount

**Image optimization:**
- Logo: `loading="eager"` (above fold), `width`/`height` explicit
- Laudo images (if any): `loading="lazy"`, thumbnail + link to full size

**Bundle impact:**
- No new heavy deps (reuse existing React, Tailwind)
- Portal feature module ~50 KB (components + hooks + pages)
- Expected main chunk impact: +8–12 KB gzip (acceptable)

**Rendering optimizations:**
- `React.memo` on LaudoCard (repeated in list)
- `useCallback` for click handlers passed to children
- `useMemo` for filtered/sorted laudo list (if >50 items)
- Firestore listener cleanup in `useEffect` return

**Web Vitals baseline:**
- LCP: <2.0s (portal loads in <1.5s baseline, <2.5s with network delay)
- INP: <200ms (button clicks, form submissions)
- CLS: <0.05 (no layout shift on laudo cards, skeleton prevents jank)
