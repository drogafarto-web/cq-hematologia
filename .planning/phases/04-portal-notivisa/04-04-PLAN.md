**Rules implementation pattern (must exist in firestore.rules):**

```firestore
rules_version = '2';
service cloud.firestore {
  match /labs/{labId} {
    // Patient portal laudo read isolation
    match /laudos/{laudoId} {
      allow read: if isPortalPatient() && matchesCPF(resource.data.pacienteCpf);
    }

    // Patient sessions (server-side only)
    match /patientSessions/{sessionId} {
      allow read, write: if false;
    }

    // Portal config (public read)
    match /portal-configuracao/{doc=**} {
      allow read: if true;
      allow write: if hasRole('admin');
    }

    // NOTIVISA queue (immutable)
    match /notivisa-outbox/{eventId} {
      allow read: if hasRole('admin', 'rt', 'quality-manager');
      allow write: if false;
    }

    // Portal access logs (audit-only, immutable)
    match /portal-access-logs/{logId} {
      allow read: if hasRole('admin', 'auditor');
      allow write: if false;
    }
  }

  function isPortalPatient() {
    return request.auth != null &&
           request.auth.token.role == 'portal-patient';
  }

  function matchesCPF(pacienteCpf) {
    return request.auth.token.cpfHash == hashCPF(pacienteCpf);
  }

  function hasRole(roles...) {
    return request.auth != null &&
           request.auth.token.role in roles;
  }

  function hashCPF(cpf) {
    return crypto.sha256(cpf); // Matches backend hash (HMAC-SHA256)
  }
}
```

```bash
#!/bin/bash
# Phase 4 Cloud Logs monitoring — run every 30min for 24h post-deploy

set -e

PROJECT="hmatologia2"
HOURS_AGO=${1:-24}  # Default: last 24h
POLL_INTERVAL=${2:-30}  # Default: poll every 30min

echo "=== Phase 4 Cloud Logs Monitor ==="
echo "Monitoring last $HOURS_AGO hours"
echo "Started: $(date)"
echo ""

# Define log filter for Phase 4 functions
LOG_FILTER='
  resource.type="cloud_function" AND
  (
    function_name=~"(validatePatientToken|downloadLaudoPDF|processNotiVisaQueue|.*Portal.*)" OR
    text_payload=~"(patient-auth|patient-pdf|notivisa)"
  ) AND
  timestamp>="'$(date -u -d "$HOURS_AGO hours ago" '+%Y-%m-%dT%H:%M:%SZ')'"
'

# 1. Count ERROR/CRITICAL messages
echo "ERROR/CRITICAL count (last $HOURS_AGO hours):"
ERROR_COUNT=$(gcloud logging read "$LOG_FILTER AND severity>=ERROR" \
  --limit=1000 --format=json --project=$PROJECT | jq 'length')
echo "  Total: $ERROR_COUNT"
if [ "$ERROR_COUNT" -gt 0 ]; then
  echo "  WARNING: Found errors. Details below."
fi
echo ""

# 2. Show last 10 ERROR/CRITICAL
echo "Last 10 ERROR/CRITICAL messages:"
gcloud logging read "$LOG_FILTER AND severity>=ERROR" \
  --limit=10 --format="table(timestamp, severity, labels.function_name, textPayload)" \
  --project=$PROJECT | head -20
echo ""

# 3. Function call distribution
echo "Function call distribution (all severities):"
gcloud logging read "$LOG_FILTER" \
  --limit=1000 --format=json --project=$PROJECT | \
  jq -r '.[].labels.function_name // "unknown"' | \
  sort | uniq -c | sort -rn
echo ""

# 4. Per-function summary
for FUNC in validatePatientToken downloadLaudoPDF processNotiVisaQueue; do
  COUNT=$(gcloud logging read "$LOG_FILTER AND labels.function_name=\"$FUNC\"" \
    --limit=1000 --format=json --project=$PROJECT | jq 'length')
  ERROR=$(gcloud logging read "$LOG_FILTER AND labels.function_name=\"$FUNC\" AND severity>=ERROR" \
    --limit=1000 --format=json --project=$PROJECT | jq 'length')
  echo "$FUNC: $COUNT calls, $ERROR errors"
done
echo ""

# 5. Failure rate (NOTIVISA queue specific)
NOTIVISA_TOTAL=$(gcloud logging read "$LOG_FILTER AND text_payload=~\"notivisa\"" \
  --limit=1000 --format=json --project=$PROJECT | jq 'length')
NOTIVISA_FAILED=$(gcloud logging read "$LOG_FILTER AND text_payload=~\"notivisa.*failed\"" \
  --limit=1000 --format=json --project=$PROJECT | jq 'length')
if [ "$NOTIVISA_TOTAL" -gt 0 ]; then
  FAILURE_RATE=$((NOTIVISA_FAILED * 100 / NOTIVISA_TOTAL))
  echo "NOTIVISA queue — Failure rate: $FAILURE_RATE% ($NOTIVISA_FAILED / $NOTIVISA_TOTAL)"
  if [ "$FAILURE_RATE" -gt 10 ]; then
    echo "  ALERT: Failure rate >10%. Investigate."
  fi
fi
echo ""

# 6. Export to JSON for analysis
EXPORT_FILE="phase4-logs-$(date +%Y%m%d-%H%M%S).json"
gcloud logging read "$LOG_FILTER" \
  --limit=1000 --format=json --project=$PROJECT > "$EXPORT_FILE"
echo "Full log export: $EXPORT_FILE"
echo ""

echo "Report generated: $(date)"
```

## Trust Boundaries

| Boundary                           | Description                                                                                                 | Direction                                                                   |
| ---------------------------------- | ----------------------------------------------------------------------------------------------------------- | --------------------------------------------------------------------------- |
| **Patient ↔ Email System**         | Patient receives email link, clicks link, lands on portal. Email can be forwarded (security risk).          | Untrusted: email is medium, not authentication                              |
| **Patient ↔ Portal (Client)**      | Patient provides browser context (user-agent, IP, cookies). Portal stores JWT in localStorage.              | Untrusted: client-side storage vulnerable to XSS                            |
| **Portal (Client) ↔ Backend APIs** | Client sends JWT, calls validatePatientToken / downloadLaudoPDF / fetchLaudoList callables.                 | Untrusted: client can modify request (CPF param tampering attempt)          |
| **Backend ↔ Firestore**            | Cloud Functions read/write Firestore (rules-enforced access). Rules are source of truth for access control. | Trusted: Rules immutable, enforced at database layer                        |
| **Backend ↔ NOTIVISA Sandbox API** | Cloud Function submits laudo data to NOTIVISA sandbox endpoint. API key stored in Cloud Secrets Manager.    | Untrusted: gov API may be compromised; submissions logged locally for audit |
| **Portal ↔ Cloud Logging**         | Events logged to Cloud Logging (Cloud Functions write, GCP manages storage).                                | Trusted: GCP Cloud Logging with Cloud Audit Logs sink prevents tampering    |

## STRIDE Threat Register

| Threat ID   | Category               | Component                                                | Disposition | Mitigation Plan                                                                                                                                                                                                                                          |
| ----------- | ---------------------- | -------------------------------------------------------- | ----------- | -------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |

| **T-04-03** | Tampering              | Laudo data modified before NOTIVISA submission           | mitigate    | Laudo marked read-only once published (status='released'). | **T-04-04** | Repudiation            | Patient denies receiving laudo notification              | mitigate    | Email audit trail: log every link generation (who, when, recipient email). | **T-04-05** | Information Disclosure | NOTIVISA API key exposed in code/logs                    | mitigate    | API key stored in Cloud Secret Manager (not in git, not in environment vars). | **T-04-07** | Denial of Service      | NOTIVISA queue flooded (spam laudo publications)         | accept      | Queue processes max 10 events/hour (slow cron + batch limit).
