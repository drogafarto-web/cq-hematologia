**3. patientSessions** — ephemeral session records (JWT-based):

```firestore-rules
match /labs/{labId}/patientSessions/{sessionId} {
  allow create: if
    request.auth == null &&
    request.resource.data.tokenExpiry > now &&
    request.resource.data.cpfHash.size() == 64 &&
    request.resource.data.jwtToken.size() > 0;
  allow read, delete: if
    request.auth != null &&
    request.auth.uid == resource.data.operatorId;
}
```

**Helper function** in rules block (add to existing helper section):

```firestore-rules
function isPatientWithValidToken(labId) {
  // Check if caller has valid JWT in patientSessions
  // Used for patient-portal reads only
  return get(/databases/$(database)/documents/labs/$(labId)/patientSessions/$(request.auth.uid)).data.expiresAt > now;
}
```

```typescript
export interface PatientSession {
  cpf: string; // Patient CPF (plaintext in session, for server filtering)
  firstName: string;
  jwtToken: string; // Signed with HCQ_SESSION_SECRET
  expiresAt: number; // Unix timestamp (30 days from now)
  labId: string;
  createdAt: number;
}

export function usePatientAuth(labId: string) {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  // Validate one-time email link token
  async function validateToken(token: string): Promise<PatientSession | null> {
    setLoading(true);
    setError(null);
    try {
      const validatePatientTokenCallable = httpsCallable(getFirestore(), 'validatePatientToken');
      const result = await validatePatientTokenCallable({
        token,
        labId,
      });
      const session: PatientSession = result.data.session;
      localStorage.setItem(`patientSession_${labId}`, JSON.stringify(session));
      return session;
    } catch (err: any) {
      const message =
        err.code === 'permission-denied'
          ? 'Link expired or already used'
          : err.message || 'Authentication failed';
      setError(message);
      return null;
    } finally {
      setLoading(false);
    }
  }

  // Get current session from localStorage
  function getSession(): PatientSession | null {
    const stored = localStorage.getItem(`patientSession_${labId}`);
    if (!stored) return null;
    const session: PatientSession = JSON.parse(stored);
    if (session.expiresAt < Date.now()) {
      logout();
      return null;
    }
    return session;
  }

  // Clear session
  function logout() {
    localStorage.removeItem(`patientSession_${labId}`);
    setError(null);
  }

  return { validateToken, getSession, logout, loading, error };
}
```

```typescript
export async function getPortalConfig(labId: string): Promise<PortalConfig | null> {
  const docRef = doc(getFirestore(), `labs/${labId}/portal-configuracao`, 'config');
  const snap = await getDoc(docRef);
  return snap.exists() ? (snap.data() as PortalConfig) : null;
}

export function subscribeToLaudos(
  labId: string,
  cpf: string,
  callback: (laudos: Laudo[]) => void,
): Unsubscribe {
  const q = query(
    collection(getFirestore(), `labs/${labId}/laudos`),
    where('paciente.cpf', '==', cpf),
    where('status', '==', 'published'),
    where('visibilidadePaciente', '==', true),
    orderBy('dataPublicacao', 'desc'),
  );
  return onSnapshot(q, (snap) => {
    const laudos = snap.docs.map((doc) => ({
      id: doc.id,
      ...doc.data(),
    })) as Laudo[];
    callback(laudos);
  });
}

export async function getSingleLaudo(labId: string, laudoId: string): Promise<Laudo | null> {
  const docRef = doc(getFirestore(), `labs/${labId}/laudos`, laudoId);
  const snap = await getDoc(docRef);
  return snap.exists() ? ({ id: snap.id, ...snap.data() } as Laudo) : null;
}
```

**Function 1: validatePatientToken**

- Input: { token: string, labId: string }
- Lookup token in `laudos[labId]` collection (field `patientAccessTokens[]` array)
- Validate: token exists, expiry > now, `usedAt` is null (single-use)
- Generate JWT: `{ cpf, firstName, labId, expiresAt: Date.now() + 30d }`
- Sign with `HCQ_SESSION_SECRET` (from env)
- Mark token as used: set `patientAccessTokens[i].usedAt = now`
- Create entry in `patientSessions/{sessionId}`: { cpf, firstName, expiresAt, operatorId: 'patient-system', ... }
- Log to audit trail: `{ action: 'patient-auth', labId, tokenHash: sha256(token), cpf: sha256(cpf), success: true, timestamp }`
- Return: { session: { cpf (plaintext), firstName, jwtToken, expiresAt, labId }, success: true }
- Error cases:
  - token not found → 404 "Link not found"
  - token expired → 403 "Link expired"
  - token already used → 400 "Link already consumed"

**Function 2: downloadLaudoPDF**

- Input: { laudoId: string, labId: string, sessionToken: string }
- Decode JWT from sessionToken (verify signature with HCQ_SESSION_SECRET)
- Fetch laudo doc from `laudos/{labId}/{laudoId}`
- Validate: status === 'published' AND visibilidadePaciente === true AND paciente.cpf === decoded.cpf
- Generate PDF (reuse existing Puppeteer cloud function from Phase 3)
- Upload to Storage: `gs://hmatologia2.appspot.com/patients/{labId}/{laudoId}/laudo.pdf`
- Create signed URL (expires in 1 hour)
- Log: `{ action: 'patient-pdf-download', laudoId, labId, cpfHash: sha256(cpf), timestamp }`
- Return: { pdfUrl: 'https://...' }
- Error cases: CPF mismatch → 403, laudo not published → 404, decode JWT fail → 401

**Function 3: updatePatientPreferences**

<threat_model>

## Trust Boundaries

| Boundary                | Description                                                                                                                                 |
| ----------------------- | ------------------------------------------------------------------------------------------------------------------------------------------- |
| **Email → Token**       | Untrusted email link (patient's inbox compromised?) — mitigated by token expiry (7d) + single-use + hash validation                         |
| **Client → API**        | Untrusted JWT in localStorage (XSS attack?) — mitigated by JWT signature validation (server-side), httpOnly flag not needed since stateless |
| **Patient → Own Laudo** | Patient claims access to laudo with wrong CPF — mitigated by server-side CPF filter in callable + Rules validation                          |
| **Portal → Firestore**  | Untrusted rules logic (rules bypass?) — mitigated by `isPatientWithValidToken()` helper checking JWT in patientSessions                     |
| **Laudo Publication**   | Lab staff marks laudo `visibilidadePaciente=false` to hide from patient — valid, read-only on patient side, no risk                         |

## STRIDE Threat Register

| Threat ID | Category               | Component                            | Disposition  | Mitigation Plan                                                                                                                          |
| --------- | ---------------------- | ------------------------------------ | ------------ | ---------------------------------------------------------------------------------------------------------------------------------------- |
| T-04-01   | Spoofing               | Email link token                     | **Mitigate** | UUID v4 (2^128 uniqueness) + HMAC-SHA256 hash validation + rate-limit per IP (10 attempts/min on validate endpoint)                      |
| T-04-02   | Spoofing               | JWT session                          | **Mitigate** | Token signed with HCQ_SESSION_SECRET (separate from HMAC key), 30d expiry, server-side verify on every read                              |
| T-04-03   | Tampering              | patientAccessTokens array            | **Mitigate** | Append-only in Firestore (no client edits), marked used immediately after validation, batch write ensures atomicity                      |
| T-04-04   | Tampering              | JWT payload                          | **Mitigate** | Signed token cannot be modified; decode includes HMAC verification                                                                       |
| T-04-05   | Repudiation            | Patient PDF download                 | **Mitigate** | All downloads logged: { action: 'patient-pdf-download', laudoId, cpfHash, timestamp } in audit collection                                |
| T-04-06   | Information Disclosure | CPF in transit                       | **Mitigate** | CPF returned as plaintext in validateToken response (needed for server filtering), but only to authorized patient; hashed in audit trail |
| T-04-07   | Information Disclosure | Cross-patient laudo leak             | **Mitigate** | Server-side CPF filter in callable + Rules `where paciente.cpf == session.cpf`, code review of all patient reads                         |
| T-04-08   | Information Disclosure | Session hijacking (localStorage XSS) | **Mitigate** | CSP header `script-src 'self'`, DOMPurify on user input, no sensitive data in HTML attributes                                            |
| T-04-09   | Denial of Service      | Token generation (caller spam)       | **Mitigate** | Email send rate-limit (Phase 6 notifications), per-IP token validation limit (10/min), max 5 active tokens per laudo                     |
| T-04-10   | Denial of Service      | PDF generation (cold start)          | **Mitigate** | Cache PDF for 7 days (avoid regenerating), monitor Cloud Logs for Puppeteer cold starts >30s, alert on >5 per hour                       |
