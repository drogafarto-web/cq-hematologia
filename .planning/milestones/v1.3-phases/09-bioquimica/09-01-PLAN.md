### Task 8: Firestore rules + indexes (1 dia)

- Em `firestore.rules` adicionar bloco:

  ```
  match /labs/{labId}/bioquimica/{document=**} {
    function isValidAnalito(d) {
      return d.labId == labId
        && d.nome is string && d.nome.size() > 0
        && d.unidade is string && d.unidade.size() > 0
        && d.rangeBiologico.min is number
        && d.rangeBiologico.max is number;
    }
    function isValidLot(d) {
      return d.labId == labId
        && d.equipmentIds is list && d.equipmentIds.size() > 0
        && d.lote is string
        && d.validade is timestamp;
    }
    function isValidRun(d) {
      return d.labId == labId
        && d.equipmentId is string
        && d.lotId is string
        && d.resultados is map
        && validSignature(d);
    }

    match /config/{singleton} {
      allow read: if isActiveMemberOfLab(labId);
      allow write: if isAdminOrOwner(labId);
    }
    match /analitos/{analitoId} {
      allow read: if isActiveMemberOfLab(labId);
      allow create: if isActiveMemberOfLab(labId) && isValidAnalito(request.resource.data);
      allow update: if isActiveMemberOfLab(labId) && keepsLabId() && keepsCreatedAt();
      allow delete: if false;
    }
    match /lotes/{lotId} {
      allow read: if isActiveMemberOfLab(labId);
      allow create: if isActiveMemberOfLab(labId) && isValidLot(request.resource.data);
      allow update: if isActiveMemberOfLab(labId) && keepsLabId() && keepsCreatedAt();
      allow delete: if false;
    }
    match /runs/{runId} {
      allow read: if isActiveMemberOfLab(labId);
      allow create: if false;  // só via Cloud Function (Plan 09-04)
      allow update: if false;
      allow delete: if false;
    }
    match /traceability-events/{eventId} {
      allow read: if isActiveMemberOfLab(labId);
      allow create: if false;  // só via Cloud Function
      allow update: if false;
      allow delete: if false;
    }
    match /audit/{logId} {
      allow read: if isActiveMemberOfLab(labId);
      allow create, update, delete: if false;
    }
  }
  ```
