### Task 1: westgardRulesCLSI.ts — engine puro (2 dias)

- Criar `src/features/bioquimica/utils/westgardRulesCLSI.ts`:

  ```typescript
  export interface WestgardCheckInput {
    current: {
      value: number;
      analitoId: string;
      nivelId: string;
      equipmentId: string;
      capturaEm: Timestamp;
    };
    history: Array<{ value: number; capturaEm: Timestamp }>; // ordem cronológica DESC, mesmo (analito, nivel, equipamento)
    stats: { mean: number; sd: number };
  }

  export function checkWestgardCLSI(input: WestgardCheckInput): WestgardViolation[] {
    const violations: WestgardViolation[] = [];
    const z = (input.current.value - input.stats.mean) / input.stats.sd;

    // 1-2s — warn (1 ponto fora 2σ)
    if (Math.abs(z) > 2 && Math.abs(z) <= 3) {
      violations.push({
        rule: '1-2s',
        severity: 'warn',
        detail: `Valor a ${z.toFixed(2)}σ` /* ... */,
      });
    }
    // 1-3s — reject (1 ponto fora 3σ)
    if (Math.abs(z) > 3) {
      violations.push({
        rule: '1-3s',
        severity: 'reject',
        detail: `Valor a ${z.toFixed(2)}σ — fora 3σ` /* ... */,
      });
    }
    // 2-2s — reject (2 runs consecutivas mesmo lado fora 2σ)
    if (input.history[0]) {
      const zPrev = (input.history[0].value - input.stats.mean) / input.stats.sd;
      if (Math.abs(z) > 2 && Math.abs(zPrev) > 2 && Math.sign(z) === Math.sign(zPrev)) {
        violations.push({
          rule: '2-2s',
          severity: 'reject',
          detail: `2 runs consecutivos a ${z.toFixed(2)}σ e ${zPrev.toFixed(2)}σ` /* ... */,
        });
      }
    }
    // R-4s — reject (range 4σ entre 2 runs consecutivos)
    if (input.history[0]) {
      const zPrev = (input.history[0].value - input.stats.mean) / input.stats.sd;
      if (Math.abs(z - zPrev) > 4) {
        violations.push({
          rule: 'R-4s',
          severity: 'reject',
          detail: `Range 4σ entre runs (${z.toFixed(2)}σ e ${zPrev.toFixed(2)}σ)` /* ... */,
        });
      }
    }
    return violations;
  }

  export function suggestStatus(violations: WestgardViolation[]): 'Aprovada' | 'Rejeitada' {
    return violations.some((v) => v.severity === 'reject') ? 'Rejeitada' : 'Aprovada';
  }
  ```

- Edge cases tratados:
  - `sd === 0` → retornar `[]` (não calcular z) e logar warning
  - `history.length === 0` → só roda 1-2s e 1-3s (não 2-2s nem R-4s)
  - Stats inválidas (NaN, undefined) → retornar `[]` + warning
- Constantes exportadas para Plan 04 server-side (mesmo arquivo, importável de functions/src se simbol-link ou copy)

**Acceptance:** Cobertura unit ≥98% (todas as 4 regras + 6 edge cases).
