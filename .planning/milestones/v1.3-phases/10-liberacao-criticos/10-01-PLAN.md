### Task 2: State machine engine (1 dia)

- `src/features/liberacao/utils/stateMachine.ts`:

  ```typescript
  type Transition = {
    from: ReleaseState;
    to: ReleaseState;
    allowedRoles: ('RT' | 'RT-Substituto' | 'Sistema')[];
  };

  const TRANSITIONS: Transition[] = [
    { from: 'Pendente', to: 'Em Revisão', allowedRoles: ['RT', 'RT-Substituto'] },
    { from: 'Pendente', to: 'Auto-Liberado', allowedRoles: ['Sistema'] },
    { from: 'Em Revisão', to: 'Liberado', allowedRoles: ['RT', 'RT-Substituto'] },
    { from: 'Em Revisão', to: 'Pendente', allowedRoles: ['RT', 'RT-Substituto'] }, // unlock
    { from: 'Liberado', to: 'Comunicado', allowedRoles: ['RT', 'RT-Substituto', 'Sistema'] },
    { from: 'Auto-Liberado', to: 'Comunicado', allowedRoles: ['RT', 'RT-Substituto', 'Sistema'] },
    { from: 'Liberado', to: 'Superado', allowedRoles: ['RT', 'RT-Substituto'] }, // retificação
    { from: 'Auto-Liberado', to: 'Superado', allowedRoles: ['RT', 'RT-Substituto'] },
    { from: 'Comunicado', to: 'Superado', allowedRoles: ['RT', 'RT-Substituto'] },
  ];

  export function validateTransition(
    from: ReleaseState,
    to: ReleaseState,
    role: string,
    ctx?: { hasComunicado: boolean },
  ): { allowed: boolean; reason?: string } {
    const t = TRANSITIONS.find((x) => x.from === from && x.to === to);
    if (!t) return { allowed: false, reason: `Transição ${from} → ${to} não permitida` };
    if (!t.allowedRoles.includes(role as any))
      return { allowed: false, reason: `Role ${role} não autorizado` };
    if (to === 'Superado' && from !== 'Comunicado' && ctx?.hasComunicado) {
      return {
        allowed: false,
        reason: 'Laudo comunicado precisa transição via Comunicado→Superado',
      };
    }
    return { allowed: true };
  }

  export function nextStates(from: ReleaseState, role: string): ReleaseState[] {
    return TRANSITIONS.filter((t) => t.from === from && t.allowedRoles.includes(role as any)).map(
      (t) => t.to,
    );
  }
  ```

### Task 3: ExameClassifier (0.5 dia)

- `src/features/liberacao/utils/exameClassifier.ts`:

  ```typescript
  export function classifyExame(config: ExameConfig): ExamClassification {
    return config.classification;
  }

  export function shouldAutoRelease(
    laudo: Laudo,
    config: ExameConfig,
    context: { hasWestgardReject: boolean; hasCritico: boolean; hasMaterialRestrito: boolean },
  ): { autoRelease: boolean; reason: string } {
    if (!config.autoReleaseEnabled)
      return { autoRelease: false, reason: 'Auto-release desativado pelo lab' };
    if (config.classification === 'revisao-rt')
      return { autoRelease: false, reason: 'Exame requer revisão de RT' };
    if (config.classification === 'bloqueio-critico' && context.hasCritico)
      return { autoRelease: false, reason: 'Valor crítico bloqueia auto-liberação' };
    if (context.hasWestgardReject)
      return { autoRelease: false, reason: 'Westgard rejection presente' };
    if (context.hasMaterialRestrito)
      return { autoRelease: false, reason: 'Amostra com restrição requer revisão RT' };
    return { autoRelease: true, reason: 'Critérios de auto-liberação atendidos' };
  }
  ```

**Acceptance:** Coverage unit 100%; tests cobrem 8 cenários.
